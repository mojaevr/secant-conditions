\chapter{Симметричные квазиньютоновские обновления с контролем секущих условий}\label{ch:symmetric}

\section{Мотивация}\label{sec:ss_motivation}

В предыдущих разделах мультисекущий Бройден решил задачу сохранения
секущих условий для общих систем $F(x) = 0$, однако матрица $B_k$ при
этом несимметрична. Для задач безусловной оптимизации,
где $F = \nabla f$ и $J^* = \nabla^2 f(x^*)$ симметричен, естественно
требовать $B_k = B_k^\top$.

Классические симметричные квазиньютоновские обновления~---
SR1 (\emph{symmetric rank-one}, Дэвидон~\cite{davidon1991}) и~PSB
(\emph{Powell symmetric Broyden}, Пауэлл~\cite{powell1970})~---
удовлетворяют текущему секущему условию $B_{k+1} s_k = y_k$,
но не контролируют невязку прошлых секущих
$R_{\mathrm{past}} = B_k S_{\mathrm{past}} - Y_{\mathrm{past}}$.
Классический результат Шнабеля~\cite{schnabel1983} показывает,
что одновременное точное выполнение всех секущих и симметрии
в общем случае невозможно; ниже воспроизведена компактная
формулировка этой леммы в форме, удобной для дальнейшего анализа.

\begin{lemma}[Шнабель~\cite{schnabel1983}; необходимое условие совместности симметрии и одновременного сохранения всех секущих]\label{lem:incompat}
Пусть $B_k\in\RR^{n\times n}$ удовлетворяет всем прошлым секущим
условиям $B_k s_j = y_j$ ($j=0,\dots,k-1$), и пусть симметричное
обновление $B_{k+1} = B_k + \Delta B$, $\Delta B = \Delta B^\top$,
сохраняет все секущие до~$k$-й включительно:
\[
  B_{k+1} s_j = y_j \;\;(j = 0,\dots,k-1),
  \qquad B_{k+1} s_k = y_k.
\]
В~терминах поправки $\Delta B$ это означает
\[
  \Delta B\, s_j = 0 \;\;(j=0,\dots,k-1),
  \qquad \Delta B\, s_k = r_k,
  \qquad r_k := y_k - B_k s_k.
\]
Тогда остаток $r_k$ обязан быть ортогонален всем прошлым шагам:
\begin{equation}\label{eq:incompat}
  s_j^\top r_k = 0 \qquad \forall\, j < k.
\end{equation}
\end{lemma}

\begin{proof}
Для произвольного $j < k$:
\begin{equation}\label{eq:incompat-chain}
  s_j^\top r_k
  \;\stackrel{(1)}{=}\; s_j^\top (\Delta B\, s_k)
  \;\stackrel{(2)}{=}\; s_j^\top \Delta B^\top s_k
  \;\stackrel{(3)}{=}\; (\Delta B\, s_j)^\top s_k
  \;\stackrel{(4)}{=}\; 0,
\end{equation}
где~(1)~--- текущее секущее условие $\Delta B\, s_k = r_k$;
(2)~--- симметрия $\Delta B = \Delta B^\top$;
(3)~--- тождество $u^\top A^\top v = (A u)^\top v$;
(4)~--- сохранение $j$-й прошлой секущей $\Delta B\, s_j = 0$.
\end{proof}

\begin{remark}[Когда условие~\eqref{eq:incompat} выполнено]
Пусть $F = \nabla f$, $f\in C^2$. По~интегральной форме теоремы
о~среднем для градиента
\[
  y_j \;=\; \nabla f(x_j + s_j) - \nabla f(x_j) \;=\; \bar{H}_j\, s_j,
  \qquad
  \bar{H}_j \;:=\; \int_0^1 \nabla^2 f(x_j + t\, s_j)\, dt,
\]
где $\bar{H}_j$~--- средний гессиан вдоль $j$-го шага (симметричен).
Если $B_k$ симметричен и удерживает прошлые секущие $B_k s_j = y_j$,
то $s_j^\top B_k = (B_k s_j)^\top = (\bar{H}_j s_j)^\top = s_j^\top \bar{H}_j$,
и левая часть~\eqref{eq:incompat} принимает вид
\begin{equation}\label{eq:incompat-hessian}
  s_j^\top r_k
  \;=\; s_j^\top y_k - s_j^\top B_k s_k
  \;=\; s_j^\top \bar{H}_k s_k - s_j^\top \bar{H}_j s_k
  \;=\; s_j^\top (\bar{H}_k - \bar{H}_j)\, s_k.
\end{equation}

\emph{Случай квадратичной $f$.} Если $f(x) = \tfrac12 x^\top A x + b^\top x + c$,
то $\nabla^2 f \equiv A$, и потому $\bar{H}_k = \bar{H}_j = A$ для любых $j,k$.
Правая часть~\eqref{eq:incompat-hessian} обращается в~нуль тождественно,
условие~\eqref{eq:incompat} выполнено автоматически~--- симметричное
обновление, одновременно удерживающее все секущие, существует.

\emph{Случай неквадратичной $f$.} Если $\nabla^2 f$~--- не~константа, то
$\bar{H}_k - \bar{H}_j$ зависит от~траектории и~ненулева в~общем положении;
разность исчезает лишь асимптотически, $x_k, x_j \to x^*$, по~непрерывности
гессиана. Поэтому в~произвольной конечной окрестности
условие~\eqref{eq:incompat} не~выполнено~--- симметричного обновления,
точно удерживающего все секущие, в~общем случае \emph{не существует}.
Это и~есть та~жёсткость, которую в~\S\ref{sec:ss_construction}
снимает~SS-конструкция: вместо точного удержания прошлых секущих
минимизируется их~остаток в~ортогональном к~$s_k$ дополнении.
\end{remark}

\section{Конструкция и алгоритм SS-\texorpdfstring{$\mathcal{U}$}{U}}
\label{sec:ss_construction}

\begin{definition}[Секантно-стабилизированное обновление]
\label{def:ss}
Пусть $\mathcal{U}$~--- базовое симметричное квазиньютоновское
обновление, удовлетворяющее текущему секущему условию
$\mathcal{U}(B_k, s_k, y_k)\, s_k = y_k$,
$B'_k:=\mathcal{U}(B_k,s_k,y_k)$, $R'_k:=B'_k S_p - Y_p$.
\emph{Секантно-стабилизированное расширение}
$\mathrm{SS}\text{-}\mathcal{U}$ задаётся ранг-1 коррекцией
\begin{equation}\label{eq:ss_update}
  B_{k+1} \;=\; B'_k \;+\; \sigma_k^*\, u_k^*\, u_k^{*\top},
  \qquad u_k^* \perp s_k,\;\; \|u_k^*\| = 1,
\end{equation}
направленной на минимизацию остатка прошлых секущих в~$s_k^\perp$,
\begin{equation}\label{eq:ss_varprob}
  (\sigma_k^*, u_k^*) \;\in\; \argmin\bigl\{\,\|R'_k + \sigma\, u u^\top S_p\|_F^{\,2}
     \;:\; \|u\|=1,\; u\perp s_k,\; \sigma\in\RR\,\bigr\}.
\end{equation}
Точная задача~\eqref{eq:ss_varprob} не~имеет замкнутой формы
(Лемма~\ref{lem:ss_variational}, п.~2); в~Алгоритме~\ref{alg:ss_u}
применяется гибридная стратегия. Сначала $u_k^*$~--- ведущий по~$|\lambda|$
собственный вектор симметричной матрицы $P_k M'_k P_k$ на~$s_k^\perp$,
где $P_k = I - \hat s_k\hat s_k^\top$, $M'_k:=\tfrac12(R'_k S_p^\top + S_p R_k^{\prime\top})$;
выбор $u_k^*$ отвечает упрощённой задаче после замены знаменателя
$\|S_p^\top u\|^2$ на~$\|u\|^2 = 1$ в~\eqref{eq:ss_phi_min_sigma}
(Лемма~\ref{lem:ss_variational}, п.~3). Затем при~зафиксированном~$u=u_k^*$
коэффициент $\sigma_k^*$ берётся \emph{точным} условным оптимумом
функционала~\eqref{eq:ss_varprob}:
\begin{equation}\label{eq:ss_sigma}
  \sigma_k^* \;=\; -\frac{u_k^{*\top} R'_k S_p^{\top} u_k^*}
                       {\|S_p^{\top} u_k^*\|^2}.
\end{equation}
В~результате пара $(u_k^*,\sigma_k^*)$~--- точный условный минимум
по~$\sigma$ при~приближённо найденном направлении~$u_k^*$;
устойчивость знаменателя обеспечивается safeguard~(А3)
Леммы~\ref{lem:bd_composite}.
\end{definition}

\begin{lemma}[Условный оптимум $\sigma$ и характеризация $u_k^*$ в~Алгоритме~\ref{alg:ss_u}]
\label{lem:ss_variational}
Пусть $R'_k = B'_k S_p - Y_p$, $M'_k = \tfrac{1}{2}(R'_k S_p^\top + S_p R_k^{\prime\top})$,
$S_p$~--- полного столбцового ранга,
$\Phi(\sigma, u) := \|R'_k + \sigma\, u u^\top S_p\|_F^2$.
Тогда:
\begin{enumerate}
  \item для каждого $u\in s_k^\perp$ с~$\|u\|=1$ и~$\|S_p^\top u\|\ne 0$
    условный оптимум $\sigma$ при фиксированном~$u$ равен
    $\sigma_k^*(u) = -(u^\top M'_k u)/\|S_p^\top u\|^2$, и подстановка даёт
    \begin{equation}\label{eq:ss_phi_min_sigma}
      \min_\sigma \Phi(\sigma, u)
      \;=\; \|R'_k\|_F^2 \;-\; \frac{(u^\top M'_k u)^2}{\|S_p^\top u\|^2};
    \end{equation}
  \item точная минимизация~\eqref{eq:ss_phi_min_sigma} по~$u\in s_k^\perp \cap S^{n-1}$
    приводит к~самосогласованной нелинейной задаче
    \[
      P_k\bigl(M'_k - \lambda(u)\, S_p S_p^\top\bigr)\, u
      \;=\; \tfrac{1}{2}(u^\top M'_k u)\, u,
      \qquad \lambda(u) := \tfrac{u^\top M'_k u}{2\,\|S_p^\top u\|^2},
    \]
    которая, в~отличие от~стандартной обобщённой задачи на~собственные
    значения, не~допускает замкнутого решения (множители $\lambda(u)$
    и~$\tfrac12 u^\top M'_k u$ зависят от~$u$);
  \item замена $\|S_p^\top u\|^2 \to \|u\|^2 = 1$ в~знаменателе~\eqref{eq:ss_phi_min_sigma}
    сводит задачу к~максимизации $(u^\top M'_k u)^2$ на~$s_k^\perp\cap S^{n-1}$,
    у~которой ведущий по~$|\lambda|$ собственный вектор $P_k M'_k P_k$
    на~$s_k^\perp$ доставляет точный максимум $\lambda_*^2$
    (тождество Куранта--Фишера для $P_k M'_k P_k$).
    В Алгоритме~\ref{alg:ss_u} в~качестве $u_k^*$ берётся именно этот
    собственный вектор; соответствующее $\sigma_k^* := \sigma_k^*(u_k^*)$ задано
    формулой~\eqref{eq:ss_sigma} из~п.~1.
\end{enumerate}
\end{lemma}

\begin{proof}
\textbf{П.~1.} Раскрытие $\|R'_k + \sigma uu^\top S_p\|_F^2$ через
$\|A+B\|_F^2 = \|A\|_F^2 + 2\,\mathrm{tr}(A^\top B) + \|B\|_F^2$ даёт
\begin{equation}\label{eq:ss_phi_quad}
  \Phi(\sigma, u)
  = \|R'_k\|_F^2 + 2\sigma\,(u^\top M'_k u) + \sigma^2\,\|S_p^\top u\|^2,
\end{equation}
где использованы $\mathrm{tr}(R_k^{\prime\top} u u^\top S_p) = u^\top S_p R_k^{\prime\top} u = u^\top M'_k u$
(полусумма скаляров $u^\top R'_k S_p^\top u = u^\top S_p R_k^{\prime\top} u$) и
$\|u u^\top S_p\|_F^2 = \|u\|^2\,\|S_p^\top u\|^2 = \|S_p^\top u\|^2$ при~$\|u\|=1$.
Стационарность по~$\sigma$: $\partial_\sigma \Phi = 0$
даёт~$\sigma_k^*(u)$; подстановка в~\eqref{eq:ss_phi_quad}~--- \eqref{eq:ss_phi_min_sigma}.

\textbf{П.~2.} Обозначим $\rho(u) := (u^\top M'_k u)^2/\|S_p^\top u\|^2$,
$f := u^\top M'_k u$, $g := \|S_p^\top u\|^2$. Случай $f=0$ соответствует
глобальному минимуму $\rho(u)=0$ и~достигается на~любом $u\perp s_k$
с~$u^\top M'_k u=0$; ниже рассматриваем нетривиальный случай $f\ne 0$.

По~правилу множителей Лагранжа стационарность $\rho$ на~$\|u\|=1, u\perp s_k$
эквивалентна $\nabla\rho = 2\alpha u + \beta s_k$ для некоторых $\alpha, \beta\in\RR$.
Скалярно умножая на~$u$ (с использованием $u\perp s_k$, $\|u\|=1$):
$u^\top\nabla\rho = 2\alpha$. Прямой расчёт
$\nabla\rho = (4f/g)\,M'_k u - (2f^2/g^2)\,S_p S_p^\top u$ даёт
$u^\top\nabla\rho = 4f^2/g - 2f^2/g = 2f^2/g = 2\rho(u)$, откуда
$\alpha = \rho(u) = f^2/g > 0$ при~$f\ne 0$.
Подстановка $\alpha = \rho(u)$ в~$\nabla\rho - 2\alpha u - \beta s_k = 0$
и~проектирование на~$s_k^\perp$ ($P_k u = u$) дают
\[
  P_k\bigl[(4f/g)\,M'_k - (2f^2/g^2)\,S_p S_p^\top\bigr]\, u
  \;=\; (2f^2/g)\,u.
\]
Деление на~$4f/g$ и~подстановка $\lambda = f/(2g)$:
$P_k(M'_k - \lambda S_p S_p^\top)\,u = (f/2)\,u$. Поскольку $\lambda$ и $f/2$
зависят от~$u$, эта задача не~является обобщённой задачей на собственные значения.

\textbf{П.~3.} Подстановка $\|S_p^\top u\|^2 \equiv 1$ в~\eqref{eq:ss_phi_min_sigma}
сводит её к~$\|R'_k\|_F^2 - (u^\top M'_k u)^2$, и~задача превращается в~максимизацию
$h(u) := (u^\top M'_k u)^2$ на~$\{u\in\RR^n: \|u\|=1,\,u\perp s_k\}$.
Поскольку $P_k u = u$ на~$s_k^\perp$, имеем $u^\top M'_k u = u^\top P_k M'_k P_k u
=: u^\top A_k u$, где $A_k := P_k M'_k P_k$ симметрична. Стационарность~$h$
под ограничениями $\|u\|=1,\,u\perp s_k$ по правилу множителей Лагранжа
эквивалентна $\nabla h = 2\lambda u + \mu s_k$ с~$\nabla h = 4\,(u^\top A_k u)\,A_k u$;
проектируя на~$s_k^\perp$ ($P_k$, причём $P_k A_k u = A_k u$):
$4\,(u^\top A_k u)\,A_k u = 2\lambda u$, т.\,е.\
$A_k u = \mu_* u$ с~$\mu_* = \lambda/\bigl(2\,u^\top A_k u\bigr)$. Таким образом,
стационарные точки совпадают с~собственными векторами $A_k$ на~$s_k^\perp$
(стандартный факт о~экстремумах квадратичной формы на~сфере;
ср.\ Голуб--ван~Лоан~\cite[\S 8.1]{golub2013}).
В~каждом стационарном векторе $u$ имеем $u^\top A_k u = \mu_*$, поэтому
$h(u) = \mu_*^2$; максимум достигается ведущим по~$|\mu_*|$ собственным вектором,
давая $h_{\max} = \lambda_*^2 := \max_{i:\,v_i\perp s_k}\mu_i^2$.
\end{proof}

\begin{remark}[Качество упрощения при~условии safeguard]
\label{rem:ss_variational_alg}
Точное решение~\eqref{eq:ss_varprob} (Лемма~\ref{lem:ss_variational}, п.~2)
и~используемое в~Алгоритме~\ref{alg:ss_u} упрощение (п.~3)
отличаются заменой $\|S_p^\top u\|^2$ на~$\|u\|^2 = 1$ в~знаменателе
условного оптимума по~$\sigma$. При safeguard~(A3) Леммы~\ref{lem:bd_composite}
($\|S_p^\top u\|^2 \ge \varepsilon_S$) знаменатель условного оптимума ограничен
снизу, и~оценка bounded deterioration (Лемма~\ref{lem:bd_composite})
корректно опирается на~точное $\sigma_k^*$ из~\eqref{eq:ss_sigma}
независимо от того, является ли выбранный $u_k^*$ точным максимизатором
$\rho(u)$ или ведущим собственным вектором $P_k M'_k P_k$.
\end{remark}

\paragraph{Базовые обновления $\mathcal{U}$.}
Алгоритм~SS-$\mathcal{U}$, приведённый ниже, не~зависит от~выбора
$\mathcal{U}$, кроме явного вычисления $B'_k=\mathcal{U}(B_k,s_k,y_k)$.
В~численных экспериментах настоящей главы рассматриваются
два классических индефинитных симметричных обновления:
\[
  \mathcal{U}_{\mathrm{SR1}}(B,s,y)
   \;=\; B + \frac{r r^\top}{r^\top s},
  \qquad
  \mathcal{U}_{\mathrm{PSB}}(B,s,y)
   \;=\; B + \frac{r s^\top + s r^\top}{s^\top s}
        - \frac{(s^\top r)\, s s^\top}{(s^\top s)^2},
  \qquad r = y - Bs.
\]
SR1 требует skip-rule~\cite[\S 6.2]{bfgs}: при
$|s^\top r|<\varepsilon_{\mathrm{skip}}\Norm{s}\Norm{r}$
обновление пропускается, $B'_k = B_k$. PSB skip-правил не~требует.
SS-расширение строится по~единому шаблону~\eqref{eq:ss_update},
ниже~--- общий псевдокод.

\medskip
Численные защиты вне теории~--- вырождение $u_k^*$ после репроекции
на~$s_k^\perp$ (порог $\varepsilon_u$), устойчивость знаменателя
$\sigma_k^*$ (порог $\varepsilon_S$) и собственный skip-rule базового
$\mathcal{U}$ (для $\mathcal{U}=\mathrm{SR1}$)~--- обсуждаются
в~замечании~\ref{rem:ss_u_safeguards}; в~псевдокоде эти пороги опущены
ради ясности и заменены проверкой $u_k^*\ne 0$, $S_p^\top u_k^*\ne 0$.
Ключевое наблюдение для собственной задачи $P_k M'_k P_k$:
$M'_k=\tfrac12(R'_k S_p^\top+S_p R_k^{\prime\top})$~--- сумма двух
матриц ранга $\le p+1$, поэтому
\begin{equation}\label{eq:ss_low_rank}
  \mathrm{rank}(P_k M'_k P_k)
   \;\le\; \mathrm{rank}(M'_k)
   \;\le\; 2(p+1),
\end{equation}
и весь ненулевой спектр $A_k=P_k M'_k P_k$ лежит
в~$\mathrm{range}(P_k[S_p,R'_k])$. Собственная задача переносится
из~$\RR^n$ в~подпространство размерности $r\le 2(p+1)$ через
QR-ортогонализацию этого базиса (приём, стандартный для блочных
Ланцош-итераций~\cite[Гл.~13]{parlett1998}; ср.\ также подход
Fang--Saad~\cite{fang2009} для мультисекантных схем).
Алгоритм~\ref{alg:ss_u} непосредственно реализует эту редукцию;
эквивалентность с~прямой формой $\mathtt{eigh}(A_k)$ и~детализация
стоимости~--- в~Замечании~\ref{rem:ss_u_cost}.

\begin{breakablealgorithm}
\caption{SS-$\mathcal{U}$\ $(\nabla f,\,x_0,\,p_{\max},\,\varepsilon;\,\mathcal{U})$}
\label{alg:ss_u}
\begin{algorithmic}
\Require $\nabla f\colon\RR^n\to\RR^n$,\ $x_0\in\RR^n$,\
         $p_{\max}\ge 0$,\ $\varepsilon>0$
\Require базовое симметричное обновление $\mathcal{U}(B,s,y)$, удерживающее
         текущую секущую $\mathcal{U}(B,s,y)\,s=y$ (например,
         $\mathcal{U}_{\mathrm{SR1}}$ со~skip-rule или $\mathcal{U}_{\mathrm{PSB}}$)
\State $B_0 \leftarrow I_n$,\quad $k \leftarrow 0$
\While{$\Norm{\nabla f(x_k)}\ge \varepsilon$}
  \State Решить $B_k\,d_k = -\nabla f(x_k)$
  \State $x_{k+1}\leftarrow x_k + d_k$,\quad
         $s_k\leftarrow d_k$,\quad
         $y_k\leftarrow \nabla f(x_{k+1})-\nabla f(x_k)$
  \State $B'_k \leftarrow \mathcal{U}(B_k, s_k, y_k)$
       \Comment{базовая симметричная итерация (с её собственным skip-rule)}
  \State $p \leftarrow \min(p_{\max},k)$,\quad
         $S_p\leftarrow [s_k\mid\cdots\mid s_{k-p}]$,\quad
         $Y_p\leftarrow [y_k\mid\cdots\mid y_{k-p}]$
  \State $R'_k \leftarrow B'_k S_p - Y_p$,\quad
         $M'_k \leftarrow \tfrac{1}{2}(R'_k S_p^{\top} + S_p R^{\prime\top}_k)$
  \State $\hat s_k\leftarrow s_k/\Norm{s_k}$,\quad
         $P_k \leftarrow I-\hat s_k\hat s_k^{\top}$
  \State $(\lambda_k^{*},u_k^{*}) \leftarrow$ ведущая по~$|\lambda|$ собственная пара
         $P_k M'_k P_k$;\ $\Norm{u_k^{*}}=1$
       \Comment{$\mathrm{rank}\le 2(p{+}1)$~--- замечание~\ref{rem:ss_u_cost}}
  \If{$u_k^{*}\ne 0$ \textbf{and} $S_p^{\top} u_k^{*}\ne 0$}
    \State $\sigma_k^{*}\leftarrow -(u_k^{*\top} R'_k S_p^{\top} u_k^{*})/\Norm{S_p^{\top} u_k^{*}}^2$
    \State $B_{k+1}\leftarrow B'_k + \sigma_k^{*}\,u_k^{*} u_k^{*\top}$
  \Else
    \State $B_{k+1}\leftarrow B'_k$
       \Comment{коррекция пропущена (включая $k{=}0$, окно пусто)}
  \EndIf
  \State $k\leftarrow k+1$
\EndWhile
\Return $x_k$
\end{algorithmic}
\end{breakablealgorithm}

\begin{remark}[Стоимость и эквивалентность с~прямой формой]\label{rem:ss_u_cost}
Псевдокод оставляет вычисление ведущей пары $(\lambda_k^*,u_k^*)$
матрицы $P_k M'_k P_k$ абстрактным. В реализации используется редукция
к~пространству $\mathrm{range}(P_k[S_p,R'_k])$: ортогонализация
$W = (I-\hat s_k\hat s_k^\top)[S_p\mid R'_k]\in\RR^{n\times 2(p{+}1)}$
тонким QR в~$Q\in\RR^{n\times r}$, $r\le 2(p+1)$, и эрмитов
собственный разбор $\tilde M = Q^\top M'_k Q\in\RR^{r\times r}$;
поднятие $u_k^* = Q v_{j^*}$, $j^*=\arg\max_i|\lambda_i|$. Стоимость:
$R'_k$~--- $O(n^2 p)$ на~плотном $B'_k$, QR~--- $O(np^2)$, $\mathtt{eigh}(\tilde M)$~---
$O(p^3)$; суммарно $O(n^2 p + np^2 + p^3)$ на~итерацию, память $O(np)$
против $O(n^2)$ при~хранении полного $A_k$.

Эквивалентная прямая форма~--- собрать $A_k=P_k M'_k P_k\in\RR^{n\times n}$
явно и~вызвать $\mathtt{eigh}(A_k)$ напрямую (QR-итерация на
трёхдиагонализованной форме~\cite[\S 8.3]{golub2013}, $O(n^3)$); в~силу
оценки ранга~\eqref{eq:ss_low_rank} обе формы дают ту~же ведущую
собственную пару $(\lambda^*,u_k^*)$ с~различием порядка машинной
точности. В~численных экспериментах главы $n\le 50$, и обе формы
пренебрежимы по~стоимости относительно вычисления $\nabla f$ и
LU-решения системы $B_k d_k=-\nabla f$ (последнее~--- $O(n^3)$);
фактически используется прямая форма $\mathtt{eigh}(A_k)$.
Редукция~\eqref{eq:ss_low_rank} становится существенной при $n\gg p$
(типично $p\le 10$, $n\gtrsim 100$ даёт $\sim n/p$-кратное ускорение).
\end{remark}

\begin{remark}[Статус численных порогов]\label{rem:ss_u_safeguards}
Дефолты $\varepsilon_{\mathrm{skip}}=10^{-8}$ (skip-rule базового
$\mathcal{U}=\mathrm{SR1}$, ср.\ \cite[\S 6.2]{bfgs}), $\varepsilon_u=10^{-12}$
и $\varepsilon_S=10^{-15}$ (\eqref{eq:ss_sigma}) выбраны как мягкие
численные защиты вблизи машинной точности~--- цель каждого охранника:
не допустить деление на малое~/ потерю ортогональности после репроекции
на $s_k^\perp$. Дефолты не основаны на параметрическом анализе
чувствительности и могут потребовать адаптации для новых классов задач,
особенно с экстремальной обусловленностью гессиана либо в~смешанной
точности. Качественно: $\varepsilon_{\mathrm{skip}}$ контролирует
консервативность skip-rule (увеличение даёт более частый skip); $\varepsilon_u$,
$\varepsilon_S$ срабатывают редко на гладких задачах настоящей работы.
Систематический sweep по $(\varepsilon_{\mathrm{skip}},\varepsilon_u,\varepsilon_S)$
выходит за рамки локального анализа и обозначен как направление
дальнейших исследований.
\end{remark}

\section{Свойства}\label{sec:ss_properties}

\begin{theorem}[Свойства SS-$\mathcal{U}$]\label{thm:ss_props}
Обновление~\eqref{eq:ss_update} обладает следующими свойствами:
\begin{enumerate}
  \item \textbf{Текущее секущее условие:}
    $B_{k+1} s_k = y_k$ (точно).
  \item \textbf{Симметрия:}
    если $\mathcal{U}$ сохраняет симметрию, то $B_{k+1} = B_{k+1}^\top$.
  \item \textbf{Монотонность невязки секущих.} Обозначим
    $R_{k+1} := B_{k+1} S_p - Y_p$ (невязка всех~$p{+}1$ секущих
    после композитного шага); в~силу п.~1 первый столбец $R_{k+1}$
    (отвечающий~$s_k$) и~$R'_k$ нулевой, поэтому $\|R_{k+1}\|_F$ и~$\|R'_k\|_F$
    совпадают с~нормами невязок \emph{прошлых} секущих. Тогда
    \[
      \|R_{k+1}\|_F^2
        \;=\; \|R'_k\|_F^2
        \;-\; \frac{(u_k^{*\top} M'_k\, u_k^*)^2}{\|S_p^\top u_k^*\|^2}
        \;=\; \|R'_k\|_F^2 - (\sigma_k^*)^2\,\|S_p^\top u_k^*\|^2.
    \]
  \item \textbf{Ранг обновления:}
    $\mathrm{rank}(B_{k+1} - B_k) \leq \mathrm{rank}(\mathcal{U}) + 1$.
\end{enumerate}
\end{theorem}

\begin{proof}\leavevmode
\begin{enumerate}
  \item Из $u_k^* \perp s_k$:
    $B_{k+1} s_k = B'_k s_k + \sigma_k^*(u_k^{*\top} s_k)\, u_k^* = y_k + 0 = y_k$.
  \item $B_{k+1} = B'_k + \sigma_k^*\, u_k^*\, u_k^{*\top}$~---
    сумма двух симметричных матриц.
  \item Пусть $\tilde{B} = B'_k + \sigma\, u u^\top$. Тогда
    $\|\tilde{B} S_p - Y_p\|_F^2
    = \|R'\|_F^2 + 2\sigma\,(u^\top M' u) + \sigma^2\,\|S_p^\top u\|^2$.
    Подстановка $\sigma_k^*$ из~\eqref{eq:ss_sigma}~--- точного условного
    оптимума по~$\sigma$~--- даёт убывание ровно на
    $(u^\top M' u)^2 / \|S_p^\top u\|^2 = (\sigma_k^*)^2\,\|S_p^\top u\|^2$
    (равенство, не~неравенство).
  \item $B_{k+1} - B_k = \underbrace{(B'_k - B_k)}_{\mathrm{rank}\,\le\,\mathrm{rank}(\mathcal{U})}
        + \underbrace{\sigma_k^*\, u_k^*\, u_k^{*\top}}_{\mathrm{rank}\,\le\,1}$,
        и~$\mathrm{rank}(A+B)\le\mathrm{rank}(A)+\mathrm{rank}(B)$. \qedhere
\end{enumerate}
\end{proof}

\section{Связь с известными подходами}\label{sec:ss_related}

Несовместность симметрии и одновременного выполнения нескольких
секущих условий~\cite{schnabel1983} (Лемма~\ref{lem:incompat})
разрешается в литературе по симметричным multi-secant
квазиньютоновским обновлениям по разным траекториям; конструкция
SS-$\mathcal{U}$ из~\eqref{eq:ss_update} занимает в~этом~ряду
характерную позицию.

Gratton, Malmedy и~Toint~\cite{gratton2015} вводят весовую multi-secant
модификацию: $B_{k+1}$~--- решение задачи $\min \|B-B_k\|_F^2$ при
штрафе $\sum_{j} w_j\|B s_j - y_j\|^2$ за~невязку прошлых секущих и
ограничении симметрии; текущая секущая выполняется только в~смысле
того~же штрафа, не~точно. Boutet, Haelterman и~Degroote~\cite{boutet2020}
конкретизируют эту схему как~расширение~PSB через весовую multi-secant
невязку; решение имеет ранг $\le 2(p{+}1)$ и~получается из уравнения
Сильвестра. В~последующих работах те~же авторы развивают подход в~двух
направлениях: gPSB~\cite{boutet2021} жертвует точной симметрией ради
точного выполнения нескольких секущих, а grouped/ordered~\cite{boutet2022}
разрешает противоречие иерархически~--- через лексикографическое
выполнение упорядоченных групп секущих, без~явной декомпозиции
$\RR^n = \operatorname{span}(s_k)\oplus s_k^\perp$.

Scieur, Liu, Pumir и~Boumal~\cite{scieur2021} формулируют симметричный
multi-secant как задачу наименьших квадратов на~всём окне с~PSD-регуляризацией
$\mu I$; все секущие, включая текущую, выполняются приближённо, направления
коррекции выделенной структуры в~$s_k^\perp$ не~имеют.

Конструкция~\eqref{eq:ss_update} отличается от~перечисленных схем тремя
структурными признаками. \emph{Во-первых}, текущая секущая
$B_{k+1}s_k = y_k$ и~симметрия выполняются \emph{точно}~--- по
построению базы $B'_k = \mathcal{U}(B_k,s_k,y_k)$ и~ортогональности
$u_k^*\perp s_k$ (теорема~\ref{thm:ss_props}, п.~1--2);
\emph{во-вторых}, коррекция имеет ранг~1 ($\sigma_k^* u_k^* u_k^{*\top}$),
а~не~ранг $\le 2(p{+}1)$ как~у~\cite{gratton2015,boutet2020};
\emph{в-третьих}, целевая функция~--- невязка прошлых секущих
$\|R'_k + \sigma\, u u^\top S_p\|_F^2$ в~подпространстве $s_k^\perp$,
а~не близость к~$B_k$, что приводит к~обобщённой собственной задаче
$M'_k u = \lambda S_p S_p^\top u$ на~$s_k^\perp$
(Лемма~\ref{lem:ss_variational}). Двухэтапная схема «симметричное базовое
$\mathcal{U}$~--- затем rank-1 надстройка в~$s_k^\perp$» оставляет выбор
базы независимым: в~настоящей работе $\mathcal{U}\in\{\mathrm{SR1},\mathrm{PSB}\}$
(\S\ref{sec:ss_construction}), но~алгоритм~\ref{alg:ss_u} применим к~любому
симметричному обновлению, удерживающему текущую секущую.

\section{Сверхлинейная сходимость}\label{sec:ss_superlinear}

\begin{theorem}[Условная Q-сверхлинейная сходимость SS-$\mathcal{U}$]
\label{thm:ss_dm}
Пусть выполнены предположения:
\begin{enumerate}
  \item[\textup{(H1)}] $F = \nabla f$, гессиан $\nabla^2 f$ липшицев в окрестности
    $x^*$ с константой $L_H$;
  \item[\textup{(H2)}] метод SS-$\mathcal{U}$ сходится: $x_k \to x^*$, $s_k \to 0$,
    $\nabla f(x^*) = 0$;
  \item[\textup{(H3)}] предельный гессиан $J^* := \nabla^2 f(x^*)$ невырожден;
  \item[\textup{(H4)}] начиная с некоторого $k_0$ глобализация выдаёт полный
    квазиньютоновский шаг $s_k = -B_k^{-1}\nabla f(x_k)$, причём
    $\|B_k^{-1}\| \le M_*$ равномерно при $k \ge k_0$;
  \item[\textup{(H5)}] bounded deterioration в форме Денниса--Море:
    обозначим $E_k := B_k - J^*$~--- ошибку аппроксимации гессиана;
    существует последовательность $\beta_k \ge 0$ с
    $\sum_{k\ge k_0}\beta_k < \infty$ такая, что в окрестности $x^*$
    \[
      \|E_{k+1}\|_F^2 \;\le\; \|E_k\|_F^2 \;-\; \frac{\|E_k s_k\|^2}{\|s_k\|^2}
                          \;+\; \beta_k.
    \]
\end{enumerate}
Тогда:
\begin{enumerate}
  \item \textup{(условие Денниса--Море для $B_k$)}
    \[
      \frac{\|(B_k - J^*)\, s_k\|}{\|s_k\|} \;\xrightarrow{k\to\infty}\; 0;
    \]
  \item \textup{(Q-сверхлинейность)}
    $\dfrac{\|x_{k+1} - x^*\|}{\|x_k - x^*\|} \to 0$.
\end{enumerate}
\end{theorem}

\begin{proof}
\textit{Шаг 1 (telescoping (H5) даёт сходимость ряда).}
Суммируя~(H5) от~$k_0$ до~$N$:
\[
  \sum_{k=k_0}^{N} \frac{\|E_k\, s_k\|^2}{\|s_k\|^2}
  \;\le\; \|E_{k_0}\|_F^2 \;-\; \|E_{N+1}\|_F^2
       \;+\; \sum_{k=k_0}^{N}\beta_k
  \;\le\; \|E_{k_0}\|_F^2 \;+\; \sum_{k\ge k_0}\beta_k
  \;<\;\infty,
\]
где использована неотрицательность $\|E_{N+1}\|_F^2$ и
суммируемость $\beta_k$. Переходя к пределу $N\to\infty$,
получаем $\sum_{k\ge k_0}\|E_k s_k\|^2/\|s_k\|^2 < \infty$, откуда
необходимое условие сходящегося ряда даёт
\begin{equation}\label{eq:ss_dm_bk}
  \frac{\|(B_k - J^*)\,s_k\|}{\|s_k\|}
  \;=\; \frac{\|E_k s_k\|}{\|s_k\|} \;\xrightarrow{k\to\infty}\; 0,
\end{equation}
т.\,е.\ пункт~1 теоремы (условие Денниса--Море для~$B_k$).

\textit{Шаг 2 (Q-сверхлинейность через характеризацию Денниса--Море).}
По~(H3)--(H4) шаг $s_k = -B_k^{-1}\nabla f(x_k)$ корректно определён при $k \ge k_0$,
так что $\nabla f(x_k) = -B_k s_k$ и
\[
  \|\nabla f(x_k) + J^* s_k\|
  = \|(J^* - B_k) s_k\|
  = \|(B_k - J^*)\, s_k\|.
\]
Согласно характеризации Денниса--Море~\cite[Th.~2.2]{dennis1974},
для сходящейся к корню $x^*$ ($\nabla f(x^*) = 0$) последовательности
$x_{k+1} = x_k - B_k^{-1}\nabla f(x_k)$ при невырожденном $J^*$ Q-сверхлинейная
сходимость эквивалентна условию
\[
  \frac{\|\nabla f(x_k) + J^* s_k\|}{\|s_k\|} \xrightarrow{k\to\infty} 0,
\]
т.\,е. в наших обозначениях ровно DM-условию пункта~1, установленному
на Шаге~1. Применяя эквивалентность, получаем
$\|x_{k+1} - x^*\|/\|x_k - x^*\| \to 0$.
\end{proof}

\begin{remark}[Содержательная роль теоремы]\label{rem:ss_dm_role}
Теорема~\ref{thm:ss_dm} утверждает скорость сходимости \emph{при условии} (H2)~---
самой сходимости итерации $x_k\to x^*$, которая для базовых индефинитных
обновлений ($\mathcal{U}\in\{\mathrm{SR1},\mathrm{PSB}\}$) без
стабилизирующей MS-коррекции в~общем случае не гарантирована
(см.~\cite{conn1991} и~Замечание~\ref{rem:ss_dm_assumptions}).
\emph{Если} итерация сходится (что эмпирически обеспечивается MS-коррекцией
и~backtracking-Армихо), то~скорость~--- Q-сверхлинейная.
\end{remark}

\begin{remark}[Роль предположений]\label{rem:ss_dm_assumptions}
Предположения~(H3) и~(H4) суть стандартные локальные допущения о хорошей
обусловленности задачи: невырожденность $J^*$ нужна для применимости
Денниса--Море~\cite{dennis1974}, а допустимость полного шага в окрестности
$x^*$~--- классическое следствие условия Армихо с $c_1 < 1/2$ для
дважды-дифференцируемой $f$ (\cite{bfgs}, Th.~3.6).

Условие~(H5) воспроизводит классическую форму bounded deterioration по
Деннису--Море~\cite[Th.~2.2]{dennis1974} и наследуется от выбранного
базового обновления~$\mathcal{U}$: для $\mathcal{U}\in\{\mathrm{BFGS},
\mathrm{PSB}\}$ оно установлено в~\cite{bfgs} и~\cite[Th.~8.2.4]{dennis1996},
для~$\mathcal{U}=\mathrm{SR1}$ со skip-rule~--- в~\cite{conn1991}.
Для композитного обновления $\mathcal{U}\to\mathrm{SS}\text{-}\mathcal{U}$
сохранение~(H5) формально установлено в~Лемме~\ref{lem:bd_composite}
(см.\ ниже) при дополнительном предположении~(A4) о~входной
Q-линейной сходимости. Без MS-коррекции~SR1 может нарушать~(H5),
что даёт практические расходимости~\cite{conn1991}, наблюдаемые в~численных
экспериментах \S\ref{sec:ss_numerics} (см.~рис.~\ref{fig:ndim_stat_sr1}).
\end{remark}

\begin{lemma}[Композитное bounded deterioration для $\mathrm{SS}\text{-}\mathcal{U}$]
\label{lem:bd_composite}
Пусть выполнены следующие условия в окрестности $B_\delta(x^*)=\{x:\|x-x^*\|\le\delta\}$:
\begin{itemize}
  \item[\textup{(A1)}] $F=\nabla f$, $\nabla^2 f$ липшицев с~константой $L_H$ в~$B_\delta(x^*)$;
  \item[\textup{(A2)}] базовое обновление $\mathcal{U}$ удовлетворяет
        \emph{Lipschitz-контролируемому bounded deterioration}: существует
        $c_{\mathcal{U}}\ge 0$ такая, что в~$B_\delta(x^*)$
        \begin{equation}\label{eq:bd_base}
          \|E'_k\|_F^{\,2}
          \;\le\; \|E_k\|_F^{\,2}
                 \;-\; \frac{\|E_k s_k\|^2}{\|s_k\|^2}
                 \;+\; c_{\mathcal{U}}\, L_H\, \alpha_k,
          \qquad
          \alpha_k:=\tfrac12\bigl(\|x_k-x^*\|+\|x_{k+1}-x^*\|\bigr);
        \end{equation}
        (Деннис--Шнабель~\cite[Th.~8.2.4]{dennis1996} для~$\mathcal{U}=\mathrm{PSB}$;
         Конн--Гулд--Тойнт~\cite{conn1991} для~$\mathcal{U}=\mathrm{SR1}_{\mathrm{skip}}$);
  \item[\textup{(A3)}] Алгоритм~\ref{alg:ss_u} применяет SS-коррекцию только при
        $\|S_p^{\top} u_k^*\|^2 \ge \varepsilon_S$; в~противном случае
        $B_{k+1}=B'_k$ и~SS-вклад равен нулю;
  \item[\textup{(A4)}] начальная точка достаточно близка к~$x^{*}$:
        $\|E_{k_0}\|_F < \delta'$ и~$\|x_{k_0}-x^{*}\| < \delta''$, где
        $\delta',\delta''>0$ зависят от~$L_H,\|J^{*}\|_2,\|(J^{*})^{-1}\|_2,
        \varepsilon_S,c_{\mathcal{U}},p$ (явный выбор дан в~Шаге~7
        доказательства).
\end{itemize}
Здесь $E_k:=B_k-J^*$, $E'_k:=B'_k-J^*$, $E_{k+1}:=B_{k+1}-J^*$~---
ошибки аппроксимации до~базового шага~$\mathcal{U}$, после него и~после
композитной SS-коррекции соответственно. Тогда существуют темп $\rho_*\in[0,1)$,
константа $\Sigma=\Sigma(\rho_*,L_H,\varepsilon_S,p,c_{\mathcal{U}},
\|E_{k_0}\|_F,\|x_{k_0}-x^*\|)<\infty$ и~неотрицательная последовательность
$\beta_k$ с~$\sum_{k\ge k_0}\beta_k\le\Sigma$ такие, что для всех $k\ge k_0$:
\begin{equation}\label{eq:bd_composite}
  \|E_{k+1}\|_F^{\,2}
  \;\le\; \|E_k\|_F^{\,2}
         \;-\; \frac{\|E_k s_k\|^2}{\|s_k\|^2}
         \;+\; \beta_k,
\end{equation}
а~итерация $\{x_k\}$ Q-линейно сходится к~$x^{*}$ с~темпом~$\rho_*$:
$\|x_{k+1}-x^{*}\|\le\rho_*\,\|x_k-x^{*}\|$. В~частности,
$\|E_k\|_F\le\sqrt{\|E_{k_0}\|_F^{\,2}+\Sigma}$ равномерно по~$k\ge k_0$.
\end{lemma}

\begin{proof}
Без ограничения общности предположим, что safeguard~(A3) не~срабатывает~---
иначе $B_{k+1}=B'_k$, и~\eqref{eq:bd_composite} сразу следует из~\eqref{eq:bd_base}.

\medskip
\textbf{Шаг 1 (декомпозиция и проектируемое тождество).}
SS-коррекция:
\[
  B_{k+1} \;=\; B'_k \;+\; \Delta_k^{SS},
  \qquad
  \Delta_k^{SS} \;:=\; \sigma_k^{*}\, u_k^{*}\, u_k^{*\top},
  \qquad u_k^{*}\perp s_k,\ \|u_k^{*}\|=1.
\]
Соответственно $E_{k+1} = E'_k + \Delta_k^{SS}$. Из~$u_k^{*}\perp s_k$:
\begin{equation}\label{eq:bd_skk}
  E_{k+1}\,s_k \;=\; E'_k\, s_k\, +\, \sigma_k^{*} u_k^{*}(u_k^{*\top} s_k) \;=\; E'_k\, s_k,
\end{equation}
т.\,е.\ SS-коррекция не~меняет направленный остаток вдоль~$s_k$.

По линейности скалярного произведения Фробениуса:
\[
  \|E_{k+1}\|_F^{\,2}
  \;=\; \|E'_k\|_F^{\,2}
   \;+\; 2\sigma_k^{*}\langle E'_k,\, u_k^{*} u_k^{*\top}\rangle_F
   \;+\; (\sigma_k^{*})^2\,\|u_k^{*} u_k^{*\top}\|_F^{\,2}.
\]
Поскольку $\langle E'_k,\,u u^{\top}\rangle_F = \mathrm{tr}(E_k'{}^\top uu^\top) = u^{\top}E'_k u$
(скаляр) и~$\|u u^{\top}\|_F^{\,2}=\|u\|^4=1$:
\begin{equation}\label{eq:bd_frob_exp}
  \|E_{k+1}\|_F^{\,2}
  \;=\; \|E'_k\|_F^{\,2}
   \;+\; 2\sigma_k^{*}\,(u_k^{*\top}\,E'_k\,u_k^{*})
   \;+\; (\sigma_k^{*})^2.
\end{equation}

\medskip
\textbf{Шаг 2 (оценка SS-возмущения через треугольное неравенство).}
По Коши--Шварцу и $\|u_k^{*}\|=1$:
$|u_k^{*\top}E'_k u_k^{*}|\le\|E'_k\|_2\le\|E'_k\|_F$. Поэтому из~\eqref{eq:bd_frob_exp}:
\begin{equation}\label{eq:bd_step2}
  \|E_{k+1}\|_F^{\,2}
  \;\le\; \|E'_k\|_F^{\,2}
       \;+\; 2|\sigma_k^{*}|\,\|E'_k\|_F
       \;+\; (\sigma_k^{*})^2.
\end{equation}
Объединяя с~базовым~\eqref{eq:bd_base} (тождество~\eqref{eq:bd_skk}
$E_{k+1}s_k = E'_k s_k$ удерживается как~отдельный инвариант,
используемый ниже при~переходе к~условию Денниса--Море):
\begin{equation}\label{eq:bd_comb}
  \|E_{k+1}\|_F^{\,2}
  \;\le\; \|E_k\|_F^{\,2}
       \;-\; \frac{\|E_k s_k\|^2}{\|s_k\|^2}
       \;+\; \underbrace{c_{\mathcal{U}}L_H\alpha_k}_{=:\beta_k^{(\mathcal{U})}}
       \;+\; \underbrace{2|\sigma_k^{*}|\,\|E'_k\|_F + (\sigma_k^{*})^2}_{=:\beta_k^{(SS)}}.
\end{equation}
Полный композитный $\beta_k:=\beta_k^{(\mathcal{U})}+\beta_k^{(SS)}\ge 0$.

\medskip
\textbf{Шаг 3 (оценка $|\sigma_k^{*}|$ через $\|R'_k\|_F$).}
По формуле~\eqref{eq:ss_sigma} и~Коши--Шварцу:
\[
  |\sigma_k^{*}|
  \;=\; \frac{|u_k^{*\top}\,R'_k\,S_p^{\top}u_k^{*}|}{\|S_p^{\top}u_k^{*}\|^2}
  \;\le\; \frac{\|R'_k\|_2\,\|S_p^{\top}u_k^{*}\|}{\|S_p^{\top}u_k^{*}\|^2}
  \;\le\; \frac{\|R'_k\|_F}{\|S_p^{\top}u_k^{*}\|}.
\]
Применяя safeguard~(A3) $\|S_p^{\top}u_k^{*}\|^2\ge\varepsilon_S$:
\begin{equation}\label{eq:bd_sigma_R}
  |\sigma_k^{*}|^2 \;\le\; \varepsilon_S^{-1}\,\|R'_k\|_F^{\,2}.
\end{equation}

\medskip
\textbf{Шаг 4 (оценка $\|R'_k\|_F$ через $\|E'_k\|_F$ и~Lipschitz).}
Пусть $e_i\in\RR^{p+1}$~--- $i$-й канонический базисный вектор;
тогда $i$-й столбец $R'_k$ ($i=0,1,\dots,p$) имеет вид
\[
  R'_k e_i \;=\; B'_k s_{k-i} - y_{k-i}
          \;=\; E'_k s_{k-i} \;-\; q_{k-i},
  \qquad q_{k-i}\,:=\,y_{k-i} - J^{*}\,s_{k-i}.
\]
По интегральной форме теоремы о~среднем и~(A1):
$y_{k-i} = \bigl(\int_0^1 \nabla^2 f(x_{k-i}+t s_{k-i})\,dt\bigr)\, s_{k-i}$,
откуда
\[
  \|q_{k-i}\|
   \;=\; \biggl\|\int_0^1 \bigl[\nabla^2 f(x_{k-i}+t s_{k-i}) - J^{*}\bigr]\,dt\;\cdot\,s_{k-i}\biggr\|
   \;\le\; L_H\,\bar\alpha_{k-i}\,\|s_{k-i}\|,
\]
где $\bar\alpha_{k-i}\,:=\,\max_{t\in[0,1]}\|x_{k-i}+t s_{k-i}-x^{*}\|
\le\max(\|x_{k-i}-x^{*}\|,\|x_{k-i+1}-x^{*}\|)\le D_k$,
$D_k:=\max_{j\in W_k}\|x_j-x^*\|$, $W_k:=\{k-p,\dots,k+1\}$.
Поэтому
\[
  \|R'_k e_i\|^2
  \;\le\; 2\,\|E'_k\|_2^{\,2}\|s_{k-i}\|^2
       + 2\,L_H^2\, D_k^2\,\|s_{k-i}\|^2,
\]
и~суммируя по~$i$:
\begin{equation}\label{eq:bd_R_bound}
  \|R'_k\|_F^{\,2}
  \;=\; \sum_{i=0}^p \|R'_k e_i\|^2
  \;\le\; 2\,\|S_p\|_F^{\,2}\bigl(\|E'_k\|_F^{\,2} + L_H^2\,D_k^2\bigr).
\end{equation}

\medskip
\textbf{Шаг 5 (явная асимптотика $\beta_k$ под (A4)).}
Подставляя~\eqref{eq:bd_R_bound} в~\eqref{eq:bd_sigma_R}:
\[
  (\sigma_k^{*})^2 \;\le\; 2\varepsilon_S^{-1}\,\|S_p\|_F^{\,2}\,
                          \bigl(\|E'_k\|_F^{\,2} + L_H^2 D_k^2\bigr).
\]
Для смешанного члена $2|\sigma_k^{*}|\|E'_k\|_F$ применяем
$|\sigma_k^{*}|\le\varepsilon_S^{-1/2}\sqrt{2}\,\|S_p\|_F\,(\|E'_k\|_F+L_H D_k)$
и~AM--GM на~смешанный остаток $L_H D_k\|E'_k\|_F\le\tfrac12(L_H^2 D_k^2 + \|E'_k\|_F^2)$:
\[
  2|\sigma_k^{*}|\|E'_k\|_F
  \;\le\; 2\sqrt{2}\,\varepsilon_S^{-1/2}\,\|S_p\|_F\bigl(\|E'_k\|_F^{\,2}
                                                   + L_H D_k\|E'_k\|_F\bigr)
  \;\le\; 3\sqrt{2}\,\varepsilon_S^{-1/2}\,\|S_p\|_F\|E'_k\|_F^{\,2}
       \;+\; \sqrt{2}\,\varepsilon_S^{-1/2}\,L_H^2\,\|S_p\|_F\,D_k^2.
\]
Складывая:
\begin{equation}\label{eq:bd_beta_decomp}
  \beta_k^{(SS)}
  \;\le\; C_E(\|S_p\|_F)\,\|E'_k\|_F^{\,2}
       \;+\; C_D(\|S_p\|_F)\,D_k^2,
\end{equation}
с~
\[
  C_E(\xi)\,:=\,3\sqrt{2}\,\varepsilon_S^{-1/2}\,\xi + 2\varepsilon_S^{-1}\,\xi^2,
  \qquad
  C_D(\xi)\,:=\,L_H^2\bigl(\sqrt{2}\,\varepsilon_S^{-1/2}\,\xi + 2\varepsilon_S^{-1}\,\xi^2\bigr).
\]

\medskip
\textbf{Шаг 6 (суммируемость при Q-линейном темпе~$\rho_*$).}
Шаги 6 и~7 проводятся совместно: пусть $\rho_*\in[0,1)$~--- темп, который
будет установлен индуктивно в~Шаге~7. По~индуктивной гипотезе Q-линейности
(I.b) Шага~7 при $j=k_0,\dots,k$:
$\|x_j-x^{*}\|\le\rho_*^{j-k_0}\,\|x_{k_0}-x^{*}\|$, поэтому
\[
  \|s_j\| \;\le\; \|x_{j+1}-x^{*}\|+\|x_j-x^{*}\| \;\le\; (1+\rho_*)\,\rho_*^{j-k_0}\,\|x_{k_0}-x^{*}\|.
\]
Отсюда:
\begin{align}
  \alpha_k &\;\le\; \rho_*^{k-k_0}\,\|x_{k_0}-x^{*}\|, \notag\\
  D_k      &\;=\; \max_{j\in W_k}\|x_j-x^{*}\| \;\le\; \rho_*^{k-p-k_0}\,\|x_{k_0}-x^{*}\|, \notag\\
  \|S_p^{(k)}\|_F^{\,2}
           &\;=\; \sum_{i=0}^p \|s_{k-i}\|^2
            \;\le\; (1+\rho_*)^2\,\|x_{k_0}-x^{*}\|^2 \sum_{i=0}^p \rho_*^{2(k-i-k_0)}
            \;\le\; C_{\rho_*}\,\rho_*^{2(k-p-k_0)}\,\|x_{k_0}-x^{*}\|^2, \label{eq:bd_Sp_geom}
\end{align}
где $C_{\rho_*}\,:=\,(1+\rho_*)^2/(1-\rho_*^2) = (1+\rho_*)/(1-\rho_*)$ зависит только
от~$\rho_*$ (не~от~$p$): сумма $\sum_{i=0}^p\rho_*^{2(p-i)}\le 1/(1-\rho_*^2)$
оценивается равномерно по~$p$. Все три величины убывают геометрически
со~скоростью $\rho_*^{k}$, $\rho_*^{k-p}$, $\rho_*^{k-p}$ соответственно.

\medskip
\textbf{Шаг 7 (совместная индукция: ограниченность~$\|E_k\|_F$ и~Q-линейность).}
Положим $M_{k_0}\,:=\,\|E_{k_0}\|_F$, $M\,:=\,2M_{k_0}$ и~зафиксируем кандидат
\begin{equation}\label{eq:bd_rho_star}
  \rho_*
   \;:=\; 2\,\|(J^{*})^{-1}\|_2\,M
        \;+\; \|(J^{*})^{-1}\|_2\,L_H\,\delta''.
\end{equation}
Покажем, что при~достаточно малых $\delta',\delta''>0$ одновременно выполняются
два инварианта для всех $k\ge k_0$:
\begin{enumerate}
  \item[\textup{(I.a)}] $\|E_k\|_F \le M$;
  \item[\textup{(I.b)}] $\|x_{k+1}-x^{*}\| \le \rho_*\,\|x_k-x^{*}\|$ с~$\rho_*<1$.
\end{enumerate}
Параметры $\delta',\delta''$ подчиним двум условиям:
\begin{equation}\label{eq:bd_delta_choice}
  \delta' \;<\; \tfrac{1}{4}\|(J^{*})^{-1}\|_2^{-1},
  \qquad
  4\,\|(J^{*})^{-1}\|_2\,(\delta' + \tfrac14 L_H\,\delta'') \;<\; 1,
\end{equation}
что обеспечивает $\rho_*<1$ (поскольку $M=2M_{k_0}<2\delta'$), и~ещё одно
условие на~малость~$\Sigma$ (см.~ниже~\eqref{eq:bd_delta_Sigma}).

\textit{База ($k=k_0$).} (I.a) выполнена по~определению~$M$. Для~(I.b):
из~$\|E_{k_0}\|_F<\delta'<\tfrac12\|(J^{*})^{-1}\|_2^{-1}$ лемма Банаха
о~возмущении обратимого оператора даёт $\|B_{k_0}^{-1}\|_2\le 2\|(J^{*})^{-1}\|_2$.
Раскладывая $\nabla f(x_{k_0}) = J^{*}(x_{k_0}-x^{*}) + r_{k_0}$ с~$\|r_{k_0}\|\le\tfrac12 L_H\|x_{k_0}-x^{*}\|^2$
(из~(A1) через интегральную форму остатка Тейлора), получаем
\[
  x_{k_0+1}-x^{*}
  \;=\; (I - B_{k_0}^{-1}J^{*})(x_{k_0}-x^{*})
        \;-\; B_{k_0}^{-1}\,r_{k_0},
\]
и~$\|I - B_{k_0}^{-1}J^{*}\|_2 = \|B_{k_0}^{-1}E_{k_0}\|_2 \le 2\|(J^{*})^{-1}\|_2 M$.
Отсюда $\|x_{k_0+1}-x^{*}\|\le\rho_*\,\|x_{k_0}-x^{*}\|$ при $\rho_*$ из~\eqref{eq:bd_rho_star},
причём $\rho_*<1$ по~выбору~\eqref{eq:bd_delta_choice}.

\textit{Индуктивный шаг.} Пусть (I.a),~(I.b) выполнены для всех $j=k_0,\dots,k$.
Тогда телескопически $\|x_j-x^{*}\|\le\rho_*^{j-k_0}\,\|x_{k_0}-x^{*}\|$, и~Шаги~5--6
с~параметром~$\rho_*$ дают (с~$\xi_j\,:=\,\|S_p^{(j)}\|_F\le\sqrt{C_{\rho_*}}\,\rho_*^{j-p-k_0}\,\|x_{k_0}-x^{*}\|$):
из~\eqref{eq:bd_base} $\|E'_j\|_F\le M':=\sqrt{M^2+c_{\mathcal{U}}L_H\|x_{k_0}-x^{*}\|}$,
и~\eqref{eq:bd_beta_decomp} приводит к~оценке
\[
  \beta_j
   \;\le\; c_{\mathcal{U}}L_H\,\alpha_j
        \;+\; (3\sqrt{2}\,\varepsilon_S^{-1/2}\xi_j + 2\varepsilon_S^{-1}\xi_j^2)\,(M')^2
        \;+\; L_H^2\,(\sqrt{2}\,\varepsilon_S^{-1/2}\xi_j + 2\varepsilon_S^{-1}\xi_j^2)\,D_j^2.
\]
Доминирующие слагаемые: $\xi_j (M')^2 = O(\rho_*^{j})$ и~$\xi_j D_j^2 = O(\rho_*^{3j})$
(квадратичные $\xi_j^2(\cdot)$ ещё быстрее); все~--- геометрические со~знаменателем $\rho_*<1$,
суммируются:
\begin{equation}\label{eq:bd_Sigma}
  \Sigma
  \;:=\; \sum_{j\ge k_0}\beta_j
  \;\le\; \frac{c_{\mathcal{U}}L_H\,\|x_{k_0}-x^{*}\|}{1-\rho_*}
       \;+\; A_1(\rho_*,p,\varepsilon_S,L_H,\|x_{k_0}-x^{*}\|)\,(M')^2
       \;+\; A_2(\rho_*,p,\varepsilon_S,L_H,\|x_{k_0}-x^{*}\|),
\end{equation}
где~$A_1,A_2$~--- $O(\|x_{k_0}-x^{*}\|)$ при~$\|x_{k_0}-x^{*}\|\to 0$.

\textit{Замыкание (I.a) на шаге $k+1$.} Телескопия~\eqref{eq:bd_comb} от~$k_0$ до~$k$ даёт
$\|E_{k+1}\|_F^{\,2}\le\|E_{k_0}\|_F^{\,2}+\sum_{j=k_0}^{k}\beta_j\le M_{k_0}^2+\Sigma$.
Для $\|E_{k+1}\|_F\le M=2M_{k_0}$ достаточно
\begin{equation}\label{eq:bd_delta_Sigma}
  \Sigma\;\le\;3M_{k_0}^2.
\end{equation}
Из~оценки~\eqref{eq:bd_Sigma} и~явных выражений для~$A_1,A_2$ из~Шага~6
следует $\Sigma\le K\,\|x_{k_0}-x^{*}\|\,(1+M_{k_0}^2)$ с~константой
$K=K(L_H,\|J^{*}\|_2,\|(J^{*})^{-1}\|_2,\varepsilon_S,p,c_{\mathcal U},\rho_*)$,
зависящей только от~параметров задачи (не~от~старта).
Подставляя~$M_{k_0}<\delta'$ и~$\|x_{k_0}-x^{*}\|<\delta''$, получаем
$\Sigma\le K\,\delta''\,(1+\delta'^{\,2})$; для выполнения~\eqref{eq:bd_delta_Sigma}
при~всех допустимых~$M_{k_0}\le\delta'$ достаточно, чтобы $\delta'',\delta'$
удовлетворяли соотношению
\begin{equation}\label{eq:bd_delta_link}
  K\,\delta''\,(1+\delta'^{\,2})\;\le\;3\,\delta'^{\,2}.
\end{equation}
Это связывает~$\delta''$ с~$\delta'$: например, выбор~$\delta'\le\tfrac14\|(J^{*})^{-1}\|_2^{-1}$
из~\eqref{eq:bd_delta_choice} и~$\delta''\le 3\delta'^{\,2}/(K(1+\delta'^{\,2}))$
обеспечивает~\eqref{eq:bd_delta_Sigma} вместе с~\eqref{eq:bd_delta_choice}
одновременно при~$M_{k_0}<\delta'$, $\|x_{k_0}-x^{*}\|<\delta''$.
В~этом смысле радиус~$\delta''$ \emph{масштабируется как~$\delta'^{\,2}$};
такая связь типична для локального анализа квазиньютоновских методов
(ср.\ Деннис--Шнабель~\cite[Th.~8.2.2]{dennis1996}).

\textit{Замыкание (I.b) на шаге $k+1$.} Из~(I.a) на шаге $k+1$: $\|E_{k+1}\|_F\le M<2\delta'<\tfrac12\|(J^{*})^{-1}\|_2^{-1}$,
поэтому $\|B_{k+1}^{-1}\|_2\le 2\|(J^{*})^{-1}\|_2$. Тот~же~аргумент Тейлора
с~$x_{k+1}$ вместо $x_{k_0}$ (используя $\|x_{k+1}-x^{*}\|\le\rho_*^{k+1-k_0}\,\|x_{k_0}-x^{*}\|<\delta''$)
и~$\|I-B_{k+1}^{-1}J^{*}\|_2\le 2\|(J^{*})^{-1}\|_2 M$ дают
$\|x_{k+2}-x^{*}\|\le\rho_*\,\|x_{k+1}-x^{*}\|$. Индукция замыкается.

\smallskip
Таким образом, (I.a) обеспечивает~\eqref{eq:bd_composite}, а~(I.b)~--- Q-линейную
сходимость $\{x_k\}$ с~темпом~$\rho_*\in[0,1)$ как~следствие, не~как~входное
предположение.
\end{proof}


\begin{remark}[Об условии~(A4) как структурной гипотезе локальной близости]\label{rem:bd_A4_origin}
Условие~(A4) формулируется как требование локальной близости старта
($\|E_{k_0}\|_F<\delta'$, $\|x_{k_0}-x^{*}\|<\delta''$), а~не~как предположение
о~Q-линейной сходимости итерации. Q-линейность с~темпом~$\rho_*\in[0,1)$~---
\emph{выходное свойство} Леммы~\ref{lem:bd_composite}, устанавливаемое
совместной индукцией Шага~7 одновременно с~ограниченностью~$\|E_k\|_F$.

Механизм совместной индукции~--- стандартный возмущенный аргумент локального
анализа квазиньютоновских методов: при~малом $\|E_k\|_F\le M$ лемма Банаха
даёт $\|B_k^{-1}\|_2\le 2\|(J^{*})^{-1}\|_2$, разложение $\nabla f$ вокруг $x^{*}$
с~Lipschitz-остатком (из~(A1)) даёт сжатие $\|x_{k+1}-x^{*}\|\le\rho_*\|x_k-x^{*}\|$
с~$\rho_*= 2\|(J^{*})^{-1}\|_2 M + \|(J^{*})^{-1}\|_2 L_H\delta''$
(ср.\ Бройден--Деннис--Море~\cite[Th.~3.1]{broyden1973}
или~Деннис--Шнабель~\cite[Th.~5.4.1]{dennis1996}). Композитная BD-оценка,
в~свою очередь, при~$\rho_*$-геометрии $\|s_k\|$ даёт $\sum\beta_k<\infty$,
сохраняя $\|E_{k+1}\|_F\le M$.

Совокупный результат: SS-$\mathcal{U}$ \emph{локально} Q-линейна в~окрестности
$x^{*}$ при условии достаточной близости $\|E_{k_0}\|_F<\delta'$,
$\|x_{k_0}-x^{*}\|<\delta''$; темп~$\rho_*$ и~радиусы~$\delta',\delta''$ явно
выражены в~Шаге~7 через~$L_H,\|J^{*}\|_2,\|(J^{*})^{-1}\|_2,\varepsilon_S,
c_{\mathcal U},p$.
\end{remark}


\begin{remark}[Замыкание (H5) Теоремы~\ref{thm:ss_dm}]\label{rem:bd_in_thm}
Лемма~\ref{lem:bd_composite} закрывает аргумент сохранения~(H5) под
SS-коррекцией строгой цепочкой:
\emph{(А1)+(А2)+(А3)+(A4)} $\Longrightarrow$ \emph{(H5) в~композитной форме}
$\stackrel{\text{telescoping}}{\Longrightarrow}$ \emph{Денниса--Море для~$B_k$}
$\stackrel{\text{характеризация}}{\Longrightarrow}$ \emph{Q-сверхлинейность}. В~совместной
формулировке Леммы~\ref{lem:bd_composite} и~Теоремы~\ref{thm:ss_dm} условие~(H5)
перестаёт быть гипотезой~--- вместо него выступают~(А2) [базовое BD]
и~(А4) [локальная близость старта]; гипотеза~(H1)~--- Lipschitz $\nabla^2 f$~---
сохраняется для применения Шага~4 доказательства Леммы (оценка~$\|q_{k-i}\|$
через~$L_H$); гипотезы~(H3),~(H4) остаются без изменений. Q-линейная сходимость
итерации с~некоторым $\rho_*<1$~--- выходное свойство Леммы (см.~Замечание~\ref{rem:bd_A4_origin}),
устанавливаемое совместной индукцией одновременно с~ограниченностью~$\|E_k\|_F$;
тем самым снимается кажущаяся цикличность~``Q-линейность~$\to$~Q-сверхлинейность''
и~весь аргумент Теоремы~\ref{thm:ss_dm} держится на~одном структурном допущении~(A4)
о~локальной близости старта.
\end{remark}


\begin{remark}[Связь с~SS-инвариантом теоремы~\ref{thm:ss_props}]\label{rem:bd_to_rpast}
Хотя теорема~\ref{thm:ss_props}~п.~3 доказывает \emph{строгое} равенство
$\|R_{k+1}\|_F^2 = \|R'_k\|_F^2 - (\sigma_k^*)^2\,\|S_p^\top u_k^*\|^2$~--- структурное
сокращение накопленной невязки на~одном и~том же~окне~$S_p$,~--- в~Лемме~\ref{lem:bd_composite}
эта монотонность не~используется. Шагу~4 достаточно одностороннего верхнего
ограничителя~\eqref{eq:bd_R_bound}, который выражается через $\|E'_k\|_F$ и~$D_k$
без~ссылки на~SS-механизм; так разделяются два контура: геометрия сокращения
невязки (п.~3 теоремы) и~передача bounded deterioration (Лемма).
Эмпирическая роль монотонности отражена на~рис.~\ref{fig:ndim_stat_rpast_psb}
как прямая верификация того, что MS-коррекция действительно ограничивает
$\Norm{B_k S_p - Y_p}_F$ в~численных режимах.
\end{remark}


\begin{remark}[Минимальность гипотез]\label{rem:bd_minimality}
Гипотеза~(A4) о~локальной близости старта~--- \emph{достаточная} для
замыкания совместной индукции Шага~7 и~извлечения Q-линейного темпа~$\rho_*<1$.
Альтернативная посылка «$\sum\|s_k\|^2<\infty$» (что эмпирически наблюдается
в~численных экспериментах §~\ref{sec:ss_numerics}) дала бы тот~же~вывод
суммируемости $\sum\beta_k<\infty$ без~возмущенного аргумента; однако
формулировка через~$(\delta',\delta'')$ согласуется со~стандартным циклом
локального анализа квазиньютоновских методов и~явно фиксирует область применимости.

Гипотеза~(A1) о~Липшице гессиана~--- \emph{использована} в~Шаге~4
(оценка $\|q_{k-i}\|$) и~в~Шаге~7 (оценка остатка Тейлора $r_k$);
ослабление до~«$\nabla^2 f$ непрерывен в~$x^{*}$» даёт~$\|q_{k-i}\|=o(\|s_{k-i}\|)$
вместо $O(\|s_{k-i}\|\cdot D_k)$, что нарушает геометрическое убывание в~Шаге~6
и~количественную оценку~$\rho_*$ в~Шаге~7. Поэтому~(A1) в~Lipschitz-форме
сохраняется как~\emph{рабочая}, а~не~излишняя гипотеза.
\end{remark}

\begin{remark}[Роль MS-коррекции: стабилизация без изменения асимптотической
  скорости]\label{rem:ss-rate-invariance}
MS-коррекция не~влияет на~сверхлинейную скорость: $\Delta_k^{SS}$ действует
в~$s_k^\perp$ (так как $u_k^*\perp s_k$), а~условие Денниса--Море
$\|(B_k-J^*)s_k\|/\|s_k\|\to 0$~--- только в~направлении~$s_k$. Роль
коррекции~--- стабилизация $B_k$ (условие~(H5) и~невырожденность $B_k$),
обеспечивающая устойчивую сходимость на~практике. Более того, доказательство
Th.~\ref{thm:ss_dm} использует от~SS-$\mathcal{U}$ только текущее секущее
условие $B_{k+1}s_k=y_k$, выполняемое и~базовым обновлением; поэтому
при выполнении~(H1)--(H5) тем же~аргументом Q-сверхлинейная скорость
следует и~для базового~SR1 (классический результат Конна--Гулда--Тойнта,
Халфана--Бёрда--Шнабеля). Узкое место в~этом случае~--- посылка~(H2)
(сходимость траектории) и~условие~(H5), ровно те~два пункта,
которые и~предназначена закрывать MS-коррекция.
\end{remark}

\begin{remark}[Следствие для дизайна численных экспериментов]\label{rem:ss-rate-empirical}
Поскольку асимптотическая скорость, заявленная в~Th.~\ref{thm:ss_dm},
разделяется с~базовым обновлением (Замечание~\ref{rem:ss-rate-invariance}),
\emph{отдельная} визуализация скорости в~гл.~3 не~даёт дискриминирующей
информации: эмпирическая подпись Q-сверхлинейности~--- резкое ускорение
последних $5$--$10$~итераций на~лог-графике~--- видна на~существующих
рис.~\ref{fig:ndim_stat_sr1} и~\ref{fig:ndim_stat_psb} для любой
сошедшейся траектории~(SR1, SS-SR1, PSB, SS-PSB). Дискриминатор
SS-конструкции~--- предпосылка~(H2), т.\,е.\ \emph{наличие} сходимости,
а~не~её~скорость; этот аспект закрыт существующими convergence-
и~rpast-фигурами.
\end{remark}

\section{Численные результаты}\label{sec:ss_numerics}

\paragraph{Постановка статистического эксперимента.}
\label{par:ssn-stat-setup}
Для общего случая ведущий собственный вектор $u_k^*$ матрицы $P_k M'_k P_k$
(см.\ Опр.~\ref{def:ss}) вычисляется через симметричную задачу на
собственные значения~(\texttt{numpy.linalg.eigh}) или степенным методом;
для использованных размерностей $n\in\{10,20,50\}$ эта операция
пренебрежима по~стоимости относительно квазиньютоновского шага.

Чтобы избежать артефактов одной траектории (где результат заметно
зависит от того, попал ли стартовый шаг в~«удачное» окно линейного
поиска), эксперимент построен по~paired-design схеме: для каждой
задачи генерируются $50$ единичных направлений
$\hat u_i\in S^{n-1}$ с фиксированным seed~$20\,260\,503$
(\texttt{numpy.random.default\_rng}), и~стартовая точка
$x_0^{(i)} = R_0\,\hat u_i$ берётся на~сфере радиуса
$R_0 = \Norm{x_0^{\mathrm{std}}}$, где $x_0^{\mathrm{std}}$~--- стандартная
стартовая точка задачи. Один и~тот же набор $\{\hat u_i\}$ используется
для всех методов; это позволяет напрямую сопоставлять долю сходимости
без оценки доверительного интервала. Все методы используют общий
backtracking-Armijo~($c_1=10^{-4}$, откат $\alpha/2$, до~$40$~откатов),
$B_0=I$, окно прошлых секущих $p=5$, критерий остановки
$\Norm{\nabla f}_2\le 10^{-8}$ или $500$~итераций; направление шага
$d_k = -B_k^{-1}\nabla f(x_k)$ через~LU-разложение системы.

Тестовый набор:
гладкие задачи MGH \emph{Rosenbrock~chained}~($n=10$,
Nocedal--Wright Eq.~2.22) и~\emph{Extended~Rosenbrock}~($n=10$, MGH~\#21)~---
контроль на~отсутствие деградации; задача с узкой искривлённой долиной
\emph{Extended~Curved~Valley}~(блочно по~$2$~координаты,
$f_{\text{блок}}(x_1,x_2) = \tfrac{\alpha}{2}(x_2-\beta x_1^2)^2 + \tfrac12(x_1^2+x_2^2)$,
$\alpha=100$, $\beta=1{,}5$, $x_0^{\mathrm{std}}$ блочно $=(1{,}8;\,1{,}8)$)~---
стресс-тест на~стабилизацию, прогоняемый при $n\in\{10,20,50\}$ для
оценки масштабируемости.

Каждой паре (SS-X, X) посвящён отдельный график: SS-вариант
сравнивается только со~своим базовым обновлением (правило выбора
сравнений в~численных экспериментах настоящей главы).
Ниже последовательно разбираются обе пары~--- SR1/SS-SR1 и~PSB/SS-PSB.

\paragraph{Прямая иллюстрация теоремы~\ref{thm:ss_props}, п.~3:
  динамика накопленной невязки на паре PSB\,/\,SS-PSB.}\label{par:ssn-rpast}
Convergence-кривые рис.~\ref{fig:ndim_stat_sr1}
и~\ref{fig:ndim_stat_psb} показывают итоговую долю сходимости
и~медианное число итераций, но~оставляют открытым прямой
вопрос: действительно ли MS-коррекция \emph{ограничивает} накопленную
невязку $\Norm{B_k S_p - Y_p}_F$ в~смысле теоремы~\ref{thm:ss_props}, п.~3?
Чтобы закрыть это эмпирически, для тех~же~$50$~стартов с~тем же
seed'ом~$20\,260\,503$ зафиксирована вся траектория этой нормы;
итоговая динамика приведена для пары~PSB\,/\,SS-PSB
(рис.~\ref{fig:ndim_stat_rpast_psb}; медиана жирной линией,
заливка~--- межквартильный размах $q_{25}$--$q_{75}$ по~$50$~стартам;
расходящиеся траектории включены через pad-by-last).

Пара PSB\,/\,SS-PSB выбрана как чистая иллюстрация по~двум причинам.
Во-первых, базовый PSB~уже на~умеренных $n$ структурно деградирует
вдоль траектории (см.\ рис.~\ref{fig:ndim_stat_psb}); это даёт
MS-коррекции достаточно \emph{материала}~для работы и~делает разрыв
$\Norm{B_k S_p - Y_p}_F$ визуально разрешимым. Во-вторых, для пары
SR1\,/\,SS-SR1 базовый~SR1 на~использованных режимах часто и~без
MS-коррекции удерживает $\Norm{B_k S_p - Y_p}_F$ на~приемлемом
уровне (что отражает ограничение самой теоремы~Th.~\ref{thm:ss_dm}:
MS-коррекция нужна там, где~SR1 \emph{реально}~деградирует, и~не
создаёт преимущества там, где уже не~деградирует), так что~прямая
rpast-картинка для этой пары читается хуже, чем косвенная сходимостная
метрика рис.~\ref{fig:ndim_stat_sr1}; пара оставлена иллюстрироваться
через~convergence на~рис.~\ref{fig:ndim_stat_sr1}.

Картина на~рис.~\ref{fig:ndim_stat_rpast_psb} согласуется
со~структурным предсказанием. На~всех пяти конфигурациях
SS-PSB системно держит медиану $\Norm{B_k S_p - Y_p}_F$ ниже,
чем~PSB; качественный разрыв особенно ясен на~режиме~ECV, где~PSB
теряет устойчивость, а~SS-PSB монотонно снижает невязку.
Это прямая эмпирическая поддержка утверждения~3~теоремы~\ref{thm:ss_props}:
оценка
$\Norm{R_{k+1}}_F^2 \le \Norm{R'_k}_F^2 -
(u_k^{*\top} M'_k u_k^*)^2/\Norm{S_p^{\top} u_k^*}^2$
выполнена по~построению на~каждом шаге; интегральный эффект~---
системно меньшая медиана $\Norm{B_k S_p - Y_p}_F$~--- наблюдается
именно там, где базовое обновление действительно теряет устойчивость.

\paragraph{Изолированная роль гипотезы~(H4): эксперименты без~Armijo.}
\label{par:ssn-noarmijo}
Теорема~\ref{thm:ss_dm} формулирует Q-сверхлинейность при~условии~(H4):
после некоторого~$k_0$ глобализация выдаёт полный QN-шаг
$s_k=-B_k^{-1}\nabla f(x_k)$. В~основных экспериментах
\S\ref{par:ssn-stat-setup} (H4) обеспечивается backtracking-Армихо и
отдельно не визуализирована. Чтобы изолировать её роль, повторим
paired-дизайн (тот же~seed~$20\,260\,503$, $50$~стартов) с~полным шагом
$x_{k+1}=x_k+d_k$ \emph{без} backtracking, на~двух типах задач:
выпуклой квадратике, где~(H4) выполнено тривиально, и~Rosenbrock chained,
где~(H4) нетривиально.

\emph{Квадратика} $f(x)=\frac12 x^{\top} A x$,
$A=Q\,\mathrm{diag}(1,\,\ldots,\,\kappa)\,Q^{\top}$, $Q\in O(n)$, $x^*=0$,
$\|x_0\|=R$. На~рис.~\ref{fig:ss_local_quadratic} приведены
два режима: $(n=20,\,\kappa=10^3)$~--- оба метода сходятся, и
$(n=50,\,\kappa=10^2)$~--- PSB \emph{не} сходится за~отведённый бюджет
итераций, а~SS-PSB сходится. SR1-семейство из~рисунка опущено: по
рамarka~\ref{rem:ss-rate-invariance} асимптотическая Q-сверхлинейная
скорость SS совпадает с~базовой, а~на~квадратике SR1 со skip-rule имеет
конечную сходимость за~$\le n+1$~шагов~\cite{conn1991}, так что~для
SR1-пары MS-добавке нечего улучшать.

Картина одинакова для радиусов $R\in[0.1,\,3]$ (рисунок построен
при~$R=0.3$): на~локальной выпуклой квадратике поведение определяется
парой $(n,\kappa)$, не~радиусом старта. На~режиме
$(n=20,\,\kappa=10^3)$ SS-PSB видимо ускоряет пре-асимптотическую фазу
PSB; с~ростом~$n$ при той~же~$\kappa$ PSB перестаёт сходиться раньше,
чем SS-PSB ($(n=50,\,\kappa=10^2)$~--- правая панель). Это
эмпирическая поддержка теоремы~\ref{thm:ss_dm} в~режиме, где~(H4)
выполнено тривиально: SS-коррекция \emph{не} меняет асимптотическую
скорость (замечание~\ref{rem:ss-rate-invariance}), но~улучшает
пре-асимптотический режим за~счёт лучшего приближения $\nabla^2 f(x^*)$
матрицей~$B_k$ на~окне прошлых секущих и~расширяет диапазон $(n,\kappa)$,
в~котором базовое обновление вообще достигает сходимости без~line search.

\emph{Rosenbrock chained}, $n=10$, $x^*=\mathbf 1$, $50$~стартов

а~лишь расширяет локальную область устойчивой сходимости (особенно
у~PSB-семейства). Глобализация в~теореме~\ref{thm:ss_dm}~--- содержательное
предположение, не~факультативная инженерная деталь.

\paragraph{Резюме.}\label{par:ssn-summary}
На SR1 и~PSB SS-коррекция эмпирически подтверждает роль ограничителя
накопленной невязки $\Norm{B_k S_p - Y_p}_F$ (теорема~\ref{thm:ss_props},
п.~3, и~Лемма~\ref{lem:bd_composite}): на~рис.~\ref{fig:ndim_stat_rpast_psb}
медиана $\Norm{B_k S_p - Y_p}_F$ у~SS-PSB системно ниже PSB, а~на~ECV~$n=20$
PSB не~сходится, тогда как SS-PSB монотонно снижает невязку. Эксперименты
без~backtracking (\S\ref{par:ssn-noarmijo}) подтверждают также нетривиальность
гипотезы~(H4) Теоремы~\ref{thm:ss_dm}: на~локально-выпуклой квадратике
SS-PSB укорачивает пре-асимптотическую фазу PSB; на~Rosenbrock chained полный
QN-шаг расходится даже из~окрестности~$x^*$, и~MS-коррекция расширяет лишь
область устойчивости, не~подменяя глобализацию. Дальнейшие направления~---
формальная глобализация и~перенос на~негладкий случай~--- обсуждаются
в~Заключении.
