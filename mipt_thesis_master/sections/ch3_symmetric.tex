\section{Секантно-стабилизированные симметричные методы}\label{sec:ss}

\subsection{Мотивация}\label{sec:ss_motivation}

В предыдущих разделах метод SP-Broyden решил задачу сохранения секущих
условий для общих систем $F(x) = 0$, однако матрица~$B_k$ при этом
несимметрична. Для задач безусловной оптимизации, где $F = \nabla f$ и
$J^* = \nabla^2 f(x^*)$ симметричен, естественно требовать $B_k = B_k^\top$.

Классические симметричные методы~--- SR1, BFGS, PSB~---
удовлетворяют текущему секущему условию $B_{k+1} s_k = y_k$,
но не контролируют невязку прошлых секущих
$R_{\mathrm{past}} = B_k S_{\mathrm{past}} - Y_{\mathrm{past}}$.
Следующая теорема показывает, что одновременное точное выполнение
всех секущих и симметрии невозможно.

\begin{theorem}[Несовместность]\label{thm:incompat}
Пусть $\Delta B = \Delta B^\top$ и $\Delta B\, s_j = 0$ для всех
$j < k$, где $\{s_j\}$~--- прошлые шаги. Тогда
\begin{equation}\label{eq:incompat}
  s_j^\top (\Delta B\, s_k) = (\Delta B\, s_j)^\top s_k = 0
  \quad \forall\, j < k.
\end{equation}
Следовательно, секущее условие $\Delta B\, s_k = r_k$
совместно с~\eqref{eq:incompat} тогда и только тогда, когда
$s_j^\top r_k = 0$ для всех~$j < k$.
\end{theorem}

\begin{proof}
Из $\Delta B = \Delta B^\top$ и $\Delta B\, s_j = 0$ следует
$s_j^\top \Delta B = (\Delta B\, s_j)^\top = 0$.
Применяя к~$s_k$: $s_j^\top \Delta B\, s_k = 0$.
\end{proof}

\begin{remark}
Для $F = \nabla f$ дефект совместности оценивается через разность
средних гессианов:
\[
  s_j^\top r_k = s_j^\top (\bar{H}_k - \bar{H}_j)\, s_k
  = O(L_H \cdot \|x_k - x_j\| \cdot \|s_j\| \cdot \|s_k\|),
\]
где $\bar{H}_i = \int_0^1 \nabla^2 f(x_i + t\, s_i)\, dt$ и $L_H$~---
константа Липшица гессиана. Дефект мал вблизи~$x^*$, но не равен нулю.
На квадратичной функции $\bar{H}_k = \bar{H}_j$, и совместность
выполнена тождественно.
\end{remark}

\subsection{Конструкция SS-\texorpdfstring{$\mathcal{U}$}{U}}
\label{sec:ss_construction}

\begin{definition}[Секантно-стабилизированное обновление]
\label{def:ss}
Пусть $\mathcal{U}$~--- базовое симметричное квазиньютоновское
обновление, удовлетворяющее текущему секущему условию
$\mathcal{U}(B_k, s_k, y_k)\, s_k = y_k$.
\emph{Секантно-стабилизированное расширение}
$\mathrm{SS}\text{-}\mathcal{U}$:
\begin{equation}\label{eq:ss_update}
  B_{k+1} = \underbrace{\mathcal{U}(B_k, s_k, y_k)}_{\displaystyle B'_k}
  + \sigma_k^*\, u_k^*\, u_k^{*\top},
  \qquad u_k^* \perp s_k,\;\; \|u_k^*\| = 1,
\end{equation}
где $u_k^*$~--- ведущий собственный вектор матрицы
$P_k M'_k P_k$,
\[
  M'_k = \tfrac{1}{2}(R'_k S_p^\top + S_p R_k^{\prime\top}),
  \qquad
  R'_k = B'_k S_p - Y_p,
  \qquad
  P_k = I - \hat{s}_k \hat{s}_k^\top,
\]
а~$\sigma_k^*$ выбирается из минимума $\|R'_k + \sigma\, u^*\, u^{*\top} S_p\|_F^2$:
\begin{equation}\label{eq:ss_sigma}
  \sigma_k^* = -\frac{u_k^{*\top} R'_k S_p^\top u_k^*}
                     {\|S_p^\top u_k^*\|^2}.
\end{equation}
\end{definition}

\begin{remark}[Частные случаи]
При $\mathcal{U} = \mathrm{SR1}$ получаем \emph{SS-SR1}~---
основной метод данного раздела. Выбор $\mathcal{U} = \mathrm{PSB}$
или $\mathcal{U} = \mathrm{BFGS}$ даёт SS-PSB и SS-BFGS
соответственно. Коррекция применима к произвольному базовому методу.
\end{remark}

\begin{remark}[Связь с SP-Broyden]
SP-Broyden и SS-$\mathcal{U}$ решают одну задачу~--- контроль
деградации прошлых секущих~--- разными механизмами.
SP-Broyden модифицирует направление обновления ($s_k \to v_k^*$),
обеспечивая точные прошлые секущие ценой потери симметрии.
SS-$\mathcal{U}$ добавляет коррекцию в~$s_k^\perp$, обеспечивая
симметрию ценой приближённых прошлых секущих.
\end{remark}


% ============================================================
\subsection{Свойства}\label{sec:ss_properties}

\begin{theorem}[Свойства SS-$\mathcal{U}$]\label{thm:ss_props}
Обновление~\eqref{eq:ss_update} обладает следующими свойствами:
\begin{enumerate}
  \item \textbf{Текущее секущее условие:}
    $B_{k+1} s_k = y_k$ (точно).
  \item \textbf{Симметрия:}
    если $\mathcal{U}$ сохраняет симметрию, то $B_{k+1} = B_{k+1}^\top$.
  \item \textbf{Монотонность:}
    $\|R_{\mathrm{past}}^{(k+1)}\|_F^2
    \leq \|R_{\mathrm{past}}'\|_F^2
    - \dfrac{(u_k^{*\top} M'_k\, u_k^*)^2}{\|S_p^\top u_k^*\|^2}$.
  \item \textbf{Ранг обновления:}
    $\mathrm{rank}(B_{k+1} - B_k) \leq \mathrm{rank}(\mathcal{U}) + 1$.
\end{enumerate}
\end{theorem}

\begin{proof}\leavevmode
\begin{enumerate}
  \item Из $u_k^* \perp s_k$:
    $B_{k+1} s_k = B'_k s_k + \sigma_k^*(u_k^{*\top} s_k)\, u_k^* = y_k + 0 = y_k$.
  \item $B_{k+1} = B'_k + \sigma_k^*\, u_k^*\, u_k^{*\top}$~---
    сумма двух симметричных матриц.
  \item Пусть $\tilde{B} = B'_k + \sigma\, u u^\top$. Тогда
    $\|\tilde{B} S_p - Y_p\|_F^2
    = \|R'\|_F^2 + 2\sigma\,(u^\top M' u) + \sigma^2\,\|S_p^\top u\|^2$.
    Минимум по~$\sigma$ при~\eqref{eq:ss_sigma} даёт убывание на
    $(u^\top M' u)^2 / \|S_p^\top u\|^2$.
  \item Очевидно из конструкции. \qedhere
\end{enumerate}
\end{proof}

\subsection{Сверхлинейная сходимость}\label{sec:ss_superlinear}

\begin{theorem}[Условие Денниса--Море для SS-$\mathcal{U}$]
\label{thm:ss_dm}
Пусть $F = \nabla f$, $\nabla^2 f$ удовлетворяет условию Липшица
с~константой~$L_H$, и метод SS-$\mathcal{U}$ сходится: $x_k \to x^*$.
Тогда
\[
  \frac{\|(B_k - J^*)\, s_k\|}{\|s_k\|}
  = \frac{\|y_k - J^*\, s_k\|}{\|s_k\|}
  = O(\|x_k - x^*\|) \to 0,
\]
т.\,е. условие Денниса--Море выполнено и сходимость сверхлинейна.
\end{theorem}

\begin{proof}
Из точного секущего условия $B_{k+1} s_k = y_k$ (Теорема~\ref{thm:ss_props}.1):
\[
  (B_{k+1} - J^*)\, s_k = y_k - J^*\, s_k
  = \int_0^1 [\nabla^2 f(x_k + t\, s_k) - \nabla^2 f(x^*)]\, s_k\, dt.
\]
По условию Липшица
$\|\nabla^2 f(x_k + t\, s_k) - \nabla^2 f(x^*)\| \leq L_H(\|x_k - x^*\| + \|s_k\|)$,
откуда
$\|(B_{k+1} - J^*) s_k\| \leq L_H(\|x_k - x^*\| + \|s_k\|)\,\|s_k\|$.
Деление на~$\|s_k\|$ и $\|s_k\| \to 0$ завершают доказательство.
\end{proof}

\begin{remark}
MS-коррекция не влияет на сверхлинейную скорость
(она действует в~$s_k^\perp$, а условие Денниса--Море~---
только в~направлении~$s_k$). Роль коррекции~--- стабилизация~$B_k$,
расширяющая практический бассейн сходимости.
\end{remark}

\subsection{Модельная задача: Curved Valley}\label{sec:ss_curved}

Для численной верификации используется параметрическая задача
\begin{equation}\label{eq:curved_valley}
  f(x_1, x_2) = \frac{\alpha}{2}(x_2 - \beta x_1^2)^2
  + \frac{1}{2}(x_1^2 + x_2^2),
\end{equation}
где $\alpha > 0$~--- жёсткость поперёк долины,
$\beta \geq 0$~--- кривизна.
Решение $x^* = 0$, гессиан $\nabla^2 f(0) = \mathrm{diag}(1, 1+\alpha)$.
Градиент и гессиан задаются явно:
\[
  \nabla f(x) = \begin{pmatrix}
    x_1 - 2\alpha\beta\, x_1(x_2 - \beta x_1^2) \\
    x_2 + \alpha(x_2 - \beta x_1^2)
  \end{pmatrix}, \qquad
  \nabla^2 f(x) - \nabla^2 f(0) = 2\alpha\beta
  \begin{pmatrix}
    3\beta x_1^2 - x_2 & -x_1 \\
    -x_1 & 0
  \end{pmatrix}.
\]

Константа Липшица гессиана пропорциональна~$\alpha\beta$: дефект
несовместности~\eqref{eq:incompat} равен
$|s_j^\top r_k| = O(\alpha\beta\,\|x\|^2\,\|s\|)$.
При $\beta = 0$ задача квадратична и все методы сходятся.
С~ростом~$\beta$ собственные векторы гессиана вращаются вдоль
траектории, что приводит к накоплению ошибок прошлых секущих
в~SR1 и потере сходимости.

\subsection{Численные результаты}\label{sec:ss_numerics}

\paragraph{Скорость сходимости.}
Сравним сходимость SR1 и SS-SR1 на задаче~\eqref{eq:curved_valley}
с выключенным масштабированием начального $B_0 = I$, линейным поиском
Армихо ($c_1 = 10^{-4}$, откат $\alpha \leftarrow \alpha/2$) и стартом
$x_0 = (2, 2)$ / $x_0 = (1{,}8, 1{,}8)$ для двух уровней жёсткости.
На Рис.~\ref{fig:ss_conv} приведены траектории остатка~$\|\nabla f\|$
по итерациям и по числу вычислений~$\nabla f$.

\begin{figure}[htbp]
\centering
\includegraphics[width=\textwidth]{fig_ss_sr1_conv.pdf}
\caption{Сходимость SR1 и SS-SR1 на задаче Curved Valley при двух
  уровнях жёсткости. Слева~--- по итерациям, справа~--- по числу
  вызовов~$\nabla f$ (включая линейный поиск).}
\label{fig:ss_conv}
\end{figure}

На режиме $\alpha = 50$, $\beta = 1$ SS-SR1 сходится за~$24$ итерации
до $\|\nabla f\| \sim 10^{-16}$, тогда как SR1 стагнирует на уровне
$\|\nabla f\| \sim 1$ из-за вырождения $B_k$ и коллапса шага
(на $200$ итерациях израсходовано~$\sim 5600$ вызовов~$\nabla f$
в линейном поиске).
При возросшей жёсткости $\alpha = 100$, $\beta = 1{,}5$ SS-SR1 остаётся
устойчивым (~$75$ итераций / $800$ вызовов до сходимости), в то время
как SR1 расходится. Разрыв между методами отражает эффект MS-коррекции
в~$s_k^\perp$: удаление направленной деградации $B_k$ на каждом шаге
восстанавливает устойчивость линейной системы~$B_k d_k = -\nabla f$
и устраняет паразитарные масштабы шага.

\paragraph{Бассейн сходимости.}
Для задачи~\eqref{eq:curved_valley} при $\alpha = 100$ и различных~$\beta$
методом бисекции по начальному радиусу~$\|x_0\|$ вдоль
$N_\theta = 72$~равномерно распределённых направлений определена
граница бассейна сходимости (Рис.~\ref{fig:basin}).
При $\beta = 0{,}3$ оба метода имеют сопоставимые бассейны;
при $\beta = 1{,}0$ бассейн SS-SR1 в~${\sim}\,1{,}9$ раза больше;
при $\beta = 2{,}0$~--- в~${\sim}\,1{,}8$ раза.

\paragraph{Механизм стабилизации.}
На кривых долинах гессиан быстро вращается вдоль траектории.
SR1 обновляет~$B_k$ только в~направлении~$s_k$, игнорируя
деградацию в~остальных направлениях:
ошибки прошлых секущих накапливаются,
$\kappa(B_k)$ растёт, шаги ухудшаются.
MS-коррекция в~$s_k^\perp$ на каждом шаге уменьшает невязку
в~наиболее «повреждённом» направлении окна,
предотвращая деградацию~$B_k$.
Сверхлинейная скорость при этом не страдает
(Теорема~\ref{thm:ss_dm}): коррекция
ортогональна~$s_k$ и не влияет на условие Денниса--Море.
