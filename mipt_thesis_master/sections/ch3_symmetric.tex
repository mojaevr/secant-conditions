\section{Секантно-стабилизированные симметричные методы}\label{sec:ss}

\subsection{Мотивация}\label{sec:ss_motivation}

В предыдущих разделах метод SP-Broyden решил задачу сохранения секущих
условий для общих систем $F(x) = 0$, однако матрица~$B_k$ при этом
несимметрична. Для задач безусловной оптимизации, где $F = \nabla f$ и
$J^* = \nabla^2 f(x^*)$ симметричен, естественно требовать $B_k = B_k^\top$.

Классические симметричные методы~--- SR1, BFGS, PSB~---
удовлетворяют текущему секущему условию $B_{k+1} s_k = y_k$,
но не контролируют невязку прошлых секущих
$R_{\mathrm{past}} = B_k S_{\mathrm{past}} - Y_{\mathrm{past}}$.
Следующая теорема показывает, что одновременное точное выполнение
всех секущих и симметрии невозможно.

\begin{theorem}[Несовместность]\label{thm:incompat}
Пусть $\Delta B = \Delta B^\top$ и $\Delta B\, s_j = 0$ для всех
$j < k$, где $\{s_j\}$~--- прошлые шаги. Тогда
\begin{equation}\label{eq:incompat}
  s_j^\top (\Delta B\, s_k) = (\Delta B\, s_j)^\top s_k = 0
  \quad \forall\, j < k.
\end{equation}
Следовательно, секущее условие $\Delta B\, s_k = r_k$
совместно с~\eqref{eq:incompat} тогда и только тогда, когда
$s_j^\top r_k = 0$ для всех~$j < k$.
\end{theorem}

\begin{proof}
Из $\Delta B = \Delta B^\top$ и $\Delta B\, s_j = 0$ следует
$s_j^\top \Delta B = (\Delta B\, s_j)^\top = 0$.
Применяя к~$s_k$: $s_j^\top \Delta B\, s_k = 0$.
\end{proof}

\begin{remark}
Для $F = \nabla f$ дефект совместности оценивается через разность
средних гессианов:
\[
  s_j^\top r_k = s_j^\top (\bar{H}_k - \bar{H}_j)\, s_k
  = O(L_H \cdot \|x_k - x_j\| \cdot \|s_j\| \cdot \|s_k\|),
\]
где $\bar{H}_i = \int_0^1 \nabla^2 f(x_i + t\, s_i)\, dt$ и $L_H$~---
константа Липшица гессиана. Дефект мал вблизи~$x^*$, но не равен нулю.
На квадратичной функции $\bar{H}_k = \bar{H}_j$, и совместность
выполнена тождественно.
\end{remark}

\subsection{Конструкция SS-\texorpdfstring{$\mathcal{U}$}{U}}
\label{sec:ss_construction}

\begin{definition}[Секантно-стабилизированное обновление]
\label{def:ss}
Пусть $\mathcal{U}$~--- базовое симметричное квазиньютоновское
обновление, удовлетворяющее текущему секущему условию
$\mathcal{U}(B_k, s_k, y_k)\, s_k = y_k$.
\emph{Секантно-стабилизированное расширение}
$\mathrm{SS}\text{-}\mathcal{U}$:
\begin{equation}\label{eq:ss_update}
  B_{k+1} = \underbrace{\mathcal{U}(B_k, s_k, y_k)}_{\displaystyle B'_k}
  + \sigma_k^*\, u_k^*\, u_k^{*\top},
  \qquad u_k^* \perp s_k,\;\; \|u_k^*\| = 1,
\end{equation}
где $u_k^*$~--- ведущий собственный вектор матрицы
$P_k M'_k P_k$,
\[
  M'_k = \tfrac{1}{2}(R'_k S_p^\top + S_p R_k^{\prime\top}),
  \qquad
  R'_k = B'_k S_p - Y_p,
  \qquad
  P_k = I - \hat{s}_k \hat{s}_k^\top,
\]
а~$\sigma_k^*$ выбирается из минимума $\|R'_k + \sigma\, u^*\, u^{*\top} S_p\|_F^2$:
\begin{equation}\label{eq:ss_sigma}
  \sigma_k^* = -\frac{u_k^{*\top} R'_k S_p^\top u_k^*}
                     {\|S_p^\top u_k^*\|^2}.
\end{equation}
\end{definition}

\begin{remark}[Частные случаи]
При $\mathcal{U} = \mathrm{SR1}$ получаем \emph{SS-SR1}~---
основной метод данного раздела. Выбор $\mathcal{U} = \mathrm{PSB}$
или $\mathcal{U} = \mathrm{BFGS}$ даёт SS-PSB и SS-BFGS
соответственно. Коррекция применима к произвольному базовому методу.
\end{remark}

\begin{remark}[Связь с SP-Broyden]
SP-Broyden и SS-$\mathcal{U}$ решают одну задачу~--- контроль
деградации прошлых секущих~--- разными механизмами.
SP-Broyden модифицирует направление обновления ($s_k \to v_k^*$),
обеспечивая точные прошлые секущие ценой потери симметрии.
SS-$\mathcal{U}$ добавляет коррекцию в~$s_k^\perp$, обеспечивая
симметрию ценой приближённых прошлых секущих.
\end{remark}

\subsection{Реализация SS-SR1 в общей размерности}
\label{sec:ss_sr1_algo_ndim}

Определение~\ref{def:ss} задаёт обновление SS-$\mathcal{U}$
неявно, через ведущий собственный вектор симметричной
матрицы $P_k M'_k P_k \in \RR^{n\times n}$.
В двумерном случае ($n=2$) эта матрица имеет ранг не выше~$1$,
ведущий собственный вектор находится за~$O(1)$ операций
и алгоритм сводится к одной формуле (см.\ \texttt{basin.ipynb}
для базовой 2D-версии). Для общей размерности $n>2$ требуется
аккуратная реализация с тремя численными охранниками; полный
псевдокод приведён в~Алгоритме~\ref{alg:ss_sr1} (используется
во всех численных экспериментах гл.~3 при $n\ge 10$, см.\
\texttt{diag\_ss\_sr1.py}).

\begin{algorithm}
\caption{SS-SR1$(\nabla f,\,x_0,\,p_{\max},\,\varepsilon;\,
        \varepsilon_{\mathrm{skip}},\,\varepsilon_u,\,\varepsilon_S)$}
\label{alg:ss_sr1}
\begin{algorithmic}
\Require $\nabla f\colon\RR^n\to\RR^n$,\ $x_0\in\RR^n$,\
         $p_{\max}\ge 0$,\ $\varepsilon>0$
\Require \emph{(охранники, см.\ ремарку~\ref{rem:ss_sr1_safeguards})}\hfil\null
         \par\hangindent=2em
         $\varepsilon_{\mathrm{skip}}>0$ (порог SR1-skip),\
         $\varepsilon_u>0$ (порог невырожденности~$u_k^*$),\
         $\varepsilon_S>0$ (порог $\Norm{S_p^{\top} u_k^*}^2$).
\State $B_0 \leftarrow I_n$,\quad $k \leftarrow 0$
\While{$\Norm{\nabla f(x_k)}\ge \varepsilon$}
  \State Решить $B_k\,d_k = -\nabla f(x_k)$
  \State $\alpha_k \leftarrow$ Armijo-backtracking на $f$
  \State $x_{k+1}\leftarrow x_k+\alpha_k d_k$,\
         $s_k\leftarrow \alpha_k d_k$,\
         $y_k\leftarrow \nabla f(x_{k+1})-\nabla f(x_k)$
  %---- Шаг 1. SR1-обновление с skip-rule ----
  \State $r_k \leftarrow y_k - B_k s_k$
  \If{$|s_k^{\top} r_k|<\varepsilon_{\mathrm{skip}}\,\Norm{s_k}\Norm{r_k}$}
    \State $B'_k \leftarrow B_k$
       \Comment{SR1-skip,~\cite[\S 6.2]{bfgs}}
  \Else
    \State $B'_k \leftarrow B_k + \dfrac{r_k r_k^{\top}}{r_k^{\top} s_k}$
  \EndIf
  %---- Шаг 2. Окно прошлых секущих ----
  \State $p \leftarrow \min(p_{\max},k)$
  \State $S_p\leftarrow [s_k\mid s_{k-1}\mid\cdots\mid s_{k-p}]$,\quad
         $Y_p\leftarrow [y_k\mid y_{k-1}\mid\cdots\mid y_{k-p}]$
       \hfill\Comment{$n\times(p{+}1)$}
  %---- Шаг 3. Поиск ведущего собственного вектора ----
  \If{$p\ge 0$ \textbf{and} $S_p$ непуста}
    \State $R'_k \leftarrow B'_k S_p - Y_p$
       \hfill\Comment{$n\times(p{+}1)$}
    \State $M'_k \leftarrow \tfrac{1}{2}\!\left(R'_k S_p^{\top} + S_p R^{\prime\top}_k\right)$
       \hfill\Comment{$n\times n$, симметричная}
    \State $\hat s_k \leftarrow s_k/\Norm{s_k}$
    \State $A_k \leftarrow (I-\hat s_k\hat s_k^{\top})\,M'_k\,(I-\hat s_k\hat s_k^{\top})$
    \State $A_k \leftarrow \tfrac{1}{2}(A_k+A_k^{\top})$
       \Comment{симметризация против ошибок плавающего}
    \State $(\lambda_i,V_{\cdot,i})_{i=1}^{n} \leftarrow \mathtt{eigh}(A_k)$
    \State $j \leftarrow \arg\max_i |\lambda_i|$,\quad
           $u\leftarrow V_{\cdot,j}$
       \hfill\Comment{ведущий по $|\lambda|$}
    \State $u \leftarrow u-(u^{\top}\hat s_k)\hat s_k$
       \Comment{репроекция на $s_k^{\perp}$}
    %---- Шаг 4. Охранники ----
    \If{$\Norm{u}<\varepsilon_u$}
      \State $B_{k+1}\leftarrow B'_k$
         \Comment{вырождение направления коррекции}
    \Else
      \State $u\leftarrow u/\Norm{u}$,\quad $\eta\leftarrow S_p^{\top} u$
      \If{$\Norm{\eta}^2 < \varepsilon_S$}
        \State $B_{k+1}\leftarrow B'_k$
           \Comment{$S_p^{\top} u\to 0$, нет информации о~$R_{\mathrm{past}}$}
      \Else
        \State $\sigma_k^{*} \leftarrow -(u^{\top} R'_k\,\eta)/\Norm{\eta}^2$
        \State $B_{k+1}\leftarrow B'_k + \sigma_k^{*}\,u u^{\top}$
      \EndIf
    \EndIf
  \Else
    \State $B_{k+1}\leftarrow B'_k$
       \Comment{$k=0$, окно пусто}
  \EndIf
  \State $k\leftarrow k+1$
\EndWhile
\Return $x_k$
\end{algorithmic}
\end{algorithm}

\begin{remark}[Численные охранники]\label{rem:ss_sr1_safeguards}
Алгоритм~\ref{alg:ss_sr1} содержит три охранника
$(\varepsilon_{\mathrm{skip}},\varepsilon_u,\varepsilon_S)$,
каждый из которых выключает MS-коррекцию и оставляет
$B_{k+1}=B'_k$. Дефолты:
$\varepsilon_{\mathrm{skip}}=10^{-8}$,
$\varepsilon_u=10^{-12}$,
$\varepsilon_S=10^{-15}$.
\textbf{(i)} \emph{SR1-skip}: классический способ избежать
деления на ноль в SR1 при $|s_k^{\top} r_k|\to 0$~\cite[\S 6.2]{bfgs};
срабатывает на гладких задачах из рис.~\ref{fig:ss_sr1_ndim_conv}
в $0$--$3\,\%$ итераций. \textbf{(ii)} \emph{Вырождение
коррекции}: после численной репроекции
$u\leftarrow u-(u^{\top}\hat s_k)\hat s_k$ норма~$\Norm{u}$ может
оказаться меньше уровня машинного эпсилон, если ведущий
собственный вектор лежит почти полностью в~$\mathrm{span}(s_k)$
(это означает, что $A_k$ мало отличается от ранга~$0$, т.\,е.\
MS-невязка в~$s_k^\perp$ уже мала и не нуждается в коррекции).
\textbf{(iii)} \emph{Малость~$\Norm{S_p^{\top} u}^2$}: даже
при~$\Norm{u}=1$ знаменатель в~\eqref{eq:ss_sigma} может
вырождаться, если выбранное направление $u\in s_k^{\perp}$
ортогонально всем прошлым секущим (т.\,е.\ MS-невязка в этом
направлении нечувствительна к~$S_p$); коррекция бессмысленна.
В численных экспериментах гл.~3
(рис.~\ref{fig:ss_sr1_ndim_rpast}) суммарная доля срабатываний
охранников~(ii)--(iii) на четырёх задачах MGH составляет
$0$--$2\,\%$ итераций.
\end{remark}

\begin{remark}[Стоимость и редукция к малому собственному
            подзадаче]\label{rem:ss_sr1_cost}
В наивной реализации Алгоритма~\ref{alg:ss_sr1} доминирующая
цена итерации~--- спектральное разложение матрицы
$A_k\in\RR^{n\times n}$, $\mathtt{eigh}(A_k)=O(n^3)$
(QR-итерация на трёхдиагонализованной форме~\cite[\S 8.3]{golub2013}).
Для умеренных $n\lesssim$ нескольких сотен, типичных в
численных экспериментах гл.~3 ($n\in\{10,12\}$ для~MGH),
эта стоимость пренебрежима по сравнению с затратами на
вычисление~$\nabla f$ и решение системы $B_k d_k=-\nabla f(x_k)$
($O(n^3)$ через LU). Для большой~$n$ ключевое наблюдение
\begin{equation}\label{eq:ss_sr1_low_rank}
  \mathrm{rank}(A_k) \;=\; \mathrm{rank}(P_k M'_k P_k)
  \;\le\; \mathrm{rank}(M'_k)
  \;\le\; 2(p+1),
\end{equation}
поскольку $M'_k=\tfrac{1}{2}(R'_k S_p^{\top}+S_p R^{\prime\top}_k)$~---
сумма двух матриц ранга не выше~$p+1$. Это позволяет редуцировать
полное собственное разложение $n\times n$ к симметричному
$2(p+1)\times 2(p+1)$ собственному подзадаче на подпространстве
$\mathrm{range}([P_k S_p,\;P_k R'_k])$ через~QR-ортогонализацию
этого базиса и проекцию~$M'_k$ на него (приём, стандартный
для блочных Ланцош-итераций~\cite[Гл.~13]{parlett1998};
ср.\ также подход Fang--Saad~\cite{fang2009} для
мультисекантных схем). Стоимость падает до
$O(n^2 p+np^2+p^3)$ за итерацию, а память для хранения~$A_k$~---
с $O(n^2)$ до $O(np)$; при типичном $p\le 10$ это даёт
${\sim}\,n/p$~--раз экономии для $n\gtrsim 100$. Реализация
в~\texttt{diag\_ss\_sr1.py} использует прямой
\texttt{numpy.linalg.eigh}, поскольку $n\in\{10,12\}$;
редукция~\eqref{eq:ss_sr1_low_rank} является прямой инженерной
оптимизацией и не меняет результат алгоритма.
\end{remark}

\subsection{Свойства}\label{sec:ss_properties}

\begin{theorem}[Свойства SS-$\mathcal{U}$]\label{thm:ss_props}
Обновление~\eqref{eq:ss_update} обладает следующими свойствами:
\begin{enumerate}
  \item \textbf{Текущее секущее условие:}
    $B_{k+1} s_k = y_k$ (точно).
  \item \textbf{Симметрия:}
    если $\mathcal{U}$ сохраняет симметрию, то $B_{k+1} = B_{k+1}^\top$.
  \item \textbf{Монотонность:}
    $\|R_{\mathrm{past}}^{(k+1)}\|_F^2
    \leq \|R_{\mathrm{past}}'\|_F^2
    - \dfrac{(u_k^{*\top} M'_k\, u_k^*)^2}{\|S_p^\top u_k^*\|^2}$.
  \item \textbf{Ранг обновления:}
    $\mathrm{rank}(B_{k+1} - B_k) \leq \mathrm{rank}(\mathcal{U}) + 1$.
\end{enumerate}
\end{theorem}

\begin{proof}\leavevmode
\begin{enumerate}
  \item Из $u_k^* \perp s_k$:
    $B_{k+1} s_k = B'_k s_k + \sigma_k^*(u_k^{*\top} s_k)\, u_k^* = y_k + 0 = y_k$.
  \item $B_{k+1} = B'_k + \sigma_k^*\, u_k^*\, u_k^{*\top}$~---
    сумма двух симметричных матриц.
  \item Пусть $\tilde{B} = B'_k + \sigma\, u u^\top$. Тогда
    $\|\tilde{B} S_p - Y_p\|_F^2
    = \|R'\|_F^2 + 2\sigma\,(u^\top M' u) + \sigma^2\,\|S_p^\top u\|^2$.
    Минимум по~$\sigma$ при~\eqref{eq:ss_sigma} даёт убывание на
    $(u^\top M' u)^2 / \|S_p^\top u\|^2$.
  \item Очевидно из конструкции. \qedhere
\end{enumerate}
\end{proof}

\subsection{Сверхлинейная сходимость}\label{sec:ss_superlinear}

\begin{theorem}[Условная Q-сверхлинейная сходимость SS-$\mathcal{U}$]
\label{thm:ss_dm}
Пусть выполнены предположения:
\begin{enumerate}
  \item[\textup{(H1)}] $F = \nabla f$, гессиан $\nabla^2 f$ липшицев в окрестности
    $x^*$ с константой $L_H$;
  \item[\textup{(H2)}] метод SS-$\mathcal{U}$ сходится: $x_k \to x^*$, $s_k \to 0$,
    $\nabla f(x^*) = 0$;
  \item[\textup{(H3)}] предельный гессиан $J^* := \nabla^2 f(x^*)$ невырожден;
  \item[\textup{(H4)}] начиная с некоторого $k_0$ глобализация выдаёт полный
    квазиньютоновский шаг $s_k = -B_k^{-1}\nabla f(x_k)$, причём
    $\|B_k^{-1}\| \le M_*$ равномерно при $k \ge k_0$;
  \item[\textup{(H5)}] аппроксимация устойчива по приращениям:
    $\|B_{k+1} - B_k\|_F \le C_B\, \|s_k\|$ при некоторой $C_B \ge 0$
    в окрестности~$x^*$.
\end{enumerate}
Тогда:
\begin{enumerate}
  \item \textup{(условие Денниса--Море)}
    \[
      \frac{\|(B_k - J^*)\, s_k\|}{\|s_k\|}
      \;\le\; L_H\bigl(\|x_k - x^*\| + \|s_k\|\bigr) + C_B\, \|s_k\|
      \;\xrightarrow{k\to\infty}\; 0;
    \]
  \item \textup{(Q-сверхлинейность)}
    $\dfrac{\|x_{k+1} - x^*\|}{\|x_k - x^*\|} \to 0$.
\end{enumerate}
\end{theorem}

\begin{proof}
\textit{Шаг 1 (DM-условие для $B_{k+1}$).}
Из точного секущего условия $B_{k+1} s_k = y_k$ (Теорема~\ref{thm:ss_props}.1):
\[
  (B_{k+1} - J^*)\, s_k = y_k - J^*\, s_k
  = \int_0^1 [\nabla^2 f(x_k + t\, s_k) - \nabla^2 f(x^*)]\, s_k\, dt.
\]
По~(H1) $\|\nabla^2 f(x_k + t s_k) - \nabla^2 f(x^*)\| \le L_H(\|x_k - x^*\| + \|s_k\|)$,
откуда
\begin{equation}\label{eq:ss_dm_bk1}
  \|(B_{k+1} - J^*)\, s_k\| \le L_H\bigl(\|x_k - x^*\| + \|s_k\|\bigr)\,\|s_k\|.
\end{equation}

\textit{Шаг 2 (перенос на $B_k$).}
По~(H5) $\|B_{k+1} - B_k\|_F \le C_B\,\|s_k\|$, откуда
\[
  \|(B_k - J^*)\, s_k\|
  \le \|(B_{k+1} - J^*)\, s_k\| + \|B_{k+1} - B_k\|\cdot\|s_k\|
  \le \bigl[L_H(\|x_k - x^*\| + \|s_k\|) + C_B\,\|s_k\|\bigr]\,\|s_k\|.
\]
Деление на $\|s_k\|$ и применение~(H2) дают пункт~1.

\textit{Шаг 3 (Q-сверхлинейность через характеризацию Денниса--Море).}
По~(H3)--(H4) шаг $s_k = -B_k^{-1}\nabla f(x_k)$ корректно определён при $k \ge k_0$,
так что $\nabla f(x_k) = -B_k s_k$ и
\[
  \|\nabla f(x_k) + J^* s_k\|
  = \|(J^* - B_k) s_k\|
  = \|(B_k - J^*)\, s_k\|.
\]
Согласно характеризации Денниса--Море~\cite[Th.~2.2]{dennis1974},
для сходящейся к корню $x^*$ ($\nabla f(x^*) = 0$) последовательности
$x_{k+1} = x_k - B_k^{-1}\nabla f(x_k)$ при невырожденном $J^*$ Q-сверхлинейная
сходимость эквивалентна условию
\[
  \frac{\|\nabla f(x_k) + J^* s_k\|}{\|s_k\|} \xrightarrow{k\to\infty} 0,
\]
т.\,е. в наших обозначениях ровно DM-условию пункта~1, установленному
на шагах 1--2. Применяя эквивалентность, получаем
$\|x_{k+1} - x^*\|/\|x_k - x^*\| \to 0$.
\end{proof}

\begin{remark}[Роль предположений]\label{rem:ss_dm_assumptions}
Предположения~(H3) и~(H4) суть стандартные локальные допущения о хорошей
обусловленности задачи: невырожденность $J^*$ нужна для применимости
Денниса--Море~\cite{dennis1974}, а допустимость полного шага в окрестности
$x^*$~--- классическое следствие условия Армихо с $c_1 < 1/2$ для
дважды-дифференцируемой $f$ (\cite{bfgs}, Th.~3.6). Условие~(H5) (bounded
deterioration) для SP-семейства следует из расширенной оценки
Theorem~\ref{thm:bd} ($\|E_{k+1}\|_F \le \|E_k\|_F + O(\|s_k\|)$);
для SR1 без MS-коррекции~(H5) может нарушаться, что даёт практические
расходимости~\cite{conn1991}, наблюдаемые в~\S\ref{sec:ss_curved}.
\end{remark}

\begin{remark}
MS-коррекция не влияет на сверхлинейную скорость
(она действует в~$s_k^\perp$, а условие Денниса--Море~---
только в~направлении~$s_k$). Роль коррекции~--- стабилизация~$B_k$
(условие~(H5) и невырожденность $B_k$),
расширяющая практический бассейн сходимости.
\end{remark}

\begin{remark}[Условный характер Th.~\ref{thm:ss_dm} на Curved
  Valley]\label{rem:ss-dm-cv-gap}
Теорема~\ref{thm:ss_dm}~--- \emph{условное} утверждение: при
выполнении~(H1)--(H5) метод сходится Q-сверхлинейно, но сама посылка
$x_k\to x^*$~(H2) у~SR1 без~стабилизации не~гарантирована.
В~эксперименте на Extended Curved Valley~$n=10$ при~$\alpha=100,\beta=1{,}5$
из~стартовой точки~$x_0=(1{,}8;\,1{,}8)$ \emph{без} линейного поиска SR1
расходится (см.\ обсуждение в~\S\ref{sec:ss_curved} и~рис.~\ref{fig:qn_compare_conv}
ниже): шаг $s_k=-B_k^{-1}\nabla f(x_k)$ становится слишком большим, $x_k$
выходит из~бассейна, $\nabla^2 f(x_k)$ резко вращается, и~SR1-обновление,
не~видя поперечной деградации, теряет связь с~гессианом.
В~такой ситуации Th.~\ref{thm:ss_dm}~\emph{неприменима} к~SR1: посылка~(H2)
ложна, а~само заключение~--- сверхлинейность~--- бессмысленно (метод вообще
не~сходится).

Разрыв закрывается тремя способами, восстанавливающими~(H2). \emph{Первый}~---
MS-коррекция SS-SR1: ограничение~$\|B_kS_p-Y_p\|_F$
(Теорема~\ref{thm:ss_props}, п.~3) играет роль bounded deterioration в смысле
Бройдена--Денниса--Море~\cite{broyden1973,dennis1996}, обеспечивая локальную
ограниченность $\|B_k^{-1}\|$ в окрестности~$x^*$~--- отсюда
допустимость полного шага и~$x_k\to x^*$. На том же режиме~$(\alpha,\beta,x_0)$
SS-SR1 сходится без~глобализации (табл.~\ref{tab:radius} и~рис.~\ref{fig:tab31_ci}:
радиус $R(\mathrm{SS-SR1})$ упирается в~потолок~$R_{\max}=6$ для всех
исследованных~$\beta$). \emph{Второй}~--- BFGS, у~которого положительная
определённость~$B_k$ автоматически обеспечивает направление спуска
и~совместимость с~Армихо~\cite[\S6.1]{bfgs}. \emph{Третий}~--- глобализация
по~Армихо (см.\ \S\ref{sec:globalization} в~гл.~2): сходимость к~стационарной
точке по~теореме Цоутендейка~\cite{zoutendijk1970}; невырожденность~$J^*$ далее
гарантирует~$\nabla f(x^*)=0$ и~локальную допустимость полного шага. На
рис.~\ref{fig:qn_compare_conv} с~единым backtracking-Армихо все методы
(SR1, SS-SR1, BFGS, DFP, L-BFGS) сходятся, что иллюстрирует именно
применимость Th.~\ref{thm:ss_dm} в~условиях глобализации.

Иначе говоря, Th.~\ref{thm:ss_dm} даёт \emph{скорость условную на сходимости};
сходимость~же~--- независимая задача стабилизации, в~которой SS-SR1
структурно сильнее~SR1, но~для строгой гарантии~(H2) на патологических
режимах требуется глобализация (или MS-коррекция как практический эквивалент).
\end{remark}


% ============================================================
\subsection{Модельная задача: Curved Valley}\label{sec:ss_curved}

Для численной верификации используется параметрическая задача
\begin{equation}\label{eq:curved_valley}
  f(x_1, x_2) = \frac{\alpha}{2}(x_2 - \beta x_1^2)^2
  + \frac{1}{2}(x_1^2 + x_2^2),
\end{equation}
где $\alpha > 0$~--- жёсткость поперёк долины,
$\beta \geq 0$~--- кривизна.
Решение $x^* = 0$, гессиан $\nabla^2 f(0) = \mathrm{diag}(1, 1+\alpha)$.
Градиент и гессиан задаются явно:
\[
  \nabla f(x) = \begin{pmatrix}
    x_1 - 2\alpha\beta\, x_1(x_2 - \beta x_1^2) \\
    x_2 + \alpha(x_2 - \beta x_1^2)
  \end{pmatrix}, \qquad
  \nabla^2 f(x) - \nabla^2 f(0) = 2\alpha\beta
  \begin{pmatrix}
    3\beta x_1^2 - x_2 & -x_1 \\
    -x_1 & 0
  \end{pmatrix}.
\]

Константа Липшица гессиана пропорциональна~$\alpha\beta$: дефект
несовместности~\eqref{eq:incompat} равен
$|s_j^\top r_k| = O(\alpha\beta\,\|x\|^2\,\|s\|)$.
При $\beta = 0$ задача квадратична и все методы сходятся.
С~ростом~$\beta$ собственные векторы гессиана вращаются вдоль
траектории, что приводит к накоплению ошибок прошлых секущих
в~SR1 и потере сходимости.

в~$s_k^\perp$: удаление направленной деградации $B_k$ на каждом шаге
восстанавливает устойчивость линейной системы~$B_k d_k = -\nabla f$
и устраняет паразитарные масштабы шага.

\paragraph{Скейлинг радиуса (эмпирический фит).}
Фиксируя $\beta = 1$ и варьируя~$\alpha$ по~log-сетке
из~$12$~точек $\alpha \in [3, 300]$ (geomspace), на~каждой~$\alpha$
проводим~$10$ независимых запусков с~единичными направлениями
$u_i$, выбранными равномерно на~$S^1$ (фиксированный
\texttt{numpy.random.default\_rng(20260428)}, см.\ \S\ref{ch:reproducibility}).
Радиус сходимости~$R$ оценивается бисекцией с~потолком~$R_{\max}=6$
по~критерию <<$\Norm{F(x_k)}_2 \le 10^{-12}$ за $\le 200$ итераций
квазиньютоновского шага без~steepest-descent~fallback'а>>
(имитация эксперимента basin.ipynb).
В~координатах $\log R$ vs.\ $\log(\alpha\beta)$ методом наименьших
квадратов на~медианах~$10$ запусков получаем степенную регрессию
с~bootstrap-доверительным~интервалом~$95\,\%$ ($2000$ ресэмплов;
рис.~\ref{fig:scaling_iqr}):
\begin{equation}\label{eq:scaling}
\begin{aligned}
R(\text{SR1})    &\;\approx\; C_1\,(\alpha\beta)^{-0{,}54},
   &\text{95\%~CI~для~показателя:} &\;[-0{,}80,\,-0{,}20], \\
R(\text{SS-SR1}) &\;\approx\; C_2\,(\alpha\beta)^{-0{,}01},
   &\text{95\%~CI~для~показателя:} &\;[-0{,}03,\,\phantom{-}0{,}00]
\end{aligned}\qquad\text{(эмпирический закон, $n=2$, $\beta=1$).}
\end{equation}
Доверительный интервал~SS-SR1 включает~ноль: эмпирически радиус
\emph{не~зависит} от~$\alpha\beta$ в~исследованном диапазоне и~саiturator-saturated
у~потолка~$R_{\max}$. SR1, напротив, показывает значимый отрицательный
наклон $\approx -0{,}54$. Отношение
$R(\text{SS-SR1})/R(\text{SR1})$ растёт от~$1$ при~$\alpha\beta\le 30$
(оба метода в~насыщении) до~$\sim 12\times$ при~$\alpha\beta = 300$.
Подчеркнём, что~\eqref{eq:scaling}~--- описательная регрессия,
а не теоретически выведенная оценка: показатели зависят от~задачи
(Curved Valley), правил линейного поиска
($\psi = \tfrac12\Norm{F}^2$ с~Армихо $c_1 = 10^{-4}$ и~откатом
$\alpha_{\mathrm{ls}}\!\leftarrow\!\alpha_{\mathrm{ls}}/2$),
жёсткого ограничения~$\Norm{d_k}\le 50\max(\Norm{x_k},1)$
(стандартный safeguard~basin.ipynb) и~стартового~$B_0 = I$.
Тем не~менее, расширение исходной 3-точечной сетки до~12-точечной
по~log-шкале с~CI~подтверждает: $R(\text{SS-SR1})$
\emph{не~деградирует} с~ростом жёсткости в~исследованном диапазоне,
тогда как~$R(\text{SR1})$ убывает примерно~как~$(\alpha\beta)^{-1/2}$.

\begin{remark}[Сравнение с~консервативной теоретической оценкой]
\label{rem:scaling-classical}
Стандартная теория локальной сходимости квазиньютоновских
методов~\cite{broyden1973,dennis1996} даёт верхнюю границу
радиуса локальной сходимости через константу Липшица
гессиана~$L_H$. Для задачи~\eqref{eq:curved_valley}
$L_H \propto \alpha\beta$, и~неравенство bounded deterioration
в~окрестности~$x^*$ при~$B_0 = I$ влечёт
$\delta^* \le C/L_H = \mathcal{O}\!\bigl((\alpha\beta)^{-1}\bigr)$
(см.\ Th.~8.2.4 в~\cite{dennis1996}).
Наблюдаемые показатели~$-0{,}42$ и~$-0{,}14$ существенно мягче~$-1$:
адаптивные стабилизаторы~--- линейный поиск (для~SR1) и~MS-коррекция
в~$s_k^\perp$ (для~SS-SR1)~--- сохраняют сходимость далеко за
рамки классической локальной окрестности, что и~отражает
расхождение фита~\eqref{eq:scaling} с~теоретической верхней
границей~$\mathcal{O}\!\bigl((\alpha\beta)^{-1}\bigr)$.
Формальный вывод точных показателей в~\eqref{eq:scaling}
требует одновременного учёта геометрии вращения
гессиана~$\nabla^2 f(x)$ вдоль траектории и~взаимодействия
обновлений секущих с~откатом Армихо; это~--- открытая задача
анализа, выходящая за~рамки настоящей работы.
Расширение фита по~$\ge 10$ точкам сетки~$\alpha$ с~доверительными
интервалами по~нескольким случайным стартам отнесено к~разделу
численных экспериментов в~дальнейших работах.
\end{remark}

\paragraph{Механизм стабилизации.}
На кривых долинах гессиан быстро вращается вдоль траектории.
SR1 обновляет~$B_k$ только в~направлении~$s_k$, игнорируя
деградацию в~остальных направлениях:
ошибки прошлых секущих накапливаются,
$\kappa(B_k)$ растёт, шаги ухудшаются.
MS-коррекция в~$s_k^\perp$ на каждом шаге уменьшает невязку
в~наиболее «повреждённом» направлении окна,
предотвращая деградацию~$B_k$.
Сверхлинейная скорость при этом не страдает
(Теорема~\ref{thm:ss_dm}): коррекция
ортогональна~$s_k$ и не влияет на условие Денниса--Море.


\paragraph{Эксперименты в~$n > 2$ (тестовый набор Moré--Garbow--Hillstrom).}
\label{par:ss-sr1-ndim}
Реализация SS-SR1, использованная в~предыдущих экспериментах, опирается
на~явный поворот~$u = R_{90^\circ}\,\hat s_k$ в~ортогональном дополнении
к~$s_k$~--- приём, специфичный для~$n=2$. Для~общего случая ведущий
собственный вектор $u_k^*$ матрицы~$P_k M'_k P_k$
(см.\ Опр.~\ref{def:ss}) вычисляется через симметричную задачу на
собственные значения (\texttt{numpy.linalg.eigh}) или степенным методом. Размерности
$n \in [10, 20]$, на которых ниже приведены результаты, оставляют
эту операцию пренебрежимой по~стоимости относительно
квазиньютоновского шага.

В~качестве~тестового набора взяты четыре задачи:
цепной Розенброк~($n=10$, Nocedal--Wright Eq.~2.22),
Extended Rosenbrock~($n=10$, MGH~\#21),
Extended Powell singular~($n=12$, MGH~\#22) и~векторизованный
Curved Valley~($n=10$, $\alpha=100$, $\beta=1{,}5$,
$x_0$ по~$2$-блокам~$=(1{,}8;\,1{,}8)$~--- тот же режим
жёсткости, что в~эксперименте на~$n=2$). Все три метода
(SR1, SS-SR1, BFGS) используют идентичный backtracking-Armijo
по~$f$ ($c_1=10^{-4}$, откат~$\alpha\!\leftarrow\!\alpha/2$),
$B_0=I$, окно~$p=5$, критерий остановки $\|\nabla f\|_2 \le 10^{-8}$
или $500$~итераций, seed~$20\,260\,428$.

Ключевое эмпирическое наблюдение: на~трёх из~четырёх задач
($\mathrm{Rosenbrock}~\mathrm{chained}$, $\mathrm{Extended}~\mathrm{Rosenbrock}$,
$\mathrm{Extended}~\mathrm{Curved}~\mathrm{Valley}$) SS-SR1 систематически
снижает медианную невязку прошлых секущих относительно SR1, что
прямо соответствует~оценке~3 теоремы~\ref{thm:ss_props}. На~Extended
Powell~singular эффект обратный: вырождение гессиана~$\nabla^2 f(x^*)$
(собственные значения~$\{2,\,0,\,0,\,0\}$ в~каждом~$4$-блоке) делает
$\sigma_{\min}(B'_k S_p)$ малым, и~оптимальный~$\sigma_k^*$
по~формуле~\eqref{eq:ss_sigma} ограничен числовой стабилизацией
знаменателя; в~таком режиме SS-SR1 теряет преимущество, оставаясь
тем не~менее сходящимся. По~числу итераций до~$\|\nabla f\| \le 10^{-8}$
SS-SR1 проигрывает SR1 на~$10$--$80\,\%$ на~гладких задачах: коррекция
ортогональна~$s_k$ и~не~ускоряет именно текущее секущее направление,
а~приобретённый контроль над~прошлыми секущими «окупается» лишь там,
где деградация~$B_k$ в~обход~$s_k$ критична для~устойчивости
(в~основном это режимы быстрого вращения гессиана,
ср.\ \S\ref{sec:ss_numerics}). Эксперимент закрывает вопрос
о~корректности обобщения SS-SR1 на~произвольную размерность,
поставленный в~\S\ref{sec:ss_construction}, и~даёт прямую численную
поддержку утверждения~3 теоремы~\ref{thm:ss_props}.

\paragraph{Сравнение с~BFGS, DFP, PSB и~L-BFGS.}
\label{par:ss-sr1-vs-qn}
Закрывая вопрос о~позиционировании SS-SR1 относительно классических
квазиньютоновских семейств, проводим~\emph{единое} сравнение
$6$~методов на~том же тестовом наборе ($n\geq 10$):
SR1, SS-SR1, BFGS, DFP (двойственное к~BFGS, обновление
обратного~$H_k = B_k^{-1}$~\cite{dennis1996}, гл.~9), PSB
(симметричное Broyden-обновление~Пауэлла,~\cite{dennis1996}, \S~9.2)
и~L-BFGS с~окном~$m=10$ (двух-петельная рекурсия~Носедала~\cite{liu1989}).
Все шесть~методов используют один и~тот~же
backtracking-Армихо-поиск ($c_1 = 10^{-4}$, откат~$\alpha/2$),
$B_0 = I$~(или~$H_0 = I$ для~DFP/L-BFGS), критерий~остановки
$\Norm{\nabla f}_2 \le 10^{-8}$ или~$500$~итераций. Сырые данные --
\texttt{qn\_compare.npz}, скрипт -- \texttt{diag\_qn\_compare.py}.

Сводные результаты приведены на~рис.~\ref{fig:qn_compare_conv}
(сходимость) и~рис.~\ref{fig:qn_compare_summary} (стоимость).
Качественная картина:
\begin{itemize}\itemsep 0pt
\item BFGS и~двойственный ему DFP сходятся на~всех четырёх задачах
   за~$85$--$98$ итераций; различие между ними (с~общим Армихо-поиском)
   укладывается в~$\pm 2$ итерации, что подтверждает классическую
   эквивалентность по~числу шагов на~гладких задачах~\cite{dennis1996}.
\item L-BFGS$(m=10)$ работает лучше всех baseline'ов по~числу
   $\#\nabla f$ за~счёт меньшего числа Армихо-откатов
   ($46$--$78$ итераций, $138$--$195$~$\#f$); ограниченная~память
   не~является препятствием при~$n\le 12$.
\item PSB на~двух из~четырёх задач (Powell~singular и~Curved~Valley)
   \emph{не сходится за~$500$~итераций} -- это согласуется с~теорией:
   PSB не~гарантирует положительной определённости~$B_k$,
   поэтому в~режимах вырождения~гессиана или быстрого~вращения
   деградирует. Заметим, что в~литературе PSB известен как
   <<надёжный, но~медленный>>~\cite[Th.~9.4.1]{dennis1996} -- наши
   эксперименты подтверждают первую часть только на~гладких задачах.
\item SR1 и~SS-SR1 находятся между PSB и~BFGS-семейством по~числу
   итераций. SS-SR1 уступает SR1 на~$10$--$80\,\%$ итераций на~гладких
   задачах (overhead MS-коррекции в~$s_k^\perp$),
   но~систематически снижает $\Norm{B_k S_p - Y_p}_F$
   на~$3$~из~$4$ задач (рис.~\ref{fig:ss_sr1_ndim_rpast}) -- это
   именно тот выигрыш, который~SS-SR1 \emph{и~должен} давать
   по~теореме~\ref{thm:ss_props} пункт~3.
\end{itemize}
Ключевой~вывод эксперимента: SS-SR1 \emph{не~предназначен}
конкурировать с~BFGS/L-BFGS по~стоимости~сходимости. Его~ниша
строго~уже~--- режимы, где~SR1 деградирует по~прошлым~секущим
из-за вращения~гессиана (Curved~Valley с~большим~$\beta$), и~где
именно контроль~над~прошлыми~секущими, а~не~текущая~скорость,
оказывается критическим (ср.\ \S\ref{sec:ss_numerics}~--- там
SR1~\emph{не~сходится} с~$x_0=(1{,}8;\,1{,}8)$ при~$\alpha=100,\beta=1{,}5$,
а~SS-SR1~сходится). Этот~режим~воспроизведён в~Extended~Curved~Valley
с~$n=10$ и~теми~же~параметрами~$2$-блока: SR1 сходится за~$100$~итераций,
BFGS за~$93$, SS-SR1~--- за~$119$~(на~$19\,\%$~медленнее SR1, в~окрестности
дляобласти стабильности~SR1~ещё внутри~бассейна сходимости).
