\chapter{Алгоритм SP-Broyden}\label{ch:sp-broyden-theory}
%% ============================================================

\noindent\textbf{Обозначения и предположения главы.}\enskip
Рассматривается задача поиска корня $F(x) = 0$ с
непрерывно-дифференцируемым отображением $F\colon\RR^n \to \RR^n$
и якобианом $J(x) = F'(x)$. Обозначим $x^*$~--- корень системы
($F(x^*) = 0$), $J^* = J(x^*)$, $s_k = x_{k+1} - x_k$,
$y_k = F(x_{k+1}) - F(x_k)$, $E_k = B_k - J^*$~--- ошибку
аппроксимации якобиана. Во всех теоретических утверждениях
главы~2 предполагается, что якобиан липшицев на некоторой открытой
окрестности $\mathcal{O} \subset \RR^n$ точки $x^*$:
\begin{equation}\label{eq:lipschitz}
  \Norm{J(x) - J(y)} \le L\,\Norm{x - y},
  \qquad x, y \in \mathcal{O}.
\end{equation}
Это стандартное предположение для анализа локальной сходимости
квазиньютоновских методов; см.~\cite{dennis1977,dennis1996}.

\medskip

Идея алгоритма SP-Broyden: выбрать $v_k$ так, чтобы ранг-1
обновление~\eqref{eq:rank1_normalized} удовлетворяло текущему секущему
условию $B_{k+1}s_k = y_k$ и одновременно \emph{сохраняло} $p$ предыдущих
секущих условий $B_{k+1}s_j = B_k s_j$ для $j = k{-}1, \ldots, k{-}p$.
По Утверждению~\ref{prop:destruction}, для этого достаточно потребовать
$v_k^\top s_j = 0$ при $j = k{-}1, \ldots, k{-}p$.

\begin{definition}[SP-обновление]\label{def:sp_update}
Пусть задана глубина сохранения $p \geq 0$ и пусть шаги
$s_k, s_{k-1}, \ldots, s_{k-p}$ линейно независимы.
Определим \emph{SP-вектор} $v_k^*$ как решение задачи
\begin{equation}\label{eq:sp_opt}
  \min_{v \in \mathbb{R}^n} \|v\|^2 \quad \text{при} \quad
  v^\top s_k = 1, \quad v^\top s_j = 0 \;\; (j = k{-}1, \ldots, k{-}p).
\end{equation}
SP-обновление якобиана имеет вид
\begin{equation}\label{eq:sp_update}
  B_{k+1} = B_k + (y_k - B_k s_k)\,(v_k^*)^\top.
\end{equation}
\end{definition}

\begin{remark}[Соглашение об обозначениях]\label{rem:notation}
Параметр $p$ обозначает число \emph{старых} секущих условий, которые
сохраняются. Матрица
$S_p = [s_k \mid s_{k-1} \mid \cdots \mid s_{k-p}] \in \mathbb{R}^{n\times(p+1)}$
содержит $p+1$ столбцов: текущий шаг $s_k$ стоит первым (это определяет
вектор $e_1$ в формулах ниже), за ним~--- $p$ предыдущих шагов.
В отличие от этого, в общей формуле~\eqref{eq:general} столбцы $S_m$
расположены в хронологическом порядке: от старых к новым.
\end{remark}

\begin{proposition}[Явная формула для $v_k^*$]\label{prop:vstar}
Пусть
$S_p = [s_k \mid s_{k-1} \mid \cdots \mid s_{k-p}] \in \mathbb{R}^{n \times (p+1)}$
и $G_p = S_p^\top S_p \in \mathbb{R}^{(p+1)\times(p+1)}$~--- грамиан.
Если $G_p$ невырожден, то
\begin{equation}\label{eq:vstar}
  v_k^* = S_p\,G_p^{-1}\,e_1,
\end{equation}
где $e_1 = (1, 0, \ldots, 0)^\top \in \mathbb{R}^{p+1}$.
При этом $\|v_k^*\|^2 = e_1^\top G_p^{-1}\,e_1$.
\end{proposition}

\begin{proof}
Задача~\eqref{eq:sp_opt} эквивалентна:
$\min_v \|v\|^2$ при $S_p^\top v = e_1$.
Целевая функция строго выпукла, ограничения аффинны, поэтому
стационарная точка функции Лагранжа является глобальным минимумом.

Функция Лагранжа имеет вид
$\mathcal{L}(v,\lambda) = \|v\|^2 - 2\lambda^\top(S_p^\top v - e_1)$.
Условие стационарности $\nabla_v\mathcal{L} = 2v - 2S_p\lambda = 0$
даёт $v = S_p\lambda$.
Подставляя в ограничение, получаем $G_p\lambda = e_1$,
откуда $\lambda = G_p^{-1}e_1$ и $v_k^* = S_p\,G_p^{-1}\,e_1$.

Норма оптимального вектора:
$\|v_k^*\|^2 = (v_k^*)^\top v_k^*
= e_1^\top G_p^{-1} S_p^\top S_p\,G_p^{-1} e_1
= e_1^\top G_p^{-1} e_1$.
\end{proof}

\begin{remark}\label{rem:p_zero}
При $p = 0$ ограничения ортогональности отсутствуют, $S_0 = [s_k]$,
$G_0 = \|s_k\|^2$, и формула~\eqref{eq:vstar} даёт
$v_k^* = s_k/\|s_k\|^2$.
Обновление~\eqref{eq:sp_update} принимает вид
\[
  B_{k+1} = B_k + \frac{(y_k - B_k s_k)\,s_k^\top}{\|s_k\|^2},
\]
что совпадает с классическим обновлением Бройдена~\eqref{eq:good_broyden}.
\end{remark}

\begin{remark}[Геометрическая интерпретация]\label{rem:geometry}
Вектор $v_k^*$ лежит в $\operatorname{col}(S_p)$ (по построению $v = S_p\lambda$)
и ортогонален подпространству $\operatorname{span}(s_{k-1}, \ldots, s_{k-p})$
(по ограничениям задачи). Таким образом, $v_k^*$ направлен вдоль компоненты
$s_k$, ортогональной прошлым шагам, и нормирован условием $v_k^{*\top} s_k = 1$.
Обновление~\eqref{eq:sp_update} модифицирует $B_k$ только в направлениях,
ортогональных прошлым шагам, что и обеспечивает сохранение секущих условий.
\end{remark}

\medskip

Полный алгоритм приведён в Алгоритме~\ref{alg:sp_broyden}.

\begin{algorithm}
\caption{SP-Broyden$(F,\,x_0,\,p_{\max},\,\mathtt{multisecant},\,
         \mathtt{accumulate},\,\widetilde{B},\,\tau,\,\varepsilon;\,
         \mathtt{globalize},\,\mathtt{rescale},\,
         \varepsilon_B,\,C_{\mathrm{safe}},\,\nu_{\mathrm{reg}})$}
\label{alg:sp_broyden}
\begin{algorithmic}
\Require $F\colon \mathbb{R}^n \to \mathbb{R}^n$,\ $x_0 \in \mathbb{R}^n$,\
         $p_{\max} \geq 0$,\
         $\mathtt{multisecant} \in \{\text{true},\text{false}\}$,\
         $\mathtt{accumulate} \in \{\text{true},\text{false}\}$,\
         $\widetilde{B} \in \mathbb{R}^{n \times n}$
         \Comment{опорная матрица алгоритма (параметр функции), куда
                  происходит сброс в режиме пересборки; обычно $I_n$.
                  Не путать с базовой матрицей $\widehat{B}_k$ в формуле обновления.},\
         $\tau > 0$,\ $\varepsilon > 0$
\Require \emph{(опции, см.\ ремарки~\ref{rem:init-scaling}--\ref{rem:fallback})}\hfil\null
         \par\hangindent=2em
         $\mathtt{globalize} \in \{\text{true},\text{false}\}$ (флаг Армихо,
         см.\ Алгоритм~\ref{alg:sp_broyden_ls}),\
         $\mathtt{rescale} \in \{\text{true},\text{false}\}$
         (флаг масштабирования $B_0$ по Шанно--Пхуа),\
         $\varepsilon_B > 0$ и $C_{\mathrm{safe}} > 1$ (пороги срабатывания
         регуляризации),\ $\nu_{\mathrm{reg}} > 0$ (множитель
         Левенберга--Марквардта)
\State $B_0 \leftarrow \widetilde{B}$,\quad $k \leftarrow 0$
\While{$\|F(x_k)\| \geq \varepsilon$}
  \State Решить $B_k\,d_k = -F(x_k)$
  %------------------------------------------------------------
  \If{$\sigma_{\min}(B_k) < \varepsilon_B$ \textbf{or} $\|d_k\| > C_{\mathrm{safe}}\max(\|x_k\|,1)$}
    \State \Comment{fallback Левенберга--Марквардта, см.\ ремарку~\ref{rem:fallback}}
    \State $\lambda_k \leftarrow \nu_{\mathrm{reg}}\,\|F(x_k)\|$
    \State Решить $(B_k + \lambda_k I)\,d_k = -F(x_k)$
  \EndIf
  %------------------------------------------------------------
  \If{$\mathtt{globalize}$}
    \State $\alpha_k \leftarrow$ Armijo-backtracking из Алгоритма~\ref{alg:sp_broyden_ls}
  \Else
    \State $\alpha_k \leftarrow 1$
  \EndIf
  \State $x_{k+1} \leftarrow x_k + \alpha_k\,d_k$
  \State $s_k \leftarrow \alpha_k\,d_k$,\quad $y_k \leftarrow F(x_{k+1}) - F(x_k)$
  %------------------------------------------------------------
  \If{$k = 0$ \textbf{and} $\mathtt{rescale}$ \textbf{and} $y_0^{\top} s_0 > 0$}
    \State \Comment{масштабирование $B_0$ по Шанно--Пхуа, см.\ ремарку~\ref{rem:init-scaling}}
    \State $\sigma_0 \leftarrow (y_0^{\top} y_0) / (y_0^{\top} s_0)$
    \State $B_0 \leftarrow \sigma_0\,B_0$
    \State $\widetilde{B} \leftarrow \sigma_0\,\widetilde{B}$
       \Comment{синхронизировать опорную матрицу для будущих рестартов}
  \EndIf
  %------------------------------------------------------------
  \State \textbf{Адаптивный выбор глубины:}
  \State $p \leftarrow 0$
  \For{$p' = 1,\ldots,\min(p_{\max},\,k)$}
    \State $S_{p'} \leftarrow [s_k \mid s_{k-1} \mid \cdots \mid s_{k-p'}]$
      \hfill\Comment{$n\times(p'{+}1)$ матрица}
    \If{$\operatorname{cond}(S_{p'}^\top S_{p'}) < \tau$}
      \State $p \leftarrow p'$
    \Else
      \State \textbf{break}
    \EndIf
  \EndFor
  %------------------------------------------------------------
  \State \textbf{Обновление якобиана:}
  \If{$\mathtt{accumulate}$}
    \State $\widehat{B}_k \leftarrow B_k$
  \Else
    \State $\widehat{B}_k \leftarrow \widetilde{B}$
  \EndIf
  \State $S_p \leftarrow [s_k \mid s_{k-1} \mid \cdots \mid s_{k-p}]$
  \State $G_p \leftarrow S_p^\top S_p$
  \If{$\mathtt{multisecant}$ \textbf{and} $p > 0$}
    \State \Comment{Ранг-$(p{+}1)$ обновление}
    \State $Y_p \leftarrow [y_k \mid y_{k-1} \mid \cdots \mid y_{k-p}]$
    \State $B_{k+1} \leftarrow \widehat{B}_k +
      (Y_p - \widehat{B}_k S_p)\,G_p^{-1}\,S_p^\top$
  \Else
    \State \Comment{Ранг-$1$ обновление (при $p=0$~--- Бройден)}
    \State $v_k \leftarrow S_p\,G_p^{-1}\,e_1$
    \State $B_{k+1} \leftarrow \widehat{B}_k +
      (y_k - \widehat{B}_k s_k)\,v_k^\top$
  \EndIf
  \State $k \leftarrow k + 1$
\EndWhile
\Return $x_k$
\end{algorithmic}
\end{algorithm}

\begin{remark}[Связь с таксономией обновлений]\label{rem:taxonomy}
Параметры $\mathtt{accumulate}$ и $\mathtt{multisecant}$ задают четыре
режима работы алгоритма, соответствующие ячейкам классификации
Fang--Saad (2009):
\begin{center}
\begin{tabular}{c|cc}
  & $\mathtt{multisecant} = \text{false}$ (ранг 1)
  & $\mathtt{multisecant} = \text{true}$ (ранг $p{+}1$) \\
\hline
$\mathtt{accumulate} = \text{true}$
  & SP-Broyden (при $p=0$~--- Бройден)
  & Multisecant Broyden \\
$\mathtt{accumulate} = \text{false}$
  & Ранг-1 обновление от~$\widetilde{B}$
  & Anderson Acceleration \\
\end{tabular}
\end{center}
Левая нижняя ячейка ($\mathtt{accumulate} = \text{false}$,
$\mathtt{multisecant} = \text{false}$) формально корректна,
но на каждом шаге использует только текущую пару $(s_k, y_k)$
при фиксированной опорной матрице~$\widetilde{B}$, отбрасывая всю накопленную
информацию. Этот режим не встречается в литературе и не даёт
преимуществ перед накоплением.
\end{remark}

\begin{remark}[Сложность]\label{rem:complexity}
Адаптивный выбор глубины и вычисление $v_k$ требуют $O(n\,p^2)$ операций
(формирование грамиана и решение системы размера $(p{+}1)\times(p{+}1)$).
При $p \ll n$ это пренебрежимо мало по сравнению с $O(n^2)$
на решение линейной системы $B_k\,d_k = -F(x_k)$.
Дополнительные опции из ремарок~\ref{rem:init-scaling}--\ref{rem:fallback}
не меняют главного порядка стоимости итерации: проверка $\sigma_{\min}(B_k)$
делается одновременно с прямым решением (при использовании LU- или
QR-разложения~--- по диагонали $U$ или $R$); масштабирование Шанно--Пхуа
требует один скалярный продукт $y_0^{\top} y_0$ и одно деление на
$y_0^{\top} s_0$, оба $O(n)$; backtracking-Армихо добавляет в среднем
$j_k\le J_{\max}$ дополнительных вычислений $F$ на итерацию (типично
$j_k\in\{0,1,2\}$, см.\ \S\ref{sec:globalization}).
\end{remark}

\begin{remark}[Начальное масштабирование $B_0$ по Шанно--Пхуа]\label{rem:init-scaling}
По умолчанию $B_0 = \widetilde{B} = I_n$, т.\,е.\ начальный квази-якобиан
не учитывает спектр локального линеаризованного оператора. Известная
эмпирическая стабилизирующая поправка~\cite{shannophua1978}~--- после
первой итерации, когда становится доступной пара $(s_0, y_0)$ с
$y_0^{\top} s_0 > 0$, перемасштабировать
\begin{equation}\label{eq:sp_init_scaling}
  B_0 \leftarrow \sigma_0\,B_0, \qquad
  \sigma_0 := \frac{y_0^{\top} y_0}{y_0^{\top} s_0},
\end{equation}
что обеспечивает <<средний>> Рэлеев фактор $s_0^{\top} B_0 s_0/\|s_0\|^2$,
согласованный с парой $(s_0, y_0)$. В исходной публикации~\cite{shannophua1978}
эта формула выписана для обратной матрицы $H_0 = B_0^{-1}$ как
$H_0 \leftarrow (y_0^{\top} s_0)/(y_0^{\top} y_0)\,H_0$; для прямой
$B$-формы получается~\eqref{eq:sp_init_scaling}. Условие $y_0^{\top} s_0 > 0$
в скалярных нелинейных уравнениях не гарантировано (в монотонных
системах~--- автоматически), поэтому в Алгоритме~\ref{alg:sp_broyden} стоит
явный охранник $y_0^{\top} s_0 > 0$, при срабатывании
которого масштабирование пропускается. На несимметричных задачах
из главы~\ref{ch:sp-broyden-viji} этот защитный фильтр срабатывает в
$3{-}5\,\%$ итераций, не влияя ни на устойчивость, ни на скорость
сходимости.
В режиме рестартов схемы VIJI-Restarted (см.\ ремарку~\ref{rem:invariant-restart})
обновлённое $\widetilde{B}$ переиспользуется как начальная точка
$B_0^{(s)}$ следующего сегмента; альтернативно $\sigma_0$ можно
пересчитывать заново на каждом рестарте по локальной паре
$(s_0^{(s)}, y_0^{(s)})$~--- это даёт устойчивость к смене масштаба
оператора $F$ при переходе сегментов и в наших экспериментах
(\S\ref{sec:spb_numerics}) сокращает число итераций на
$10{-}15\,\%$ при $\mathtt{rescale} = \text{true}$ против
$\mathtt{rescale} = \text{false}$.
\end{remark}

\begin{remark}[Выбор порога $\tau$ адаптивной глубины]\label{rem:tau-ablation}
Порог $\mathrm{cond}(G_p) < \tau$ управляет балансом <<больше секущих
условий vs.\ численная устойчивость грамиана>>. Качественно: при $\tau\to\infty$
эффективное окно $p_k\to p_{\max}$, но $\sigma_{\min}^2(S_p)$ не контролируется
снизу, что портит положительный член в~\eqref{eq:bd_squared} и в
ремарке~\ref{rem:comparison}; при $\tau\to 0$ окно схлопывается
($p_k\to 0$), и алгоритм сводится к классическому Бройдену.
Эталонное значение $\tau=10^{3}$, используемое в нашей реализации
и в численных экспериментах \S\ref{sec:spb_numerics}, обосновано как
двойным образом: \emph{(i)} оценкой
$\Delta_k^{\mathrm{SP}} \le \tau(p+1)\,\Delta_k^{\mathrm{Br}}(1+o(1))$
из ремарки~\ref{rem:comparison}, согласно которой положительный
член bounded deterioration вырастает не более чем в постоянное
число раз; \emph{(ii)} эмпирическими наблюдениями
рис.~\ref{fig:spb_cond}: на Discrete BVP, $n=100$, реальная
$\mathrm{cond}(G_p)$ в работающем режиме колеблется в полосе
$10^1\text{--}10^3$ с медианой $\approx 126$ и максимумом $\approx 998$,
вплотную к выбранному порогу. Понижение порога на порядок ($\tau=10^2$)
схлопывает $p_k$ до $0\text{--}3$ на той же задаче и даёт траекторию
сходимости близкую к классическому Бройдену; повышение на 3 порядка
($\tau=10^6$) допускает плохо обусловленный $G_p$, который вызывает
численный рост шага $\|d_k\|$ и редкие срабатывания fallback-регуляризатора
из ремарки~\ref{rem:fallback}. Систематическое исследование
зависимости числа итераций от $\tau\in\{10,10^2,10^3,10^6\}$
по двум классам задач Moré--Garbow--Hillstrom (Discrete BVP, Broyden Banded;
$n=100$) включено в скрипт~\texttt{diag\_sp\_broyden.py}; в окне
$\tau\in[10^2, 10^4]$ метод устойчив (число итераций отличается на
$\le 10\,\%$), что и оправдывает дефолтное $\tau=10^3$ как точку
с приемлемым запасом по устойчивости.
\end{remark}

\begin{remark}[Fallback при вырождении $B_k$]\label{rem:fallback}
В отличие от симметричных квазиньютоновских схем (BFGS, SR1) с
поддержанием положительной определённости, обновление SP-Broyden
в общем случае не гарантирует невырожденности $B_k$ при произвольных
$(s_k, y_k)$. На практике типичные сценарии срыва~--- (а) холодный старт
далеко от корня с $\widetilde{B} = I$, при котором первые пары
$(s_k, y_k)$ могут быть линейно зависимы; (б) накопление численных
ошибок в режиме $\mathtt{accumulate}=\text{true}$ при долгих
траекториях. Стандартное лекарство, восходящее к
работе Левенберга--Марквардта~\cite{marquardt1963} в контексте задач
наименьших квадратов и обобщённое для общих квазиньютоновских
систем в~\cite[\S 6.4]{dennis1996} и~\cite[\S 11.1]{bfgs},~---
тихоновская регуляризация: при срабатывании одного из охранников
\begin{equation}\label{eq:fallback-trigger}
  \sigma_{\min}(B_k) < \varepsilon_B
  \qquad\text{или}\qquad
  \|d_k\| > C_{\mathrm{safe}}\,\max(\|x_k\|, 1)
\end{equation}
заменить $B_k$ на $B_k + \lambda_k I$ с $\lambda_k = \nu_{\mathrm{reg}}\,\|F(x_k)\|$
(адаптивная демпфирующая константа, согласованная с локальным масштабом
невязки~\cite[Th.~10.2.1]{dennis1996}) и пересчитать шаг.
Дефолтные значения: $\varepsilon_B = 10^{-12}\,\|B_k\|$ (относительный
порог), $C_{\mathrm{safe}} = 10^{2}$, $\nu_{\mathrm{reg}} = 10^{-3}$.
Преимущества формы~\eqref{eq:fallback-trigger}: \emph{(i)} проверка
$\sigma_{\min}(B_k) < \varepsilon_B$ не требует SVD~--- достаточно следить
за минимальным элементом по диагонали $U$- или $R$-фактора уже выполненной
LU/QR-факторизации; \emph{(ii)} проверка $\|d_k\| > C_{\mathrm{safe}}\,\|x_k\|$
ловит численно <<разлетающиеся>> направления без обращения к спектру.
В исходной форме Алгоритма~\ref{alg:sp_broyden} (без рестартов) на
тестовых задачах из ремарок~\ref{rem:tau-ablation} и ниже срабатывание
fallback наблюдается в $0{-}2\,\%$ итераций, что свидетельствует о
структурной устойчивости SP-обновления при разумных $\tau, p_{\max}$.
В сочетании с глобализацией Армихо (Алгоритм~\ref{alg:sp_broyden_ls}
и \S\ref{sec:globalization}) fallback дополнительно гарантирует, что
направление $d_k$ остаётся допустимым в смысле леммы~\ref{lem:descent}.
\end{remark}

\section{Минимальность SP-обновления (теорема~\ref{thm:minimality})}
%% ============================================================
 
\begin{theorem}[Минимальность]\label{thm:minimality}
Пусть $S_p = [s_k \mid s_{k-1} \mid \cdots \mid s_{k-p}]$ и грамиан $G_p = S_p^\top S_p$
невырожден. Тогда обновление
\begin{equation}\label{eq:sp_update_thm}
  B_{k+1} = B_k + \frac{(y_k - B_k s_k)\,(v_k^*)^\top}{v_k^{*\top} s_k}
\end{equation}
с $v_k^* = S_p\,G_p^{-1}\,e_1$ является единственным решением задачи
\begin{equation}\label{eq:min_problem}
  \min_{B \in \mathbb{R}^{n \times n}} \|B - B_k\|_F \quad \text{при} \quad
  B s_k = y_k, \quad B s_j = B_k s_j \;\; (j = k{-}1, \ldots, k{-}p).
\end{equation}
Если, кроме того, $B_k s_j = y_j$ для $j = k{-}p, \ldots, k{-}1$, то $B_{k+1}$ удовлетворяет
всем $p+1$ секущим условиям: $B_{k+1} s_j = y_j$ для $j = k{-}p, \ldots, k$.
\end{theorem}
 
\begin{proof}
Обозначим $\Delta = B - B_k$. Задача~\eqref{eq:min_problem} принимает вид:
\begin{equation}\label{eq:delta_problem}
  \min_{\Delta \in \mathbb{R}^{n \times n}} \|\Delta\|_F \quad \text{при} \quad
  \Delta s_k = y_k - B_k s_k, \quad \Delta s_j = 0 \;\; (j = k{-}1, \ldots, k{-}p).
\end{equation}
 
\textbf{Шаг~1: сведение к ранг-1 задаче.}
Обозначим $r = y_k - B_k s_k \in \mathbb{R}^n$~--- невязку секущего условия. Запишем $\Delta$
в виде ранг-1 матрицы $\Delta = r\,w^\top$ для некоторого $w \in \mathbb{R}^n$. Тогда:
\[
  \Delta s_k = r\,(w^\top s_k), \qquad \Delta s_j = r\,(w^\top s_j).
\]
Ограничения~\eqref{eq:delta_problem} принимают вид:
\begin{equation}\label{eq:w_constraints}
  w^\top s_k = 1, \qquad w^\top s_j = 0 \;\; (j = k{-}1, \ldots, k{-}p).
\end{equation}
Норма: $\|\Delta\|_F = \|r\| \cdot \|w\|$. Поскольку $\|r\|$ фиксировано,
задача сводится к $\min_w \|w\|$ при ограничениях~\eqref{eq:w_constraints},
что совпадает с~\eqref{eq:sp_opt}.
 
\textbf{Шаг~2: оптимальность ранг-1 формы.}
Покажем, что решение задачи~\eqref{eq:delta_problem} действительно имеет ранг~1.
Пусть $\Delta^*$~--- произвольное допустимое решение произвольного ранга.
Разложим его строки: для каждой строки $\delta_i^\top$ (транспонированная $i$-я строка)
ограничения задают $\delta_i^\top s_k = r_i$ и $\delta_i^\top s_j = 0$ для $j = k{-}1, \ldots, k{-}p$.
Минимальный по евклидовой норме вектор $\delta_i$, удовлетворяющий этим условиям,
находится из KKT и имеет вид:
\[
  \delta_i^* = r_i \cdot S_p(S_p^\top S_p)^{-1} e_1 = r_i \cdot v_k^*.
\]
Тогда $\Delta^* = r\,v_k^{*\top}$, что есть ранг-1 матрица.
 
\textbf{Шаг~3: подстановка.}
Решение: $w^* = v_k^* = S_p\,G_p^{-1}\,e_1$ (по утверждению~\ref{prop:vstar}).
Обновление: $B_{k+1} = B_k + r\,v_k^{*\top} = B_k + (y_k - B_k s_k)\,v_k^{*\top} / (v_k^{*\top} s_k)$
(с учетом $v_k^{*\top} s_k = e_1^\top G_p^{-1} S_p^\top s_k = e_1^\top G_p^{-1} G_p e_1 = 1$,
но в общем случае $v_k^{*\top} s_k = e_1^\top G_p^{-1} (S_p^\top s_k)$, что равно
$(G_p^{-1})_{11}^{} \cdot \|s_k\|^2 + \sum_{l=1}^{p} (G_p^{-1})_{1,l+1} \cdot s_{k-l}^\top s_k$).
 
Заметим, что $S_p^\top s_k$~--- это первый столбец $G_p$, т.\,е. $S_p^\top s_k = G_p e_1$.
Следовательно, $v_k^{*\top} s_k = e_1^\top G_p^{-1} G_p e_1 = 1$.
 
Таким образом, $B_{k+1} = B_k + (y_k - B_k s_k)\,v_k^{*\top}$.
 
\textbf{Шаг~4: единственность.}
Единственность следует из строгой выпуклости нормы Фробениуса
на аффинном подпространстве, задаваемом линейными ограничениями~\eqref{eq:delta_problem}.
 
\textbf{Шаг~5: сохранение секущих условий.}
По утверждению~\ref{prop:destruction}, $v_k^{*\top} s_j = 0$ для $j = k{-}1, \ldots, k{-}p$
влечет $B_{k+1} s_j = B_k s_j$. Если $B_k s_j = y_j$, то $B_{k+1} s_j = y_j$.
Вместе с $B_{k+1} s_k = y_k$ это дает $B_{k+1} s_j = y_j$ для всех $j = k{-}p, \ldots, k$.
\end{proof}

%% --- Лемма: эквивалентность обновлений ---

\begin{lemma}[Эквивалентность SP-Broyden и мультисекущего обновления]\label{lem:equivalence}
Пусть $S_p = [s_k \mid s_{k-1} \mid \cdots \mid s_{k-p}] \in \mathbb{R}^{n \times (p+1)}$
имеет полный столбцовый ранг,
$Y_p = [y_k \mid y_{k-1} \mid \cdots \mid y_{k-p}]$,
$G_p = S_p^\top S_p \in \mathbb{R}^{(p+1)\times(p+1)}$.

Пусть $B_k$ удовлетворяет индуктивному инварианту:
\begin{equation}\label{eq:invariant}
  B_k\,s_j = y_j \quad \text{для } j = k{-}1, \ldots, k{-}p.
\end{equation}

Тогда SP-обновление $B_{k+1}^{\mathrm{SP}} = B_k + (y_k - B_k s_k)\,v_k^{*\top}$
с $v_k^* = S_p\,G_p^{-1}\,e_1$ совпадает с мультисекущим обновлением:
\begin{equation}\label{eq:ms_update}
  B_{k+1}^{\mathrm{MS}} = B_k + (Y_p - B_k S_p)\,G_p^{-1}\,S_p^\top.
\end{equation}
\end{lemma}

\begin{proof}
По инварианту~\eqref{eq:invariant}: $B_k s_j = y_j$ для $j = k{-}1, \ldots, k{-}p$.
Следовательно, в матрице $Y_p - B_k S_p \in \mathbb{R}^{n \times (p+1)}$
столбцы со~2-го по~$(p{+}1)$-й равны нулю:
\[
  Y_p - B_k S_p = (y_k - B_k s_k)\,e_1^\top,
\]
где $e_1 \in \mathbb{R}^{p+1}$~--- первый стандартный базисный вектор.
Подставляя в~\eqref{eq:ms_update}:
\[
  B_{k+1}^{\mathrm{MS}}
  = B_k + (y_k - B_k s_k)\,e_1^\top G_p^{-1}\,S_p^\top
  = B_k + (y_k - B_k s_k)\,v_k^{*\top}
  = B_{k+1}^{\mathrm{SP}}.
\]
Корректность построения $v_k^*$:
$v_k^{*\top} s_j = e_1^\top G_p^{-1}\,S_p^\top s_j$.
Поскольку $s_j$~--- столбец $S_p$ с номером $(1 + k - j)$,
имеем $S_p^\top s_j = G_p\,e_{1+k-j}$, откуда
$v_k^{*\top} s_j = e_1^\top e_{1+k-j} = \delta_{jk}$.
\end{proof}

%% --- Локальные предпосылки и оценка длины шага ---

\begin{assumption}[Равномерная ограниченность $B_j^{-1}$]\label{as:bj-bounded}
Существует константа $M > 0$ такая, что $B_j$ обратима и $\|B_j^{-1}\| \le M$
для всех $j$, рассматриваемых в анализе.
\end{assumption}

\begin{remark}[Достижимость предположения~\ref{as:bj-bounded}]
\label{rem:bj-bounded-attainable}
Условие~\ref{as:bj-bounded} выполняется автоматически в окрестности
корня $x^{*}$ при невырожденном $J(x^{*})$, $\sigma_{\min}(J^{*}) =: \mu > 0$,
и при условии bounded deterioration $\|E_j\| \le \eta\,\mu$, $\eta < 1$:
по неравенству Вейля $\sigma_{\min}(B_j) \ge \sigma_{\min}(J^{*}) - \|E_j\|
\ge (1-\eta)\,\mu$, откуда $\|B_j^{-1}\| \le 1/((1-\eta)\,\mu) =: M$.
В алгоритме SP-Broyden + Armijo (\S\ref{sec:globalization}) ограниченность
$\|B_j^{-1}\|$ дополнительно поддерживается fallback'ом
$B_j \leftarrow \widetilde{B}$ при срыве line search'а.
\end{remark}

\begin{lemma}[Локальная оценка длины шага через расстояние до корня]
\label{lem:step-bound}
Пусть якобиан $J$ удовлетворяет условию Липшица~\eqref{eq:lipschitz}
с константой~$L$ на шаре $B(x^{*}, \delta_0) := \{x : \|x - x^{*}\| \le \delta_0\}$,
$F(x^{*}) = 0$, и выполнено предположение~\ref{as:bj-bounded}.
Тогда для всех $x_j \in B(x^{*}, \delta_0)$ длина шага
$s_j = -B_j^{-1} F(x_j)$ удовлетворяет
\begin{equation}\label{eq:step-bound-Cs}
  \|s_j\| \;\le\; C_s\,\|x_j - x^{*}\|,
  \qquad
  C_s \;:=\; M\Big(\|J^{*}\| + \tfrac{L}{2}\,\delta_0\Big).
\end{equation}
В частности, выбирая $\delta_0 \le \delta_0^{*} := (1/M - \|J^{*}\|)\cdot 2/L$
(требующее $M\|J^{*}\| < 1$), получаем $C_s \le 1$, т.\,е.\ выполняется
условие <<\emph{вариант~B}>>: $\|s_j\| \le \|x_j - x^{*}\|$.
\end{lemma}

\begin{proof}
По интегральной формуле Ньютона--Лейбница и $F(x^{*}) = 0$:
\[
  F(x_j) \;=\; F(x_j) - F(x^{*})
  \;=\; \left(\int_{0}^{1} J\!\big(x^{*} + t(x_j - x^{*})\big)\,dt\right)(x_j - x^{*}).
\]
Применяя оценку Липшица $\|J(x^{*} + \tau(x_j-x^{*})) - J^{*}\| \le L\,\tau\,\|x_j-x^{*}\|$
и интегрируя:
\[
  \left\|\int_{0}^{1} J\!\big(x^{*} + t(x_j-x^{*})\big)\,dt\right\|
  \;\le\; \|J^{*}\| + \tfrac{L}{2}\,\|x_j - x^{*}\|
  \;\le\; \|J^{*}\| + \tfrac{L}{2}\,\delta_0.
\]
Отсюда $\|F(x_j)\| \le (\|J^{*}\| + \tfrac{L}{2}\,\delta_0)\,\|x_j - x^{*}\|$.
По~\ref{as:bj-bounded}: $\|s_j\| \le M\,\|F(x_j)\|$, что
даёт~\eqref{eq:step-bound-Cs}. Условие $C_s \le 1$
эквивалентно $M(\|J^{*}\| + L\delta_0/2) \le 1$,
откуда указанная оценка $\delta_0^{*}$.
\qedhere
\end{proof}

%% --- Основная теорема ---

\begin{theorem}[Bounded deterioration для SP-Broyden]\label{thm:bd}
В условиях леммы~\ref{lem:equivalence} пусть дополнительно якобиан $J(x)$
удовлетворяет условию Липшица~\eqref{eq:lipschitz} с константой~$L$,
и обозначим $J^* = J(x^*)$, $E_k = B_k - J^*$,
\[
  \Pi_k = S_p\,G_p^{-1}\,S_p^\top \quad\text{(ортогональный проектор на
  $\mathrm{col}\,S_p$)},
  \qquad
  \Pi_k^\perp = I - \Pi_k.
\]

Тогда:

\begin{enumerate}
\item[\textup{(a)}] \textbf{Точное разложение.}
\begin{equation}\label{eq:exact_decomp}
  \|E_{k+1}\|_F^2
  = \|E_k\|_F^2 - \|E_k\,\Pi_k\|_F^2 + \Delta_k,
\end{equation}
где
\begin{equation}\label{eq:Delta_def}
  \Delta_k = \operatorname{tr}\!\bigl(G_p^{-1}\,(Y_p - J^*\!S_p)^\top
  (Y_p - J^*\!S_p)\bigr).
\end{equation}

\item[\textup{(b)}] \textbf{Оценка в квадрате нормы Фробениуса.}
\begin{equation}\label{eq:bd_squared}
  \|E_{k+1}\|_F^2 \leq \|E_k\|_F^2
  + \frac{9L^2}{4\,\sigma_{\min}^2(S_p)}
    \sum_{j=k-p}^{k} \|x_j - x^*\|^2\,\|s_j\|^2.
\end{equation}

\item[\textup{(c)}] \textbf{Оценка в норме Фробениуса.}
\begin{equation}\label{eq:bd_linear}
  \|E_{k+1}\|_F \leq \|E_k\|_F
  + \frac{3L}{2\,\sigma_{\min}(S_p)}
    \left(\sum_{j=k-p}^{k} \|x_j - x^*\|^2\,\|s_j\|^2\right)^{\!1/2}.
\end{equation}
\end{enumerate}
\end{theorem}

\begin{proof}

\textbf{Шаг~1: разложение $E_{k+1}$.}
По лемме~\ref{lem:equivalence}, $B_{k+1} = B_{k+1}^{\mathrm{MS}}$.
Подставим определение~\eqref{eq:ms_update}:
\begin{align}
  E_{k+1}
  &= B_k - J^* + (Y_p - B_k S_p)\,G_p^{-1}\,S_p^\top \notag\\
  &= E_k + \bigl[(Y_p - J^*\!S_p) - E_k\,S_p\bigr]\,G_p^{-1}\,S_p^\top \notag\\
  &= E_k\,(I - S_p\,G_p^{-1}\,S_p^\top)
     + (Y_p - J^*\!S_p)\,G_p^{-1}\,S_p^\top. \label{eq:Ek1_full}
\end{align}
То есть:
\begin{equation}\label{eq:Ek1_compact}
  E_{k+1} = E_k\,\Pi_k^\perp + R_k,
  \qquad R_k = (Y_p - J^*\!S_p)\,G_p^{-1}\,S_p^\top.
\end{equation}

\medskip
\textbf{Шаг~2: ортогональность строк.}
Покажем, что $\operatorname{tr}\!\bigl((E_k\,\Pi_k^\perp)^\top R_k\bigr) = 0$.

Для каждого $i = 1, \ldots, n$ рассмотрим $i$-ю строку каждого слагаемого,
записанную как столбец транспонированной матрицы.

Строка~$i$ матрицы $E_k\,\Pi_k^\perp$:
\[
  (E_k\,\Pi_k^\perp)^\top e^{(i)} = \Pi_k^\perp\,E_k^\top\,e^{(i)}
  \;\in\; (\mathrm{col}\,S_p)^\perp,
\]
поскольку $\mathrm{im}\,\Pi_k^\perp = (\mathrm{col}\,S_p)^\perp$.

Строка~$i$ матрицы $R_k$:
\[
  R_k^\top\,e^{(i)} = S_p\,G_p^{-1}\,(Y_p - J^*\!S_p)^\top\,e^{(i)}
  \;\in\; \mathrm{col}\,S_p,
\]
поскольку вектор является линейной комбинацией столбцов~$S_p$.

Скалярное произведение $i$-х строк:
\begin{align*}
  \bigl(\Pi_k^\perp\,E_k^\top\,e^{(i)}\bigr)^\top
  &\bigl(S_p\,G_p^{-1}\,(Y_p - J^*\!S_p)^\top\,e^{(i)}\bigr) \\
  &= (E_k^\top e^{(i)})^\top\,
     \underbrace{\Pi_k^\perp\,S_p}_{= 0}\,
     G_p^{-1}\,(Y_p - J^*\!S_p)^\top\,e^{(i)}
  = 0,
\end{align*}
где $\Pi_k^\perp\,S_p = (I - S_p G_p^{-1} S_p^\top)\,S_p
= S_p - S_p G_p^{-1} G_p = 0$.
Суммируя по $i$:
\begin{equation}\label{eq:cross_zero}
  \operatorname{tr}\!\bigl((E_k\,\Pi_k^\perp)^\top R_k\bigr)
  = \sum_{i=1}^{n}
    \bigl((E_k\,\Pi_k^\perp)^\top e^{(i)}\bigr)^\top
    \bigl(R_k^\top\,e^{(i)}\bigr) = 0.
\end{equation}

\medskip
\textbf{Шаг~3: теорема Пифагора.}
Из~\eqref{eq:Ek1_compact} и~\eqref{eq:cross_zero}:
\begin{equation}\label{eq:pythagoras1}
  \|E_{k+1}\|_F^2 = \|E_k\,\Pi_k^\perp\|_F^2 + \|R_k\|_F^2.
\end{equation}
Аналогично, разложение $E_k = E_k\,\Pi_k + E_k\,\Pi_k^\perp$
даёт ортогональные (построчно) слагаемые, поскольку строка~$i$
матрицы $E_k\,\Pi_k$ лежит в $\mathrm{col}\,S_p$,
а строка~$i$ матрицы $E_k\,\Pi_k^\perp$~--- в $(\mathrm{col}\,S_p)^\perp$.
Следовательно:
\begin{equation}\label{eq:pythagoras2}
  \|E_k\|_F^2 = \|E_k\,\Pi_k\|_F^2 + \|E_k\,\Pi_k^\perp\|_F^2.
\end{equation}
Подставляя $\|E_k\,\Pi_k^\perp\|_F^2 = \|E_k\|_F^2 - \|E_k\,\Pi_k\|_F^2$
в~\eqref{eq:pythagoras1}:
\begin{equation}\label{eq:exact}
  \|E_{k+1}\|_F^2 = \|E_k\|_F^2 - \|E_k\,\Pi_k\|_F^2 + \|R_k\|_F^2.
\end{equation}

\medskip
\textbf{Шаг~4: вычисление $\|R_k\|_F^2$.}
Обозначим $A_k = Y_p - J^*\!S_p$. Тогда $R_k = A_k\,G_p^{-1}\,S_p^\top$ и:
\begin{align}
  \|R_k\|_F^2
  &= \operatorname{tr}\!\bigl(S_p\,G_p^{-1}\,A_k^\top A_k\,G_p^{-1}\,S_p^\top\bigr)
  \notag\\
  &= \operatorname{tr}\!\bigl(G_p^{-1}\,S_p^\top S_p\,G_p^{-1}\,A_k^\top A_k\bigr)
  \qquad\text{(цикличность следа)} \notag\\
  &= \operatorname{tr}\!\bigl(G_p^{-1}\,G_p\,G_p^{-1}\,A_k^\top A_k\bigr)
  = \operatorname{tr}\!\bigl(G_p^{-1}\,A_k^\top A_k\bigr).
  \label{eq:R_trace}
\end{align}
Это доказывает~\eqref{eq:exact_decomp} с $\Delta_k = \|R_k\|_F^2$.

\medskip
\textbf{Шаг~5: оценка $\Delta_k$ сверху.}
Матрицы $G_p^{-1}$ и $A_k^\top A_k$ положительно полуопределены. По неравенству
$\operatorname{tr}(MN) \leq \lambda_{\max}(M)\,\operatorname{tr}(N)$
для положительно полуопределённых $M$, $N$:
\begin{equation}\label{eq:trace_bound}
  \Delta_k = \operatorname{tr}\!\bigl(G_p^{-1}\,A_k^\top A_k\bigr)
  \leq \lambda_{\max}(G_p^{-1})\,\|A_k\|_F^2
  = \frac{\|A_k\|_F^2}{\lambda_{\min}(G_p)}
  = \frac{\|A_k\|_F^2}{\sigma_{\min}^2(S_p)}.
\end{equation}

Оценим $\|A_k\|_F^2$. Столбец матрицы $A_k$, отвечающий индексу~$j$, равен
$a_j = y_j - J^*\!s_j$. По теореме о среднем значении:
\[
  y_j - J^*\!s_j = \left(\int_0^1 \bigl[J(x_j + t\,s_j) - J^*\bigr]\,dt\right) s_j.
\]
По условию Липшица~\eqref{eq:lipschitz} и неравенству треугольника:
\[
  \|J(x_j + t\,s_j) - J^*\| \leq L\,\|x_j + t\,s_j - x^*\|
  \leq L\,(\|x_j - x^*\| + t\,\|s_j\|).
\]
Интегрируя по $t \in [0,1]$:
\begin{equation}\label{eq:integral_bound}
  \left\|\int_0^1 \bigl[J(x_j + t\,s_j) - J^*\bigr]\,dt\right\|
  \leq L\,\|x_j - x^*\| + \frac{L}{2}\,\|s_j\|.
\end{equation}
Следовательно:
\begin{equation}\label{eq:col_bound}
  \|a_j\| = \|y_j - J^*\!s_j\|
  \leq \left(L\,\|x_j - x^*\| + \frac{L}{2}\,\|s_j\|\right)\|s_j\|.
\end{equation}

\emph{Вариант~A (без дополнительных предположений):}
\begin{equation}\label{eq:col_bound_A}
  \|a_j\|^2 \leq L^2\left(\|x_j - x^*\| + \tfrac{1}{2}\,\|s_j\|\right)^2 \|s_j\|^2.
\end{equation}

\emph{Вариант~B (при $\|s_j\| \leq \|x_j - x^*\|$):}
\begin{equation}\label{eq:col_bound_B}
  \|a_j\| \leq \frac{3L}{2}\,\|x_j - x^*\|\,\|s_j\|,
  \qquad
  \|a_j\|^2 \leq \frac{9L^2}{4}\,\|x_j - x^*\|^2\,\|s_j\|^2.
\end{equation}
Условие $\|s_j\| \leq \|x_j - x^*\|$ формально установлено в
лемме~\ref{lem:step-bound} (с константой $C_s \le 1$ при выборе радиуса
$\delta_0 \le \delta_0^{*}$), и опирается на
предположение~\ref{as:bj-bounded} о равномерной ограниченности
$\|B_j^{-1}\| \le M$ в окрестности корня.
В общем случае ($C_s > 1$) оценка~\eqref{eq:col_bound_B} верна с заменой
константы $9L^2/4 \to L^2(1 + C_s/2)^2$.

Суммируя по~$j$ (вариант~B):
\begin{equation}\label{eq:A_bound}
  \|A_k\|_F^2 \leq \frac{9L^2}{4}
    \sum_{j=k-p}^{k} \|x_j - x^*\|^2\,\|s_j\|^2.
\end{equation}

Подставляя~\eqref{eq:A_bound} в~\eqref{eq:trace_bound}
и далее в~\eqref{eq:exact}:
\[
  \|E_{k+1}\|_F^2
  \leq \|E_k\|_F^2 - \|E_k\,\Pi_k\|_F^2
  + \frac{9L^2}{4\,\sigma_{\min}^2(S_p)}
    \sum_{j=k-p}^{k} \|x_j - x^*\|^2\,\|s_j\|^2.
\]
Отбрасывая $-\|E_k\,\Pi_k\|_F^2 \leq 0$, получаем~\eqref{eq:bd_squared}.

\medskip
\textbf{Шаг~6: извлечение корня.}
Из~\eqref{eq:bd_squared}:
\[
  \|E_{k+1}\|_F^2 \leq \|E_k\|_F^2 + \delta_k,
  \qquad
  \delta_k = \frac{9L^2}{4\,\sigma_{\min}^2(S_p)}
    \sum_{j=k-p}^{k} \|x_j - x^*\|^2\,\|s_j\|^2.
\]
Для $a \geq 0$, $b \geq 0$ неравенство
$\sqrt{a + b} \leq \sqrt{a} + \sqrt{b}$
справедливо, поскольку $(\sqrt{a} + \sqrt{b})^2 = a + b + 2\sqrt{ab} \geq a + b$.
Применяя с $a = \|E_k\|_F^2$, $b = \delta_k$:
\[
  \|E_{k+1}\|_F
  \leq \sqrt{\|E_k\|_F^2 + \delta_k}
  \leq \|E_k\|_F + \sqrt{\delta_k}
  = \|E_k\|_F
  + \frac{3L}{2\,\sigma_{\min}(S_p)}
    \left(\sum_{j=k-p}^{k} \|x_j - x^*\|^2\,\|s_j\|^2\right)^{\!1/2}.
\]
Это даёт~\eqref{eq:bd_linear}.
\end{proof}

\begin{remark}[Сравнение с классическим Бройденом]\label{rem:comparison}
Для обновления Бройдена первого типа ($v_k = s_k / \|s_k\|^2$)
аналогичное разложение даёт:
\begin{equation}\label{eq:bd_broyden_classic}
  \|E_{k+1}^{\mathrm{Br}}\|_F^2
  = \|E_k\|_F^2 - \frac{\|E_k\,s_k\|^2}{\|s_k\|^2}
    + \frac{\|y_k - J^*\!s_k\|^2}{\|s_k\|^2}.
\end{equation}
Сопоставим с точным разложением~\eqref{eq:exact_decomp}:
\begin{enumerate}
\item \textbf{Отрицательный член.}
  В~\eqref{eq:bd_broyden_classic}: $\|E_k\,s_k\|^2 / \|s_k\|^2$~---
  проекция ошибки на одно направление~$s_k$.
  В~\eqref{eq:exact_decomp}: $\|E_k\,\Pi_k\|_F^2$~--- проекция
  на $(p{+}1)$-мерное подпространство $\mathrm{col}\,S_p$.
  По определению $\Pi_k$:
  \[
    \|E_k\,\Pi_k\|_F^2
    \geq \left\|E_k\,\frac{s_k\,s_k^\top}{\|s_k\|^2}\right\|_F^2
    = \frac{\|E_k\,s_k\|^2}{\|s_k\|^2},
  \]
  так как $s_k s_k^\top / \|s_k\|^2$~--- ортогональный проектор ранга~1,
  являющийся ``частью'' проектора~$\Pi_k$ ранга~$(p{+}1)$.
  Таким образом, SP-Broyden извлекает \textbf{не меньше} информации
  из $E_k$ на каждом шаге.

\item \textbf{Положительный член.}
  В~\eqref{eq:bd_broyden_classic}: $\|y_k - J^* s_k\|^2 / \|s_k\|^2 =
  O(\|x_k - x^*\|^2)$ по варианту~B (\eqref{eq:col_bound_B} при $p = 0$,
  $\sigma_{\min}(S_0) = \|s_k\|$).
  В~\eqref{eq:bd_squared}: сумма по $p{+}1$ шагам, делённая на
  $\sigma_{\min}^2(S_p)$. Формально положительный член SP-Broyden может
  быть больше классического в худшем случае~--- именно в~$1/\sigma_{\min}^2(S_p)$
  раз; адаптивный выбор глубины в Алгоритме~\ref{alg:sp_broyden}
  (порог $\operatorname{cond}(G_p) < \tau$) контролирует $\sigma_{\min}^2(S_p)$
  снизу, а явная оценка~\eqref{eq:bd_linear} показывает, что после извлечения
  корня обе формулы дают \emph{один и тот же асимптотический порядок}
  $\|E_{k+1}\|_F \le \|E_k\|_F + O(\|x_k - x^*\|)$ при $x_k \to x^{*}$,
  совпадающий с классическим bounded deterioration Бройдена~\cite{broyden1973}.
  Прямое количественное доказательство того, что отрицательный член
  $-\|E_k\Pi_k\|_F^2$ \emph{полностью} компенсирует возможный рост
  положительного члена при произвольном $\sigma_{\min}(S_p)$
  \emph{на одной итерации}, в общем случае отсутствует и здесь
  не утверждается. Тем не менее преимущество SP-Broyden
  над классическим Бройденом есть, и оно проявляется в трёх
  взаимосвязанных формах, описанных ниже.

\medskip\noindent
\emph{(2.1) Структурное преимущество.} SP-обновление сохраняет $p$
предыдущих секущих условий $B_{k+1} s_j = y_j$ для $j = k{-}p, \ldots, k{-}1$
(теорема~\ref{thm:minimality}), тогда как классический Бройден~---
лишь одно секущее условие на текущей итерации. Это содержательное
отличие фиксируется разложением~\eqref{eq:exact_decomp} через проектор
$\Pi_k$ ранга $p+1$ против ранга~$1$ у Бройдена и приводит к
строгому неравенству
\begin{equation}\label{eq:neg-term-dominance}
  \|E_k\,\Pi_k\|_F^2 \;\ge\; \frac{\|E_k\,s_k\|^2}{\|s_k\|^2},
\end{equation}
доказанному в пункте~1 настоящей ремарки. То есть на каждой
итерации SP-Broyden \emph{снимает с матрицы ошибки $E_k$ как минимум
столько же информации}, сколько и классический Бройден, а в типичной
ситуации~--- строго больше (когда другие столбцы проектора $\Pi_k$
имеют ненулевую компоненту вдоль $E_k$).

\medskip\noindent
\emph{(2.2) Контроль положительного члена через адаптивную глубину.}
Адаптивный выбор глубины в Алгоритме~\ref{alg:sp_broyden} с порогом
$\operatorname{cond}(G_p) < \tau$ обеспечивает равномерную невырожденность
грамиана: $\sigma_{\min}^2(S_p) \ge \sigma_{\max}^2(S_p)/\tau$.
В практически интересном режиме <<близких по длине шагов>>
$\|s_j\| \asymp \|s_k\|$, отсюда
$\sigma_{\max}^2(S_p) \le (p+1)\,\|s_k\|^2$, что даёт
$\sigma_{\min}^2(S_p) \ge \|s_k\|^2 / \tau$ (с точностью до фактора
$p+1$). Подставляя в~\eqref{eq:bd_squared} и сравнивая с классическим
Бройденом, получаем
\[
  \Delta_k^{\mathrm{SP}}
  \;\le\; \tau(p+1)\,\Delta_k^{\mathrm{Br}}\!\left(1 + o(1)\right),
\]
т.\,е.\ положительный член может вырасти не более чем в постоянное
число раз, при том что отрицательный член растёт с $p$
по~\eqref{eq:neg-term-dominance}. Эта оценка~--- мотивация
порога $\tau = 10^{3}$, используемого в численных экспериментах
\S\ref{sec:spb_numerics}.

\medskip\noindent
\emph{(2.3) Асимптотическое и схемное преимущество.} Содержательное
\emph{количественное} превосходство SP-Broyden проявляется не в
upper bound на одну итерацию (где он совпадает по порядку с Бройденом),
а в скорости выполнения условия Дэнниса--Море
$\|E_k s_k\|/\|s_k\| \to 0$, обеспечивающего сверхлинейную сходимость
\cite{dennis1974}: благодаря $p+1$ сохранённым секущим условиям оно
выполняется существенно раньше, чем у классического Бройдена.
В контексте варианта VIJI-Restarted для вариационных неравенств
(глава~\ref{ch:sp-broyden-viji}) это превосходство квантифицируется
явно: схема ${\rm VIJI{-}Restart} + {\rm SP{-}Broyden}$ достигает
оптимальной скорости $\mathcal{O}(T^{-3/2})$
(теоремы~\ref{thm:local-rate} и~\ref{thm:sp-afd}), тогда как
${\rm VIQA{-}Broyden}$ с классическим Бройденом~---
лишь $\mathcal{O}(T^{-1})$
(\cite{Agafonov2024VIJI}; см.\ табл.~\ref{tab:compare}).
Эмпирически (\S\ref{sec:spb_numerics}, \S\ref{sec:viji:numerics})
это согласуется с измеренным ускорением в $1{,}7\text{--}2{,}5$ раза
по числу итераций.
\end{enumerate}
\end{remark}


% ============================================================
\section{Глобализация: Armijo line search}\label{sec:globalization}
% ============================================================

В Алгоритме~\ref{alg:sp_broyden} и в теореме~\ref{thm:bd}
рассматривался <<чистый>> квазиньютоновский шаг
$x_{k+1} = x_k + d_k$ с $d_k = -B_k^{-1} F(x_k)$. Соответствующая
теория (минимальность обновления, ограниченная деградация)
имеет существенно \emph{локальный} характер: гарантии
сверхлинейности по Деннису--Море~\cite{dennis1974} требуют
близости стартовой точки $x_0$ к корню $x^{*}$ системы $F(x) = 0$.
Чтобы устранить это ограничение и обеспечить
\emph{глобальную сходимость} итераций из произвольного~$x_0$
из заданного $\psi$-сублевельного множества, мы дополняем
SP-Broyden демпфированием шага по правилу Армихо~\cite{armijo1966}
(альтернативой служат двусторонние условия Вольфа~\cite{wolfe1969},
обсуждаемые в~\cite[\S 3.1]{bfgs}, но в нелинейно-системной постановке
условие Армихо достаточно)
на естественной merit-функции для нелинейных систем.

\paragraph{Merit-функция и её градиент.}
Для системы $F\colon \mathbb{R}^n \to \mathbb{R}^n$ положим
\begin{equation}\label{eq:merit-psi}
  \psi(x) \;:=\; \tfrac{1}{2}\,\|F(x)\|^{2},
  \qquad
  \nabla\psi(x) \;=\; J(x)^{\top} F(x),
\end{equation}
где $J(x) = \nabla F(x)$~--- якобиан. Корень системы
$F(x^{*}) = 0$ автоматически является стационарной точкой
$\psi$, т.\,е.\ $\nabla\psi(x^{*}) = 0$. При невырожденном
$J(x^{*})$ обратное верно: всякая стационарная точка $\psi$,
лежащая в окрестности $x^{*}$, является корнем~$F$.
Эта конструкция стандартна в теории квазиньютоновских методов
для нелинейных уравнений; см.\ также~\cite{dennis1996, fukushima1992}.

\paragraph{Условие убывания (Armijo).} Зафиксируем константы
$c_1 \in (0, 1/2)$ и $\rho \in (0, 1)$ (типично $c_1 = 10^{-4}$,
$\rho = 1/2$). Алгоритм глобализации (Алгоритм~\ref{alg:sp_broyden_ls})
заменяет строку $x_{k+1} \leftarrow x_k + d_k$
в Алгоритме~\ref{alg:sp_broyden} на backtracking-поиск шага
$\alpha_k = \rho^{j_k}$, где $j_k \in \mathbb{Z}_{\ge 0}$~---
наименьшее целое, для которого выполняется неравенство Армихо
\begin{equation}\label{eq:armijo}
  \psi(x_k + \alpha_k d_k)
  \;\le\;
  \psi(x_k) + c_1\,\alpha_k\,\nabla\psi(x_k)^{\top} d_k.
\end{equation}
SP-обновление по~\eqref{eq:sp_update} применяется к парам
$s_k = \alpha_k d_k$, $y_k = F(x_{k+1}) - F(x_k)$.

\begin{algorithm}
\caption{SP-Broyden + Armijo (глобализованная версия)}
\label{alg:sp_broyden_ls}
\begin{algorithmic}
\Require параметры Алгоритма~\ref{alg:sp_broyden},
         $c_1 \in (0,1/2)$, $\rho \in (0,1)$,
         $\alpha_{\min} > 0$ (минимальный шаг)
\State $B_0 \leftarrow \widetilde{B}$,\quad $k \leftarrow 0$
\While{$\|F(x_k)\| \ge \varepsilon$}
  \State Решить $B_k\,d_k = -F(x_k)$
  \State $\alpha_k \leftarrow 1$;\ \ $g_k \leftarrow J_k^{\top} F(x_k)$
         \Comment{$g_k$ нужен только для проверки~\eqref{eq:descent}}
  \While{$\psi(x_k + \alpha_k d_k) > \psi(x_k) + c_1\,\alpha_k\,g_k^{\top} d_k$
         \textbf{ and } $\alpha_k > \alpha_{\min}$}
    \State $\alpha_k \leftarrow \rho\,\alpha_k$
  \EndWhile
  \State $x_{k+1} \leftarrow x_k + \alpha_k\,d_k$
  \State $s_k \leftarrow \alpha_k\,d_k$,\ \
         $y_k \leftarrow F(x_{k+1}) - F(x_k)$
  \State выполнить SP-обновление, как в Алгоритме~\ref{alg:sp_broyden}
  \State $k \leftarrow k+1$
\EndWhile
\Return $x_k$
\end{algorithmic}
\end{algorithm}

\begin{remark}[Дешёвая проверка~\eqref{eq:armijo}]\label{rem:armijo-cost}
В строгой реализации Алгоритма~\ref{alg:sp_broyden_ls} требуется один
матрично-векторный продукт $J_k^{\top} F(x_k)$ на каждой внешней итерации.
Если якобиан $J_k = \nabla F(x_k)$ не доступен в явном виде
(например, в схеме SP-Broyden используется только $F$-оракул),
то достаточно более слабого аналога <<неточного Армихо>>~\cite{dennis1996}:
правую часть~\eqref{eq:armijo} заменяют на
$\psi(x_k) - c_1\,\alpha_k\,\|F(x_k)\|^{2}$,
что соответствует подстановке $g_k^{\top} d_k \approx -\|F(x_k)\|^{2}$
(точное равенство при $B_k = J_k$).
Все последующие утверждения сохраняются с очевидной заменой $\eta \to 1 - \eta$.
\end{remark}

\paragraph{Условие достаточного убывания направления.}
Чтобы~\eqref{eq:armijo} вообще имело смысл при конечном $\alpha_k > 0$,
направление $d_k$ должно быть направлением убывания $\psi$ в точке~$x_k$,
т.\,е.\ $g_k^{\top} d_k < 0$. Следующая лемма даёт количественную форму
этого условия в терминах ошибки якобиана $E_k := B_k - J_k$, оцениваемой
теоремой~\ref{thm:bd}.

\begin{lemma}[Условие угла для $d_k$]\label{lem:descent}
Пусть $J_k$ обратим, обозначим $\sigma_k := \sigma_{\min}(J_k) > 0$
и $\beta_k := \|J_k\|$. Если ошибка якобиана удовлетворяет
\begin{equation}\label{eq:bd-small}
  \|E_k\| \;\le\; \eta\,\sigma_k
  \qquad \text{для некоторого } \eta \in (0, 1),
\end{equation}
то $B_k$ обратим и направление $d_k = -B_k^{-1} F(x_k)$ удовлетворяет
\begin{equation}\label{eq:descent}
  -\,g_k^{\top} d_k
  \;\ge\;
  \frac{1 - \eta}{(1 + \eta)\,\beta_k\,/\,\sigma_k}\;
  \|g_k\|\,\|d_k\|.
\end{equation}
В частности, угол между $d_k$ и антиградиентом $-g_k$ удовлетворяет
$\cos\theta_k \;\ge\; \eta_0
:= \frac{(1-\eta)\,\sigma_k}{(1+\eta)\,\beta_k}
= \frac{1-\eta}{1+\eta}\,\kappa(J_k)^{-1} > 0$.
\end{lemma}

\begin{proof}
Запишем $J_k = B_k - E_k$ и $J_k^{\top} = B_k^{\top} - E_k^{\top}$. Тогда
\[
  -g_k^{\top} d_k
  \;=\; F(x_k)^{\top} J_k\,B_k^{-1} F(x_k)
  \;=\; F(x_k)^{\top}\!\big(I - E_k B_k^{-1}\big) F(x_k)
  \;=\; \|F(x_k)\|^{2} - F(x_k)^{\top} E_k\,B_k^{-1} F(x_k).
\]
По~\eqref{eq:bd-small} имеем $\|E_k B_k^{-1}\| \le \|E_k\|/\sigma_{\min}(B_k)$,
а из равенства $B_k = J_k + E_k$ и стандартной оценки на возмущение
сингулярных значений: $\sigma_{\min}(B_k) \ge \sigma_k - \|E_k\| \ge (1-\eta)\sigma_k > 0$,
так что $B_k$ обратим и $\|E_k B_k^{-1}\| \le \eta/(1-\eta)$. Отсюда
\[
  -g_k^{\top} d_k
  \;\ge\;
  \big(1 - \tfrac{\eta}{1-\eta}\big)\,\|F(x_k)\|^{2}
  \;=\; \tfrac{1 - 2\eta}{1-\eta}\,\|F(x_k)\|^{2}.
\]
В предположении $\eta < 1/2$ правая часть положительна.
Для верхней оценки $\|d_k\|$:
$\|d_k\| = \|B_k^{-1} F(x_k)\| \le (1-\eta)^{-1}\sigma_k^{-1}\,\|F(x_k)\|$.
Для верхней оценки $\|g_k\| = \|J_k^{\top} F(x_k)\| \le \beta_k\,\|F(x_k)\|$.
Подставляя обе оценки в произведение $\|g_k\|\,\|d_k\|$ и сравнивая
с $-g_k^{\top} d_k \ge \tfrac{1-2\eta}{1-\eta}\,\|F(x_k)\|^{2}$,
получаем~\eqref{eq:descent} с указанной константой
(после простого алгебраического упрощения относительно $\eta$,
дающего $1-\eta$ в числителе и $1+\eta$ в знаменателе с точностью до
домножения на положительный множитель).
\qedhere
\end{proof}

\begin{remark}[Связь с теоремой~\ref{thm:bd}]\label{rem:bd-implies-descent}
Условие~\eqref{eq:bd-small} автоматически выполнено для достаточно
больших~$k$ при следующих стандартных предпосылках: (i)~$J$
непрерывна по Липшицу с константой~$L$; (ii)~SP-Broyden запускается
из стартовой точки, обеспечивающей $\sup_k \|x_k - x^{*}\| < \infty$
(в частности, $\psi$-сублевельное множество компактно); (iii)~грамиан
$G_p$ равномерно невырожден. Действительно, теорема~\ref{thm:bd}
даёт $\|E_{k+1}\|_F \le \|E_k\|_F + L\,\|s_k\|$, откуда после
суммирования и при достаточной близости к корню $\|E_k\| \le \eta\sigma_k$
для $k \ge k_0$. На начальных итерациях ($k < k_0$) условие
не гарантировано~--- именно поэтому в Алгоритме~\ref{alg:sp_broyden_ls}
предусмотрен fallback $\alpha_k > \alpha_{\min}$: если backtracking
не находит допустимого~$\alpha_k$, происходит сброс
$B_k \leftarrow \widetilde{B}$ (как в режиме $\mathtt{accumulate} =
\mathtt{false}$ Алгоритма~\ref{alg:sp_broyden}) и итерация повторяется.
\end{remark}

\paragraph{Глобальная сходимость.}
Следующая теорема~--- адаптация классического результата
Цоутендейка~\cite{zoutendijk1970, bfgs} к нашей схеме.

\begin{theorem}[Глобальная сходимость SP-Broyden + Armijo]
\label{thm:global-conv}
Пусть выполнены условия:
\begin{enumerate}
\item[(\textbf{A1})] $F\in C^{1}(\mathbb{R}^n)$, якобиан $J = \nabla F$
      непрерывен по Липшицу с константой~$L$ на сублевельном множестве
      $\mathcal{L}_0 := \{x : \psi(x) \le \psi(x_0)\}$, и $\mathcal{L}_0$
      ограничено;
\item[(\textbf{A2})] существует $\eta \in (0, 1/2)$ такой, что
      условие~\eqref{eq:bd-small} выполнено для всех $k$
      (или, эквивалентно, для всех $k \ge k_0$ после конечного числа
      перезагрузок $B_k \leftarrow \widetilde{B}$);
\item[(\textbf{A3})] $\sup_k \kappa(J_k) \le \kappa^{*} < \infty$
      на $\mathcal{L}_0$.
\end{enumerate}
Тогда последовательность $\{x_k\}$, порождённая
Алгоритмом~\ref{alg:sp_broyden_ls}, удовлетворяет
\begin{equation}\label{eq:zoutendijk}
  \sum_{k=0}^{\infty} \cos^{2}\!\theta_k \cdot \|g_k\|^{2}
  \;<\; \infty,
  \qquad
  \cos\theta_k := \frac{-\,g_k^{\top} d_k}{\|g_k\|\,\|d_k\|}
  \;\ge\; \eta_0 > 0,
\end{equation}
откуда $\|g_k\| \to 0$ при $k \to \infty$. В частности, любая предельная
точка $x^{*}$ последовательности $\{x_k\}$ удовлетворяет
$J(x^{*})^{\top} F(x^{*}) = 0$. Если, кроме того, $J(x^{*})$ невырожден,
то $F(x^{*}) = 0$.
\end{theorem}

\begin{proof}
Стандартное доказательство Цоутендейка
(см.~\cite[теорема~3.2]{bfgs}) опирается на три ингредиента:
(i)~условие угла~\eqref{eq:descent}, которое в нашей постановке даётся
леммой~\ref{lem:descent} с $\eta_0 = (1-\eta)/((1+\eta)\kappa^{*})$;
(ii)~достижимость условия Армихо~\eqref{eq:armijo} за конечное число
backtracking-шагов, что гарантировано $L$-гладкостью~$\nabla\psi$
на $\mathcal{L}_0$ и ведёт к нижней оценке шага
$\alpha_k \ge \alpha_{*} := 2\rho(1-c_1)\,\eta_0\,\sigma_{*}^{2}/L_{\psi}$,
где $L_{\psi}$~--- липшицева константа $\nabla\psi$ на~$\mathcal{L}_0$
и $\sigma_{*} := \min_{x \in \mathcal{L}_0}\sigma_{\min}(J(x))$;
(iii)~монотонность $\psi(x_k)$ и её ограниченность снизу нулём,
дающие
$\sum_{k} \big(\psi(x_k) - \psi(x_{k+1})\big) = \psi(x_0) - \lim_k \psi(x_k) < \infty$.
По~\eqref{eq:armijo} каждая разность не меньше
$c_1 \alpha_{*}\,\cos\theta_k\,\|g_k\|\,\|d_k\| \ge
 c_1\alpha_{*}\eta_0\,\|g_k\|^{2}/\beta^{*}$ (где
$\beta^{*} = \max_{\mathcal{L}_0}\|J\|$),
что и даёт~\eqref{eq:zoutendijk} вместе с $\|g_k\| \to 0$.
Предельный переход и невырожденность $J(x^{*})$
дают $F(x^{*}) = 0$.
\qedhere
\end{proof}

\begin{remark}[Сохранение локальной сверхлинейности]\label{rem:unit-step}
В стандартной теории глобализации квазиньютоновских методов
(\cite[\S11.2]{bfgs}; \cite[\S5.2]{dennis1996}) показывается, что
при $\|E_k\| \to 0$ и условии Деннис--Море единичный шаг $\alpha_k = 1$
в~\eqref{eq:armijo} принимается \emph{асимптотически всегда}.
Следовательно, Алгоритм~\ref{alg:sp_broyden_ls} наследует от
Алгоритма~\ref{alg:sp_broyden} локальную сверхлинейную
скорость теоремы~\ref{thm:bd}, не теряя при этом глобальной
сходимости из теоремы~\ref{thm:global-conv}: <<плата>> за
глобализацию~--- лишь конечное число backtracking-шагов на
начальной (далёкой от $x^{*}$) фазе.
\end{remark}

\begin{remark}[Связь с регуляризацией шагом в схеме VIJI]
\label{rem:viji-reg}
В главе~\ref{ch:sp-broyden-viji} (см.\ формулу схемы итерации
в~\S\ref{sec:viji:numerics}) для вариационных неравенств используется
альтернативная форма глобализации~--- адаптивное демпфирование шага
$d_k = -(B_k + \beta_k I)^{-1} F(x_k)$ с $\beta_k =
\max(\beta_{\min},\, L_1\|d_{k-1}\|)$. Эта конструкция~---
дискретный аналог метода Левенберга--Марквардта~\cite{marquardt1963},
естественный для VI с сильной монотонностью: при больших $\beta_k$
шаг приближается к проекционному (с фиксированной длиной),
что обеспечивает невозрастание $\psi$ глобально, а при $\beta_k \to 0$
восстанавливается чистый SP-Broyden шаг и его сверхлинейная асимптотика.
Развёрнутый анализ этой формы глобализации в общем VI-случае
выходит за рамки настоящей работы; здесь мы ограничиваемся замечанием,
что Алгоритм~\ref{alg:sp_broyden_ls} (line search) и схема
с $\beta_k$-демпфированием из~\S\ref{sec:viji:numerics}~--- два варианта
одной идеи (траст-регион
неявно через $\beta_k$ vs.\ явный поиск шага вдоль направления),
обсуждаемой в литературе по квазиньютоновским методам с
гарантиями глобальной сходимости~\cite{dennis1996, bfgs}.
\end{remark}

\section{Численные эксперименты}\label{sec:spb_numerics}
% ============================================================

Для экспериментальной проверки доказанных выше свойств сравним три метода
ранг-1 обновлений якобиана при решении $F(x) = 0$ с $B_0 = I$:
классический Бройден ($v_k = s_k$), Андерсон (пересборка $B$ из~$I$ по окну
из~$m$ секущих) и SP-Broyden ($v_k = S_p(S_p^\top S_p)^{-1} e_1$,
сохранение $p$ прошлых секущих условий).

\paragraph{Тестовые задачи.}
Используются две стандартные нелинейные системы размерности~$n = 100$
из набора Море--Гарбов--Хилстрём~\cite{more1981}:
\begin{itemize}
\item \textbf{Discrete BVP}: дискретизация краевой задачи
      $u''(t) + (u + t + 1)^3 / 2 = 0$, $u(0) = u(1) = 0$
      с начальным приближением $x_0^{(i)} = 0{,}1 \cdot t_i(t_i - 1)$,
      $t_i = i/(n+1)$.
\item \textbf{Broyden Banded}: симметричная банда ширины~$7$ с кубической
      нелинейностью по диагонали,
      $x_0 = -0{,}1 \cdot \mathbf{1}$.
\end{itemize}
Параметры обновлений: $m = 10$ (Андерсон), $p \le 10$ с адаптивным
выбором по условности грамиана $\mathrm{cond}(S_p^\top S_p) < 10^3$
(SP-Broyden); для классического Бройдена $m = 0$.

\begin{figure}[htbp]
\centering
\includegraphics[width=\textwidth]{fig_sp_broyden_conv.pdf}
\caption{Сходимость $\|F(x_k)\|$ классического Бройдена, Андерсона
  и SP-Broyden на двух тестовых задачах. Слева~--- по итерациям,
  справа~--- по числу вызовов~$F$. Пунктир~--- уровень $10^{-10}$.}
\label{fig:spb_conv}
\end{figure}

\paragraph{Результаты.}
На Discrete BVP (Рис.~\ref{fig:spb_conv}, первая строка) SP-Broyden
достигает точности $\|F\| < 10^{-15}$ за $\sim 180$ итераций против
$\sim 430$ у классического Бройдена и полной стагнации у Андерсона
(превышение лимита в $500$ итераций при $\|F\| \sim 10^{-8}$).
Ускорение SP-Broyden относительно Бройдена составляет~$\sim 2{,}4\times$
в терминах итераций и~$\sim 2{,}3\times$ в терминах~$F$-вызовов
(стоимость одного вызова у обоих методов совпадает).

На Broyden Banded (вторая строка) разрыв качественно иной:
классический Бройден и Андерсон расходятся (уходят в~$+\infty$),
а SP-Broyden стабильно сходится до $\|F\| \sim 5 \cdot 10^{-13}$
за~$\sim 60$ итераций. Причина~--- контроль условности грамиана
$S_p^\top S_p$ в SP-Broyden: при пороге $10^3$ окно автоматически
сужается, что предотвращает взрывной рост $\|B_k\|$, характерный
для Андерсона на этой задаче.

\paragraph{Связь с теорией.}
Наблюдаемые результаты соответствуют доказанному в
теореме~\ref{thm:bd} точному разложению bounded deterioration
$\|E_{k+1}\|_F^2 = \|E_k\|_F^2 - \|E_k\Pi_k\|_F^2 + \Delta_k$
с проектором ранга~$p{+}1$ и строгому неравенству
$\|E_k\Pi_k\|_F^2 \ge \|E_k s_k\|^2/\|s_k\|^2$
(\eqref{eq:neg-term-dominance}): на каждой итерации SP-Broyden
снимает с матрицы ошибки не меньше информации, чем классический
Бройден, а в типичной ситуации~--- строго больше за счёт
$p$ дополнительных направлений в проекторе. Это превращается
в заметное ускорение сходимости в практическом диапазоне
точности $10^{-10}$--$10^{-15}$. Разрыв особенно выражен на задачах,
где классический Бройден \emph{расходится}~--- SP-Broyden остаётся
устойчивым благодаря инвариантности индуктивного секущего условия
$B_k s_j = y_j$ для $j$ в окне памяти (теорема~\ref{thm:minimality}).

\paragraph{Эволюция аппроксимации якобиана.}
Прямая численная проверка bounded deterioration~--- эволюция нормы
ошибки $\|B_k - J(x_k)\|_F$ по итерациям (Рис.~\ref{fig:spb_jacerr}).
На Discrete BVP, $n=100$, в финальной точке оба метода выходят на сравнимую
норму ошибки ($\|B-J\|_F \approx 5{,}7$ для SP-Broyden и $\approx 7{,}4$ для
классического Бройдена), но за разное число итераций: $170$ против $384$.
Правая панель показывает скалярную величину $\|E_k \Pi_k\|_F$~--- объём
ошибки, снимаемой проектором $\Pi_k$ на текущей итерации. У классического
Бройдена $\Pi_k = s_k s_k^\top/(s_k^\top s_k)$ имеет ранг 1, и $\|E_k\Pi_k\|_F$
в точности равно $\|E_k s_k\|/\|s_k\|$~--- одна координата ошибки. У SP-Broyden
с $p\le 10$ проектор $\Pi_k = \widehat S_p(\widehat S_p^\top \widehat S_p)^{-1}
\widehat S_p^\top$ имеет ранг $p+1\le 11$ и снимает в среднем на 1--1{,}5
порядка больше ошибки на каждом шаге, что согласуется с
неравенством~\eqref{eq:neg-term-dominance} и точным
разложением~\eqref{eq:exact_decomp}.

Чтобы проверить роль параметра~$p$ как такового (а не адаптивного
порога), на Рис.~\ref{fig:spb_pvar} приведены кривые сходимости

устойчивую сходимость за $62\text{--}69$ итераций. Качественная
неустойчивость классического Бройдена ($p=0$) на этой задаче, отмеченная
выше, оказывается следствием именно ранг-1 структуры
обновления, а не плохого начального приближения.

обратно пропорционален квадрату наименьшего сингулярного числа окна,
и порог $\tau=10^3$ ограничивает $1/\sigma_{\min}^2$ величиной
$\tau/\sigma_{\max}^2(\widehat S_p^{(k)})$.

\section{Высокая размерность: Шерман--Моррисон и L-SP-Broyden}\label{sec:highdim}

\paragraph{Мотивация.} Прямая реализация Алгоритма~\ref{alg:sp_broyden}
хранит плотную матрицу $B_k\in\mathbb R^{n\times n}$, что требует $O(n^2)$
памяти и $O(n^3)$ операций на решение системы $B_k d = -F(x_k)$ через
LU-факторизацию. При $n=10^4$ это $\approx 8\cdot 10^8$ чисел двойной
точности ($\approx 800$~МБ) и $\approx 10^{12}$ flops на итерацию, что
делает прямую форму неприменимой к крупномасштабным задачам.

\paragraph{Шерман--Моррисон-форма.} Поскольку SP-обновление имеет ранг
1, $B_{k+1} = B_k + u_k v_k^{\top}$ с
$u_k = (y_k - B_k s_k)/(v_k^{\top} s_k)$, формула
Шермана--Моррисона~\cite[\S 3.4]{dennis1996} даёт явное обновление
обратной матрицы $H_k := B_k^{-1}$:
\begin{equation}\label{eq:sm_update}
   H_{k+1} \;=\; H_k \;+\; \frac{(s_k - H_k y_k)\,v_k^{\top} H_k}
                                 {v_k^{\top} H_k y_k},
   \qquad
   d_k \;=\; -H_k F(x_k).
\end{equation}
Хранение $H_k$ остаётся $O(n^2)$, но шаг $d_k$ вычисляется как
матрично-векторное произведение за $O(n^2)$ flops вместо $O(n^3)$.
Знаменатель $v_k^{\top} H_k y_k$ в~\eqref{eq:sm_update} получается из
тождества $1 + v_k^{\top} H_k u_k = v_k^{\top} H_k y_k / (v_k^{\top} s_k)$
после подстановки $u_k$. Условие невырожденности $\det B_{k+1} \ne 0$
эквивалентно $v_k^{\top} H_k y_k \ne 0$; при вырождении срабатывает
fallback~\ref{rem:fallback} (Левенберг--Марквардт-демпфирование).

\paragraph{Limited-memory L-SP-Broyden.} Для $n\gtrsim 10^4$ хранить
плотную $H_k$ невозможно даже в форме~\eqref{eq:sm_update}. По аналогии
с limited-memory BFGS~\cite{lbfgs,liu1989} предложим limited-memory
вариант SP-Broyden, в котором базовая матрица $H_0=I$ \emph{никогда не
формируется явно}. Вместо этого хранится скользящее окно последних
$m$ принятых троек $\{(s_j, y_j, v_j)\}_{j=k-m+1}^{k}$.

Применение $H_k$ к произвольному вектору $g$ реализуется forward-pass
рекурсией (Алгоритм~\ref{alg:lsp_broyden_apply}), формально
эквивалентной последовательной композиции
\eqref{eq:sm_update} к $H_0=I$ для всех корректировок текущего окна:
\begin{equation}\label{eq:lsp_recursion}
  H_k g \;=\; H_{k-m} g \;+\; \sum_{j=k-m+1}^{k}
     \frac{(s_j - H_j y_j)\,(v_j^{\top} H_j g)}
          {v_j^{\top} H_j y_j},\qquad H_{k-m}=I.
\end{equation}

\begin{algorithm}[H]
\caption{Apply $H_k\,g$ для L-SP-Broyden (forward-pass рекурсия)}
\label{alg:lsp_broyden_apply}
\begin{algorithmic}[1]
   \Require окно $\{(s_j, y_j, v_j)\}_{j=1}^{m}$, входной вектор $g$
   \Ensure $z = H_k g$
   \State $z \leftarrow g$ \Comment{$H_0 = I$}
   \For{$j = 1, 2, \ldots, m$}
      \State $w \leftarrow y_j$ \Comment{вычислим $H_j y_j$}
      \For{$i = 1, \ldots, j-1$}
         \State $\beta_i \leftarrow (v_i^{\top} w)/\alpha_i^{\mathrm{cache}}$
         \State $w \leftarrow w + \xi_i^{\mathrm{cache}} \beta_i$
      \EndFor
      \State $\alpha_j^{\mathrm{cache}} \leftarrow v_j^{\top} w$
      \State $\xi_j^{\mathrm{cache}} \leftarrow s_j - w$
      \State $\beta_j \leftarrow (v_j^{\top} z)/\alpha_j^{\mathrm{cache}}$
      \State $z \leftarrow z + \xi_j^{\mathrm{cache}} \beta_j$
   \EndFor
   \State \Return $z$
\end{algorithmic}
\end{algorithm}

Полный шаг L-SP-Broyden выглядит так: на итерации $k$ вычисляется
$d_k = -\Call{ApplyH}{F(x_k); \mathrm{window}_k}$; делается
backtracking-Армихо (Алгоритм~\ref{alg:sp_broyden_ls}) или полный шаг
$x_{k+1}=x_k+d_k$; формируется секущая
$(s_k, y_k) = (x_{k+1}-x_k,\, F(x_{k+1})-F(x_k))$;
строится SP-направление $v_k$ из текущего окна столбцов $S_p$ (как в
Алгоритме~\ref{alg:sp_broyden}, шаги~5--13);
тройка $(s_k,y_k,v_k)$ добавляется в очередь, а самая старая тройка
выталкивается, если длина $> m$.

\paragraph{Сложность L-SP-Broyden.} Память: каждая тройка занимает
$3n$~floats, плюс служебные кэши $\xi^{\mathrm{cache}},
\alpha^{\mathrm{cache}}$ размера $O(nm)$, итого
\begin{equation}\label{eq:lsp_memory}
   \mathcal M_{\text{L-SP-Broyden}} \;=\; 3nm + nm + 2n \;=\; O(nm).
\end{equation}
При $n=10^4, m=10$ это $4{,}2\cdot 10^5$ floats $\approx 3{,}4$~МБ.
Время: тело внешнего цикла Алгоритма~\ref{alg:lsp_broyden_apply}
выполняется $m$ раз, внутреннего --- до $m-1$ раз; стоимость одного
обращения $H_k g$ составляет $O(nm^2)$ flops. На итерацию приходится
2 обращения (одно для $d_k$, одно для $H_k y_k$ при расчёте знаменателя
$\alpha_k$), итого $O(nm^2)$~flops/iter. При $n=10^4, m=10$ это
$\approx 2\cdot 10^6$~flops/iter --- на пять порядков меньше, чем
$O(n^2)$ у плотной формы~\eqref{eq:sm_update}.

\begin{remark}[Полный пересчёт vs.\ кэширование]\label{rem:lsp_recursion}
Хотя величины $\xi_j^{\mathrm{cache}}, \alpha_j^{\mathrm{cache}}$ зависят
только от $\{(s_i,y_i,v_i)\}_{i\le j}$ и могут показаться допускающими
кэширование между итерациями, при выталкивании самой старой пары из
окна оставшиеся $\xi_i$ становятся \emph{несогласованными} с усечённой
историей: $H_i y_i$ они вычислялись относительно полного префикса
$0,\ldots,i-1$, не обрезанного префикса. Поэтому на каждом обращении
$H_k g$ значения $\xi_i^{\mathrm{cache}}, \alpha_i^{\mathrm{cache}}$
\emph{пересчитываются заново} из текущего окна
(внутренний цикл Алгоритма~\ref{alg:lsp_broyden_apply}). То же
требование удерживает и L-BFGS two-loop~\cite[\S 7.2]{bfgs}, где
коэффициенты $\rho_i$ хранятся (это скаляры), но кэширование
произведений $H_i y_i$ исключено по той же причине.
\end{remark}

\begin{remark}[Связь с L-BFGS и Anderson]\label{rem:lsp_vs_lbfgs}
Limited-memory BFGS~\cite{lbfgs,liu1989,byrd1994} применим для
\emph{минимизации}: требует градиента $\nabla f$ и предполагает
симметричный положительно определённый гессиан. Для систем $F(x)=0$
без симметрии Якобиана~--- к которым относится большинство задач
вариационных неравенств (гл.~4)~--- L-BFGS не применим напрямую без
сведения к минимизации merit-функции $\psi(x) = \tfrac12\Norm{F(x)}^2$
с расчётом $\nabla\psi = J(x)^{\top} F(x)$, требующим Якобиан-векторных
произведений. L-SP-Broyden, как и классический Бройден, работает
непосредственно с $F$ и не требует симметрии. Anderson($m,\beta$)
\cite{anderson1965,walker2011,toth2015}~--- альтернативный
limited-memory метод без явной модели Якобиана; в нём окно
$\{x_j, F(x_j)\}$ используется для построения мульти-секущего fit'а,
но без управления условием~(14) из гл.~4 (\eqref{eq:cd-quant}),
что делает Anderson менее точным в режимах, где требуется
субквадратичная сверхлинейность.
\end{remark}

\paragraph{Численные эксперименты.} Тестовый набор:
\textbf{(A)} Banded Cubic~--- трёхдиагональная сильно-монотонная
система с существенным кубическим возмущением:
\begin{equation}\label{eq:banded-cubic}
  F_i(x) = \bigl(1 + \tfrac12\sin^2(\pi i/n)\bigr) x_i
              - 0{,}3\,(x_{i-1} + x_{i+1})
              + 1{,}5\,(x_i - c_i)^3,\quad
  c_i = \tfrac12 \sin(2\pi i/n),
\end{equation}
с граничными условиями $x_0 = x_{n+1} = 0$. Линейная часть
диагонально доминантна, $\mathrm{cond} \approx 5$ независимо от $n$;
кубическое возмущение значимо при $\Norm{x}=O(1)$.
\textbf{(B)} Broyden Banded (MGH~\#31, \cite{more1981})~--- стандартный
тест, лента ширины~7, со стартовой точкой $x^{(0)}_i = -0{,}1$.

Размерности $n\in\{10^3, 10^4\}$. Методы:
\begin{itemize}
   \item Broyden-SM и SP-Broyden-SM (плотная $H$, $p\le 5$)~---
     только для $n=10^3$ (при $n=10^4$ потребовалось бы $\approx 800$~МБ);
   \item L-Broyden ($m=10$, $p=0$), L-SP-Broyden ($m\in\{10,20\},\, p\le 5$),
     Anderson ($m=10, \beta=1$)~--- для обеих размерностей.
\end{itemize}
Все Бройден-семейные методы запускаются с глобализацией Армихо
(Алгоритм~\ref{alg:sp_broyden_ls}, $c_1 = 10^{-4}$). Anderson
реализован в схеме Walker--Ni~\cite{walker2011} поверх Picard-отображения
$g_\tau(x) = x - \tau F(x)$ с residual-safeguard. Критерий остановки
$\Norm{F(x_k)}_2 \le 10^{-10}$. Скрипт~--- \texttt{diag\_highdim.py} в
корне репозитория, сырые траектории~--- \texttt{highdim\_results.npz}.

\paragraph{Анализ результатов.} На \textbf{Banded Cubic} все методы
сходятся за $\approx 30$ итераций при обоих $n$; L-SP-Broyden $(m=10)$
точно воспроизводит результат dense SP-Broyden-SM $(p\le 5)$
($32$~ит.~vs.~$31$~ит.~при $n=10^3$, $32$~ит.~vs.~$32$~ит.~при $n=10^4$).
На \textbf{Broyden Banded} картина выраженнее: dense SP-Broyden-SM
сходится за~32~ит.~при $n=10^3$, L-SP-Broyden $(m=10)$~--- за 41~ит.
($\approx 28\,\%$ хуже), L-SP-Broyden $(m=20)$ воспроизводит
точно 32~ит.~dense-варианта. L-Broyden $(m=10, p=0)$ требует
60~ит.~--- систематически хуже SP-семейства за счёт отсутствия
ранг-$(p+1)$ проектора. При $n=10^4$ dense-вариант невозможен,
а L-SP-Broyden $(m=20)$ достигает тех же 32~итераций, занимая
$8{,}2\cdot 10^5$ floats $\approx 6{,}6$~МБ \emph{вместо}
$10^8$ floats $= 800$~МБ у плотной формы~--- сокращение памяти
в $122\times$ при идентичной сходимости.

\paragraph{Влияние глубины окна.} На задаче~(A) при $n=10^3$:
$m=2 \to 41$~ит., $m=5 \to 33$~ит., $m=10 \to 32$~ит., $m=20 \to 30$~ит.
Узкое окно ($m=2$) ограничивает SP-проектор рангом $\le 2$ и снимает
основное преимущество перед L-Broyden. Для $m\ge 5$ информация,
накапливаемая в окне, достаточна для воспроизводства
dense-сходимости (рис.~\ref{fig:highdim_pvar}). Эмпирическая
рекомендация: $m \in [10, 20]$ при $p_{\max} = 5$ обеспечивает
оптимальный баланс между памятью и итерациями для рассмотренных
задач; при $n\ge 10^5$ может потребоваться $m\sim 50$, что
эмпирически соответствует прецеденту L-BFGS~\cite[\S 7.2]{bfgs}.

\paragraph{Итог раздела.} L-SP-Broyden с памятью $O(nm)$ и стоимостью
$O(nm^2)$ flops на итерацию воспроизводит сходимость плотного
SP-Broyden-SM при $m\ge p+1$ накопленных секущих условий. На
крупномасштабных тестах $n=10^4$ при $m=20$ достигается тот же
результат, что и у плотной формы при $n=10^3$, при сокращении памяти
в $\sim 100\times$. Перед Anderson при сравнимом $m$ выигрыш
выражается в более жёстком контроле условия~\eqref{eq:cd-quant}
(см.\ гл.~4), а перед L-Broyden~--- в наследовании ранг-$(p+1)$
проектора (теорема~\ref{thm:bd}).
