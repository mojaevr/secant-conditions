\noindent\textbf{Обозначения и предположения главы.}\enskip
Рассматривается задача поиска корня $F(x) = 0$ с
непрерывно-дифференцируемым отображением $F\colon\RR^n \to \RR^n$
и якобианом $J(x) = F'(x)$. Обозначим $x^*$~--- корень системы
($F(x^*) = 0$), $J^* = J(x^*)$, $s_k = x_{k+1} - x_k$,
$y_k = F(x_{k+1}) - F(x_k)$, $E_k = B_k - J^*$~--- ошибку
аппроксимации якобиана. Во всех теоретических утверждениях
главы~2 предполагается, что якобиан липшицев на некоторой открытой
окрестности $\mathcal{O} \subset \RR^n$ точки $x^*$:
\begin{equation}\label{eq:lipschitz}
  \Norm{J(x) - J(y)} \le L\,\Norm{x - y},
  \qquad x, y \in \mathcal{O}.
\end{equation}
Это стандартное предположение для анализа локальной сходимости
квазиньютоновских методов; см.~\cite{dennis1977,dennis1996}.

\medskip

Идея алгоритма SP-Broyden: выбрать $v_k$ так, чтобы ранг-1
обновление~\eqref{eq:rank1_normalized} удовлетворяло текущему секущему
условию $B_{k+1}s_k = y_k$ и одновременно \emph{сохраняло} $p$ предыдущих
секущих условий $B_{k+1}s_j = B_k s_j$ для $j = k{-}1, \ldots, k{-}p$.
По Утверждению~\ref{prop:destruction}, для этого достаточно потребовать
$v_k^\top s_j = 0$ при $j = k{-}1, \ldots, k{-}p$.

\begin{definition}[SP-обновление]\label{def:sp_update}
Пусть задана глубина сохранения $p \geq 0$ и пусть шаги
$s_k, s_{k-1}, \ldots, s_{k-p}$ линейно независимы.
Определим \emph{SP-вектор} $v_k^*$ как решение задачи
\begin{equation}\label{eq:sp_opt}
  \min_{v \in \mathbb{R}^n} \|v\|^2 \quad \text{при} \quad
  v^\top s_k = 1, \quad v^\top s_j = 0 \;\; (j = k{-}1, \ldots, k{-}p).
\end{equation}
SP-обновление якобиана имеет вид
\begin{equation}\label{eq:sp_update}
  B_{k+1} = B_k + (y_k - B_k s_k)\,(v_k^*)^\top.
\end{equation}
\end{definition}

\begin{remark}[Соглашение об обозначениях]\label{rem:notation}
Параметр $p$ обозначает число \emph{старых} секущих условий, которые
сохраняются. Матрица
$S_p = [s_k \mid s_{k-1} \mid \cdots \mid s_{k-p}] \in \mathbb{R}^{n\times(p+1)}$
содержит $p+1$ столбцов: текущий шаг $s_k$ стоит первым (это определяет
вектор $e_1$ в формулах ниже), за ним~--- $p$ предыдущих шагов.
В отличие от этого, в общей формуле~\eqref{eq:general} столбцы $S_m$
расположены в хронологическом порядке: от старых к новым.
\end{remark}

\begin{proposition}[Явная формула для $v_k^*$]\label{prop:vstar}
Пусть
$S_p = [s_k \mid s_{k-1} \mid \cdots \mid s_{k-p}] \in \mathbb{R}^{n \times (p+1)}$
и $G_p = S_p^\top S_p \in \mathbb{R}^{(p+1)\times(p+1)}$~--- грамиан.
Если $G_p$ невырожден, то
\begin{equation}\label{eq:vstar}
  v_k^* = S_p\,G_p^{-1}\,e_1,
\end{equation}
где $e_1 = (1, 0, \ldots, 0)^\top \in \mathbb{R}^{p+1}$.
При этом $\|v_k^*\|^2 = e_1^\top G_p^{-1}\,e_1$.
\end{proposition}

\begin{proof}
Задача~\eqref{eq:sp_opt} эквивалентна:
$\min_v \|v\|^2$ при $S_p^\top v = e_1$.
Целевая функция строго выпукла, ограничения аффинны, поэтому
стационарная точка функции Лагранжа является глобальным минимумом.

Функция Лагранжа имеет вид
$\mathcal{L}(v,\lambda) = \|v\|^2 - 2\lambda^\top(S_p^\top v - e_1)$.
Условие стационарности $\nabla_v\mathcal{L} = 2v - 2S_p\lambda = 0$
даёт $v = S_p\lambda$.
Подставляя в ограничение, получаем $G_p\lambda = e_1$,
откуда $\lambda = G_p^{-1}e_1$ и $v_k^* = S_p\,G_p^{-1}\,e_1$.

Норма оптимального вектора:
$\|v_k^*\|^2 = (v_k^*)^\top v_k^*
= e_1^\top G_p^{-1} S_p^\top S_p\,G_p^{-1} e_1
= e_1^\top G_p^{-1} e_1$.
\end{proof}

\begin{remark}\label{rem:p_zero}
При $p = 0$ ограничения ортогональности отсутствуют, $S_0 = [s_k]$,
$G_0 = \|s_k\|^2$, и формула~\eqref{eq:vstar} даёт
$v_k^* = s_k/\|s_k\|^2$.
Обновление~\eqref{eq:sp_update} принимает вид
\[
  B_{k+1} = B_k + \frac{(y_k - B_k s_k)\,s_k^\top}{\|s_k\|^2},
\]
что совпадает с классическим обновлением Бройдена~\eqref{eq:good_broyden}.
\end{remark}

\begin{remark}[Геометрическая интерпретация]\label{rem:geometry}
Вектор $v_k^*$ лежит в $\operatorname{col}(S_p)$ (по построению $v = S_p\lambda$)
и ортогонален подпространству $\operatorname{span}(s_{k-1}, \ldots, s_{k-p})$
(по ограничениям задачи). Таким образом, $v_k^*$ направлен вдоль компоненты
$s_k$, ортогональной прошлым шагам, и нормирован условием $v_k^{*\top} s_k = 1$.
Обновление~\eqref{eq:sp_update} модифицирует $B_k$ только в направлениях,
ортогональных прошлым шагам, что и обеспечивает сохранение секущих условий.
\end{remark}

\medskip

Полный алгоритм приведён в Алгоритме~\ref{alg:sp_broyden}.

\begin{algorithm}
\caption{SP-Broyden$(F,\,x_0,\,p_{\max},\,\mathtt{multisecant},\,
         \mathtt{accumulate},\,\widehat{B},\,\tau,\,\varepsilon)$}
\label{alg:sp_broyden}
\begin{algorithmic}
\Require $F\colon \mathbb{R}^n \to \mathbb{R}^n$,\ $x_0 \in \mathbb{R}^n$,\
         $p_{\max} \geq 0$,\
         $\mathtt{multisecant} \in \{\text{true},\text{false}\}$,\
         $\mathtt{accumulate} \in \{\text{true},\text{false}\}$,\
         $\widehat{B} \in \mathbb{R}^{n \times n}$
         \Comment{базовая матрица для пересборки, обычно $I_n$},\
         $\tau > 0$,\ $\varepsilon > 0$
\State $B_0 \leftarrow \widehat{B}$,\quad $k \leftarrow 0$
\While{$\|F(x_k)\| \geq \varepsilon$}
  \State Решить $B_k\,d_k = -F(x_k)$
  \State $x_{k+1} \leftarrow x_k + d_k$
  \State $s_k \leftarrow d_k$,\quad $y_k \leftarrow F(x_{k+1}) - F(x_k)$
  %------------------------------------------------------------
  \State \textbf{Адаптивный выбор глубины:}
  \State $p \leftarrow 0$
  \For{$p' = 1,\ldots,\min(p_{\max},\,k)$}
    \State $S_{p'} \leftarrow [s_k \mid s_{k-1} \mid \cdots \mid s_{k-p'}]$
      \hfill\Comment{$n\times(p'{+}1)$ матрица}
    \If{$\operatorname{cond}(S_{p'}^\top S_{p'}) < \tau$}
      \State $p \leftarrow p'$
    \Else
      \State \textbf{break}
    \EndIf
  \EndFor
  %------------------------------------------------------------
  \State \textbf{Обновление якобиана:}
  \If{$\mathtt{accumulate}$}
    \State $\widehat{B}_k \leftarrow B_k$
  \Else
    \State $\widehat{B}_k \leftarrow \widehat{B}$
  \EndIf
  \State $S_p \leftarrow [s_k \mid s_{k-1} \mid \cdots \mid s_{k-p}]$
  \State $G_p \leftarrow S_p^\top S_p$
  \If{$\mathtt{multisecant}$ \textbf{and} $p > 0$}
    \State \Comment{Ранг-$(p{+}1)$ обновление}
    \State $Y_p \leftarrow [y_k \mid y_{k-1} \mid \cdots \mid y_{k-p}]$
    \State $B_{k+1} \leftarrow \widehat{B}_k +
      (Y_p - \widehat{B}_k S_p)\,G_p^{-1}\,S_p^\top$
  \Else
    \State \Comment{Ранг-$1$ обновление (при $p=0$~--- Бройден)}
    \State $v_k \leftarrow S_p\,G_p^{-1}\,e_1$
    \State $B_{k+1} \leftarrow \widehat{B}_k +
      (y_k - \widehat{B}_k s_k)\,v_k^\top$
  \EndIf
  \State $k \leftarrow k + 1$
\EndWhile
\Return $x_k$
\end{algorithmic}
\end{algorithm}

\begin{remark}[Связь с таксономией обновлений]\label{rem:taxonomy}
Параметры $\mathtt{accumulate}$ и $\mathtt{multisecant}$ задают четыре
режима работы алгоритма, соответствующие ячейкам классификации
Fang--Saad (2009):
\begin{center}
\begin{tabular}{c|cc}
  & $\mathtt{multisecant} = \text{false}$ (ранг 1)
  & $\mathtt{multisecant} = \text{true}$ (ранг $p{+}1$) \\
\hline
$\mathtt{accumulate} = \text{true}$
  & SP-Broyden (при $p=0$~--- Бройден)
  & Multisecant Broyden \\
$\mathtt{accumulate} = \text{false}$
  & Ранг-1 обновление от~$\widehat{B}$
  & Anderson Acceleration \\
\end{tabular}
\end{center}
Левая нижняя ячейка ($\mathtt{accumulate} = \text{false}$,
$\mathtt{multisecant} = \text{false}$) формально корректна,
но на каждом шаге использует только текущую пару $(s_k, y_k)$
при фиксированной базе~$\widehat{B}$, отбрасывая всю накопленную
информацию. Этот режим не встречается в литературе и не даёт
преимуществ перед накоплением.
\end{remark}

\begin{remark}[Сложность]
Адаптивный выбор глубины и вычисление $v_k$ требуют $O(n\,p^2)$ операций
(формирование грамиана и решение системы размера $(p{+}1)\times(p{+}1)$).
При $p \ll n$ это пренебрежимо мало по сравнению с $O(n^2)$
на решение линейной системы $B_k\,d_k = -F(x_k)$.
\end{remark}

\begin{theorem}[Минимальность]\label{thm:minimality}
Пусть $S_p = [s_k \mid s_{k-1} \mid \cdots \mid s_{k-p}]$ и грамиан $G_p = S_p^\top S_p$
невырожден. Тогда обновление
\begin{equation}\label{eq:sp_update_thm}
  B_{k+1} = B_k + \frac{(y_k - B_k s_k)\,(v_k^*)^\top}{v_k^{*\top} s_k}
\end{equation}
с $v_k^* = S_p\,G_p^{-1}\,e_1$ является единственным решением задачи
\begin{equation}\label{eq:min_problem}
  \min_{B \in \mathbb{R}^{n \times n}} \|B - B_k\|_F \quad \text{при} \quad
  B s_k = y_k, \quad B s_j = B_k s_j \;\; (j = k{-}1, \ldots, k{-}p).
\end{equation}
Если, кроме того, $B_k s_j = y_j$ для $j = k{-}p, \ldots, k{-}1$, то $B_{k+1}$ удовлетворяет
всем $p+1$ секущим условиям: $B_{k+1} s_j = y_j$ для $j = k{-}p, \ldots, k$.
\end{theorem}
 
\begin{proof}
Обозначим $\Delta = B - B_k$. Задача~\eqref{eq:min_problem} принимает вид:
\begin{equation}\label{eq:delta_problem}
  \min_{\Delta \in \mathbb{R}^{n \times n}} \|\Delta\|_F \quad \text{при} \quad
  \Delta s_k = y_k - B_k s_k, \quad \Delta s_j = 0 \;\; (j = k{-}1, \ldots, k{-}p).
\end{equation}
 
\textbf{Шаг~1: сведение к ранг-1 задаче.}
Обозначим $r = y_k - B_k s_k \in \mathbb{R}^n$~--- невязку секущего условия. Запишем $\Delta$
в виде ранг-1 матрицы $\Delta = r\,w^\top$ для некоторого $w \in \mathbb{R}^n$. Тогда:
\[
  \Delta s_k = r\,(w^\top s_k), \qquad \Delta s_j = r\,(w^\top s_j).
\]
Ограничения~\eqref{eq:delta_problem} принимают вид:
\begin{equation}\label{eq:w_constraints}
  w^\top s_k = 1, \qquad w^\top s_j = 0 \;\; (j = k{-}1, \ldots, k{-}p).
\end{equation}
Норма: $\|\Delta\|_F = \|r\| \cdot \|w\|$. Поскольку $\|r\|$ фиксировано,
задача сводится к $\min_w \|w\|$ при ограничениях~\eqref{eq:w_constraints},
что совпадает с~\eqref{eq:sp_opt}.
 
\textbf{Шаг~2: оптимальность ранг-1 формы.}
Покажем, что решение задачи~\eqref{eq:delta_problem} действительно имеет ранг~1.
Пусть $\Delta^*$~--- произвольное допустимое решение произвольного ранга.
Разложим его строки: для каждой строки $\delta_i^\top$ (транспонированная $i$-я строка)
ограничения задают $\delta_i^\top s_k = r_i$ и $\delta_i^\top s_j = 0$ для $j = k{-}1, \ldots, k{-}p$.
Минимальный по евклидовой норме вектор $\delta_i$, удовлетворяющий этим условиям,
находится из KKT и имеет вид:
\[
  \delta_i^* = r_i \cdot S_p(S_p^\top S_p)^{-1} e_1 = r_i \cdot v_k^*.
\]
Тогда $\Delta^* = r\,v_k^{*\top}$, что есть ранг-1 матрица.
 
\textbf{Шаг~3: подстановка.}
Решение: $w^* = v_k^* = S_p\,G_p^{-1}\,e_1$ (по утверждению~\ref{prop:vstar}).
Обновление: $B_{k+1} = B_k + r\,v_k^{*\top} = B_k + (y_k - B_k s_k)\,v_k^{*\top} / (v_k^{*\top} s_k)$
(с учетом $v_k^{*\top} s_k = e_1^\top G_p^{-1} S_p^\top s_k = e_1^\top G_p^{-1} G_p e_1 = 1$,
но в общем случае $v_k^{*\top} s_k = e_1^\top G_p^{-1} (S_p^\top s_k)$, что равно
$(G_p^{-1})_{11}^{} \cdot \|s_k\|^2 + \sum_{l=1}^{p} (G_p^{-1})_{1,l+1} \cdot s_{k-l}^\top s_k$).
 
Заметим, что $S_p^\top s_k$~--- это первый столбец $G_p$, т.\,е. $S_p^\top s_k = G_p e_1$.
Следовательно, $v_k^{*\top} s_k = e_1^\top G_p^{-1} G_p e_1 = 1$.
 
Таким образом, $B_{k+1} = B_k + (y_k - B_k s_k)\,v_k^{*\top}$.
 
\textbf{Шаг~4: единственность.}
Единственность следует из строгой выпуклости нормы Фробениуса
на аффинном подпространстве, задаваемом линейными ограничениями~\eqref{eq:delta_problem}.
 
\textbf{Шаг~5: сохранение секущих условий.}
По утверждению~\ref{prop:destruction}, $v_k^{*\top} s_j = 0$ для $j = k{-}1, \ldots, k{-}p$
влечет $B_{k+1} s_j = B_k s_j$. Если $B_k s_j = y_j$, то $B_{k+1} s_j = y_j$.
Вместе с $B_{k+1} s_k = y_k$ это дает $B_{k+1} s_j = y_j$ для всех $j = k{-}p, \ldots, k$.
\end{proof}

%% --- Лемма: эквивалентность обновлений ---

\begin{lemma}[Эквивалентность SP-Broyden и мультисекущего обновления]\label{lem:equivalence}
Пусть $S_p = [s_k \mid s_{k-1} \mid \cdots \mid s_{k-p}] \in \mathbb{R}^{n \times (p+1)}$
имеет полный столбцовый ранг,
$Y_p = [y_k \mid y_{k-1} \mid \cdots \mid y_{k-p}]$,
$G_p = S_p^\top S_p \in \mathbb{R}^{(p+1)\times(p+1)}$.

Пусть $B_k$ удовлетворяет индуктивному инварианту:
\begin{equation}\label{eq:invariant}
  B_k\,s_j = y_j \quad \text{для } j = k{-}1, \ldots, k{-}p.
\end{equation}

Тогда SP-обновление $B_{k+1}^{\mathrm{SP}} = B_k + (y_k - B_k s_k)\,v_k^{*\top}$
с $v_k^* = S_p\,G_p^{-1}\,e_1$ совпадает с мультисекущим обновлением:
\begin{equation}\label{eq:ms_update}
  B_{k+1}^{\mathrm{MS}} = B_k + (Y_p - B_k S_p)\,G_p^{-1}\,S_p^\top.
\end{equation}
\end{lemma}

\begin{proof}
По инварианту~\eqref{eq:invariant}: $B_k s_j = y_j$ для $j = k{-}1, \ldots, k{-}p$.
Следовательно, в матрице $Y_p - B_k S_p \in \mathbb{R}^{n \times (p+1)}$
столбцы со~2-го по~$(p{+}1)$-й равны нулю:
\[
  Y_p - B_k S_p = (y_k - B_k s_k)\,e_1^\top,
\]
где $e_1 \in \mathbb{R}^{p+1}$~--- первый стандартный базисный вектор.
Подставляя в~\eqref{eq:ms_update}:
\[
  B_{k+1}^{\mathrm{MS}}
  = B_k + (y_k - B_k s_k)\,e_1^\top G_p^{-1}\,S_p^\top
  = B_k + (y_k - B_k s_k)\,v_k^{*\top}
  = B_{k+1}^{\mathrm{SP}}.
\]
Корректность построения $v_k^*$:
$v_k^{*\top} s_j = e_1^\top G_p^{-1}\,S_p^\top s_j$.
Поскольку $s_j$~--- столбец $S_p$ с номером $(1 + k - j)$,
имеем $S_p^\top s_j = G_p\,e_{1+k-j}$, откуда
$v_k^{*\top} s_j = e_1^\top e_{1+k-j} = \delta_{jk}$.
\end{proof}

%% --- Основная теорема ---

\begin{theorem}[Bounded deterioration для SP-Broyden]\label{thm:bd}
В условиях леммы~\ref{lem:equivalence} пусть дополнительно якобиан $J(x)$
удовлетворяет условию Липшица~\eqref{eq:lipschitz} с константой~$L$,
и обозначим $J^* = J(x^*)$, $E_k = B_k - J^*$,
\[
  \Pi_k = S_p\,G_p^{-1}\,S_p^\top \quad\text{(ортогональный проектор на
  $\mathrm{col}\,S_p$)},
  \qquad
  \Pi_k^\perp = I - \Pi_k.
\]

Тогда:

\begin{enumerate}
\item[\textup{(a)}] \textbf{Точное разложение.}
\begin{equation}\label{eq:exact_decomp}
  \|E_{k+1}\|_F^2
  = \|E_k\|_F^2 - \|E_k\,\Pi_k\|_F^2 + \Delta_k,
\end{equation}
где
\begin{equation}\label{eq:Delta_def}
  \Delta_k = \operatorname{tr}\!\bigl(G_p^{-1}\,(Y_p - J^*\!S_p)^\top
  (Y_p - J^*\!S_p)\bigr).
\end{equation}

\item[\textup{(b)}] \textbf{Оценка в квадрате нормы Фробениуса.}
\begin{equation}\label{eq:bd_squared}
  \|E_{k+1}\|_F^2 \leq \|E_k\|_F^2
  + \frac{9L^2}{4\,\sigma_{\min}^2(S_p)}
    \sum_{j=k-p}^{k} \|x_j - x^*\|^2\,\|s_j\|^2.
\end{equation}

\item[\textup{(c)}] \textbf{Оценка в норме Фробениуса.}
\begin{equation}\label{eq:bd_linear}
  \|E_{k+1}\|_F \leq \|E_k\|_F
  + \frac{3L}{2\,\sigma_{\min}(S_p)}
    \left(\sum_{j=k-p}^{k} \|x_j - x^*\|^2\,\|s_j\|^2\right)^{\!1/2}.
\end{equation}
\end{enumerate}
\end{theorem}

\begin{proof}

\textbf{Шаг~1: разложение $E_{k+1}$.}
По лемме~\ref{lem:equivalence}, $B_{k+1} = B_{k+1}^{\mathrm{MS}}$.
Подставим определение~\eqref{eq:ms_update}:
\begin{align}
  E_{k+1}
  &= B_k - J^* + (Y_p - B_k S_p)\,G_p^{-1}\,S_p^\top \notag\\
  &= E_k + \bigl[(Y_p - J^*\!S_p) - E_k\,S_p\bigr]\,G_p^{-1}\,S_p^\top \notag\\
  &= E_k\,(I - S_p\,G_p^{-1}\,S_p^\top)
     + (Y_p - J^*\!S_p)\,G_p^{-1}\,S_p^\top. \label{eq:Ek1_full}
\end{align}
То есть:
\begin{equation}\label{eq:Ek1_compact}
  E_{k+1} = E_k\,\Pi_k^\perp + R_k,
  \qquad R_k = (Y_p - J^*\!S_p)\,G_p^{-1}\,S_p^\top.
\end{equation}

\medskip
\textbf{Шаг~2: ортогональность строк.}
Покажем, что $\operatorname{tr}\!\bigl((E_k\,\Pi_k^\perp)^\top R_k\bigr) = 0$.

Для каждого $i = 1, \ldots, n$ рассмотрим $i$-ю строку каждого слагаемого,
записанную как столбец транспонированной матрицы.

Строка~$i$ матрицы $E_k\,\Pi_k^\perp$:
\[
  (E_k\,\Pi_k^\perp)^\top e^{(i)} = \Pi_k^\perp\,E_k^\top\,e^{(i)}
  \;\in\; (\mathrm{col}\,S_p)^\perp,
\]
поскольку $\mathrm{im}\,\Pi_k^\perp = (\mathrm{col}\,S_p)^\perp$.

Строка~$i$ матрицы $R_k$:
\[
  R_k^\top\,e^{(i)} = S_p\,G_p^{-1}\,(Y_p - J^*\!S_p)^\top\,e^{(i)}
  \;\in\; \mathrm{col}\,S_p,
\]
поскольку вектор является линейной комбинацией столбцов~$S_p$.

Скалярное произведение $i$-х строк:
\begin{align*}
  \bigl(\Pi_k^\perp\,E_k^\top\,e^{(i)}\bigr)^\top
  &\bigl(S_p\,G_p^{-1}\,(Y_p - J^*\!S_p)^\top\,e^{(i)}\bigr) \\
  &= (E_k^\top e^{(i)})^\top\,
     \underbrace{\Pi_k^\perp\,S_p}_{= 0}\,
     G_p^{-1}\,(Y_p - J^*\!S_p)^\top\,e^{(i)}
  = 0,
\end{align*}
где $\Pi_k^\perp\,S_p = (I - S_p G_p^{-1} S_p^\top)\,S_p
= S_p - S_p G_p^{-1} G_p = 0$.
Суммируя по $i$:
\begin{equation}\label{eq:cross_zero}
  \operatorname{tr}\!\bigl((E_k\,\Pi_k^\perp)^\top R_k\bigr)
  = \sum_{i=1}^{n}
    \bigl((E_k\,\Pi_k^\perp)^\top e^{(i)}\bigr)^\top
    \bigl(R_k^\top\,e^{(i)}\bigr) = 0.
\end{equation}

\medskip
\textbf{Шаг~3: теорема Пифагора.}
Из~\eqref{eq:Ek1_compact} и~\eqref{eq:cross_zero}:
\begin{equation}\label{eq:pythagoras1}
  \|E_{k+1}\|_F^2 = \|E_k\,\Pi_k^\perp\|_F^2 + \|R_k\|_F^2.
\end{equation}
Аналогично, разложение $E_k = E_k\,\Pi_k + E_k\,\Pi_k^\perp$
даёт ортогональные (построчно) слагаемые, поскольку строка~$i$
матрицы $E_k\,\Pi_k$ лежит в $\mathrm{col}\,S_p$,
а строка~$i$ матрицы $E_k\,\Pi_k^\perp$~--- в $(\mathrm{col}\,S_p)^\perp$.
Следовательно:
\begin{equation}\label{eq:pythagoras2}
  \|E_k\|_F^2 = \|E_k\,\Pi_k\|_F^2 + \|E_k\,\Pi_k^\perp\|_F^2.
\end{equation}
Подставляя $\|E_k\,\Pi_k^\perp\|_F^2 = \|E_k\|_F^2 - \|E_k\,\Pi_k\|_F^2$
в~\eqref{eq:pythagoras1}:
\begin{equation}\label{eq:exact}
  \|E_{k+1}\|_F^2 = \|E_k\|_F^2 - \|E_k\,\Pi_k\|_F^2 + \|R_k\|_F^2.
\end{equation}

\medskip
\textbf{Шаг~4: вычисление $\|R_k\|_F^2$.}
Обозначим $A_k = Y_p - J^*\!S_p$. Тогда $R_k = A_k\,G_p^{-1}\,S_p^\top$ и:
\begin{align}
  \|R_k\|_F^2
  &= \operatorname{tr}\!\bigl(S_p\,G_p^{-1}\,A_k^\top A_k\,G_p^{-1}\,S_p^\top\bigr)
  \notag\\
  &= \operatorname{tr}\!\bigl(G_p^{-1}\,S_p^\top S_p\,G_p^{-1}\,A_k^\top A_k\bigr)
  \qquad\text{(цикличность следа)} \notag\\
  &= \operatorname{tr}\!\bigl(G_p^{-1}\,G_p\,G_p^{-1}\,A_k^\top A_k\bigr)
  = \operatorname{tr}\!\bigl(G_p^{-1}\,A_k^\top A_k\bigr).
  \label{eq:R_trace}
\end{align}
Это доказывает~\eqref{eq:exact_decomp} с $\Delta_k = \|R_k\|_F^2$.

\medskip
\textbf{Шаг~5: оценка $\Delta_k$ сверху.}
Матрицы $G_p^{-1}$ и $A_k^\top A_k$ положительно полуопределены. По неравенству
$\operatorname{tr}(MN) \leq \lambda_{\max}(M)\,\operatorname{tr}(N)$
для положительно полуопределённых $M$, $N$:
\begin{equation}\label{eq:trace_bound}
  \Delta_k = \operatorname{tr}\!\bigl(G_p^{-1}\,A_k^\top A_k\bigr)
  \leq \lambda_{\max}(G_p^{-1})\,\|A_k\|_F^2
  = \frac{\|A_k\|_F^2}{\lambda_{\min}(G_p)}
  = \frac{\|A_k\|_F^2}{\sigma_{\min}^2(S_p)}.
\end{equation}

Оценим $\|A_k\|_F^2$. Столбец матрицы $A_k$, отвечающий индексу~$j$, равен
$a_j = y_j - J^*\!s_j$. По теореме о среднем значении:
\[
  y_j - J^*\!s_j = \left(\int_0^1 \bigl[J(x_j + t\,s_j) - J^*\bigr]\,dt\right) s_j.
\]
По условию Липшица~\eqref{eq:lipschitz} и неравенству треугольника:
\[
  \|J(x_j + t\,s_j) - J^*\| \leq L\,\|x_j + t\,s_j - x^*\|
  \leq L\,(\|x_j - x^*\| + t\,\|s_j\|).
\]
Интегрируя по $t \in [0,1]$:
\begin{equation}\label{eq:integral_bound}
  \left\|\int_0^1 \bigl[J(x_j + t\,s_j) - J^*\bigr]\,dt\right\|
  \leq L\,\|x_j - x^*\| + \frac{L}{2}\,\|s_j\|.
\end{equation}
Следовательно:
\begin{equation}\label{eq:col_bound}
  \|a_j\| = \|y_j - J^*\!s_j\|
  \leq \left(L\,\|x_j - x^*\| + \frac{L}{2}\,\|s_j\|\right)\|s_j\|.
\end{equation}

\emph{Вариант~A (без дополнительных предположений):}
\begin{equation}\label{eq:col_bound_A}
  \|a_j\|^2 \leq L^2\left(\|x_j - x^*\| + \tfrac{1}{2}\,\|s_j\|\right)^2 \|s_j\|^2.
\end{equation}

\emph{Вариант~B (при $\|s_j\| \leq \|x_j - x^*\|$):}
\begin{equation}\label{eq:col_bound_B}
  \|a_j\| \leq \frac{3L}{2}\,\|x_j - x^*\|\,\|s_j\|,
  \qquad
  \|a_j\|^2 \leq \frac{9L^2}{4}\,\|x_j - x^*\|^2\,\|s_j\|^2.
\end{equation}
Условие $\|s_j\| \leq \|x_j - x^*\|$ выполняется в окрестности~$x^*$
при невырожденности $J^*$ и ограниченности $\|B_j^{-1}\|$:
$\|s_j\| = \|B_j^{-1} F(x_j)\| \leq \|B_j^{-1}\|\,(\|J^*\|\,\|x_j - x^*\|
+ O(\|x_j - x^*\|^2))$.

Суммируя по~$j$ (вариант~B):
\begin{equation}\label{eq:A_bound}
  \|A_k\|_F^2 \leq \frac{9L^2}{4}
    \sum_{j=k-p}^{k} \|x_j - x^*\|^2\,\|s_j\|^2.
\end{equation}

Подставляя~\eqref{eq:A_bound} в~\eqref{eq:trace_bound}
и далее в~\eqref{eq:exact}:
\[
  \|E_{k+1}\|_F^2
  \leq \|E_k\|_F^2 - \|E_k\,\Pi_k\|_F^2
  + \frac{9L^2}{4\,\sigma_{\min}^2(S_p)}
    \sum_{j=k-p}^{k} \|x_j - x^*\|^2\,\|s_j\|^2.
\]
Отбрасывая $-\|E_k\,\Pi_k\|_F^2 \leq 0$, получаем~\eqref{eq:bd_squared}.

\medskip
\textbf{Шаг~6: извлечение корня.}
Из~\eqref{eq:bd_squared}:
\[
  \|E_{k+1}\|_F^2 \leq \|E_k\|_F^2 + \delta_k,
  \qquad
  \delta_k = \frac{9L^2}{4\,\sigma_{\min}^2(S_p)}
    \sum_{j=k-p}^{k} \|x_j - x^*\|^2\,\|s_j\|^2.
\]
Для $a \geq 0$, $b \geq 0$ неравенство
$\sqrt{a + b} \leq \sqrt{a} + \sqrt{b}$
справедливо, поскольку $(\sqrt{a} + \sqrt{b})^2 = a + b + 2\sqrt{ab} \geq a + b$.
Применяя с $a = \|E_k\|_F^2$, $b = \delta_k$:
\[
  \|E_{k+1}\|_F
  \leq \sqrt{\|E_k\|_F^2 + \delta_k}
  \leq \|E_k\|_F + \sqrt{\delta_k}
  = \|E_k\|_F
  + \frac{3L}{2\,\sigma_{\min}(S_p)}
    \left(\sum_{j=k-p}^{k} \|x_j - x^*\|^2\,\|s_j\|^2\right)^{\!1/2}.
\]
Это даёт~\eqref{eq:bd_linear}.
\end{proof}

\begin{remark}[Сравнение с классическим Бройденом]\label{rem:comparison}
Для обновления Бройдена первого типа ($v_k = s_k / \|s_k\|^2$)
аналогичное разложение даёт:
\begin{equation}\label{eq:bd_broyden_classic}
  \|E_{k+1}^{\mathrm{Br}}\|_F^2
  = \|E_k\|_F^2 - \frac{\|E_k\,s_k\|^2}{\|s_k\|^2}
    + \frac{\|y_k - J^*\!s_k\|^2}{\|s_k\|^2}.
\end{equation}
Сопоставим с точным разложением~\eqref{eq:exact_decomp}:
\begin{enumerate}
\item \textbf{Отрицательный член.}
  В~\eqref{eq:bd_broyden_classic}: $\|E_k\,s_k\|^2 / \|s_k\|^2$~---
  проекция ошибки на одно направление~$s_k$.
  В~\eqref{eq:exact_decomp}: $\|E_k\,\Pi_k\|_F^2$~--- проекция
  на $(p{+}1)$-мерное подпространство $\mathrm{col}\,S_p$.
  По определению $\Pi_k$:
  \[
    \|E_k\,\Pi_k\|_F^2
    \geq \left\|E_k\,\frac{s_k\,s_k^\top}{\|s_k\|^2}\right\|_F^2
    = \frac{\|E_k\,s_k\|^2}{\|s_k\|^2},
  \]
  так как $s_k s_k^\top / \|s_k\|^2$~--- ортогональный проектор ранга~1,
  являющийся ``частью'' проектора~$\Pi_k$ ранга~$(p{+}1)$.
  Таким образом, SP-Broyden извлекает \textbf{не меньше} информации
  из $E_k$ на каждом шаге.

\item \textbf{Положительный член.}
  В~\eqref{eq:bd_broyden_classic}: $\|y_k - J^* s_k\|^2 / \|s_k\|^2 = O(\|x_k - x^*\|^2)$.
  В~\eqref{eq:bd_squared}: сумма по $p{+}1$ шагам, делённая на $\sigma_{\min}^2(S_p)$.
  Формально положительный член SP-Broyden может быть больше.
  Однако это компенсируется бо\,льшим отрицательным членом
  и тем, что мультисекущая структура обновления не допускает
  накопления ошибок на направлениях прошлых шагов.
\end{enumerate}
\end{remark}


% ============================================================
\section{Численные эксперименты}\label{sec:spb_numerics}
% ============================================================

Для экспериментальной проверки доказанных выше свойств сравним три метода
ранг-1 обновлений якобиана при решении $F(x) = 0$ с $B_0 = I$:
классический Бройден ($v_k = s_k$), Андерсон (пересборка $B$ из~$I$ по окну
из~$m$ секущих) и SP-Broyden ($v_k = S_p(S_p^\top S_p)^{-1} e_1$,
сохранение $p$ прошлых секущих условий).

\paragraph{Тестовые задачи.}
Используются две стандартные нелинейные системы размерности~$n = 100$
из набора Море--Гарбов--Хилстрём~\cite{more1981}:
\begin{itemize}
\item \textbf{Discrete BVP}: дискретизация краевой задачи
      $u''(t) + (u + t + 1)^3 / 2 = 0$, $u(0) = u(1) = 0$
      с начальным приближением $x_0^{(i)} = 0{,}1 \cdot t_i(t_i - 1)$,
      $t_i = i/(n+1)$.
\item \textbf{Broyden Banded}: симметричная банда ширины~$7$ с кубической
      нелинейностью по диагонали,
      $x_0 = -0{,}1 \cdot \mathbf{1}$.
\end{itemize}
Параметры обновлений: $m = 10$ (Андерсон), $p \le 10$ с адаптивным
выбором по условности грамиана $\mathrm{cond}(S_p^\top S_p) < 10^3$
(SP-Broyden); для классического Бройдена $m = 0$.

\begin{figure}[htbp]
\centering
\includegraphics[width=\textwidth]{fig_sp_broyden_conv.pdf}
\caption{Сходимость $\|F(x_k)\|$ классического Бройдена, Андерсона
  и SP-Broyden на двух тестовых задачах. Слева~--- по итерациям,
  справа~--- по числу вызовов~$F$. Пунктир~--- уровень $10^{-10}$.}
\label{fig:spb_conv}
\end{figure}

\paragraph{Результаты.}
На Discrete BVP (Рис.~\ref{fig:spb_conv}, первая строка) SP-Broyden
достигает точности $\|F\| < 10^{-15}$ за $\sim 180$ итераций против
$\sim 430$ у классического Бройдена и полной стагнации у Андерсона
(превышение лимита в $500$ итераций при $\|F\| \sim 10^{-8}$).
Ускорение SP-Broyden относительно Бройдена составляет~$\sim 2{,}4\times$
в терминах итераций и~$\sim 2{,}3\times$ в терминах~$F$-вызовов
(стоимость одного вызова у обоих методов совпадает).

На Broyden Banded (вторая строка) разрыв качественно иной:
классический Бройден и Андерсон расходятся (уходят в~$+\infty$),
а SP-Broyden стабильно сходится до $\|F\| \sim 5 \cdot 10^{-13}$
за~$\sim 60$ итераций. Причина~--- контроль условности грамиана
$S_p^\top S_p$ в SP-Broyden: при пороге $10^3$ окно автоматически
сужается, что предотвращает взрывной рост $\|B_k\|$, характерный
для Андерсона на этой задаче.

\paragraph{Связь с теорией.}
Наблюдаемые результаты соответствуют доказанной оценке~\eqref{eq:bd_linear}
улучшенной ограниченной деградации SP-Broyden: прирост
$\|B_{k+1} - J^*\|_F^2 - \|B_k - J^*\|_F^2 = \mathcal O(\|x_k - x^*\|^2)$
за шаг (против $\mathcal O(\|x_k - x^*\|)$ у классического Бройдена)
превращается в заметное ускорение сходимости в практическом диапазоне
точности $10^{-10}$--$10^{-15}$. Разрыв особенно выражен на задачах,
где классический Бройден \emph{расходится}~--- SP-Broyden остаётся
устойчивым благодаря инвариантности индуктивного секущего условия
$B_k s_j = y_j$ для $j$ в окне памяти (теорема~\ref{thm:minimality}).
