\section*{Базовые объекты}
\addcontentsline{toc}{section}{Базовые объекты}

\begin{description}[leftmargin=5.5em,labelwidth=5em,labelindent=0pt,font=\normalfont]
  \item[$\RR^n$]   пространство вещественных векторов размерности $n$
  \item[$F$]       нелинейное отображение $F\colon \RR^n \to \RR^n$ в задаче $F(x)=0$
                   (главы~1,~2,~4) или градиент $F = \nabla f$ задачи оптимизации (глава~3)
  \item[$f$]       скалярная целевая функция $f\colon\RR^n\to\RR$ в оптимизации
  \item[$J(x)$]    якобиан $F$ в точке $x$; в задаче оптимизации~--- гессиан $\nabla^2 f(x)$
  \item[$J^*$]     якобиан в решении: $J^* = J(x^*) = \nabla F(x^*)$
                   (единое обозначение во всех главах; макрос \texttt{\textbackslash Jstar})
  \item[$J_k$]     локальный якобиан в центре сегмента VIJI-Restarted:
                   $J_k = \nabla F(v_k)$ (только глава~4 и
                   приложение~\ref{app:sp-afd-proof}); следует отличать от $J^*$
  \item[$x^*$]     решение: $F(x^*)=0$ или решение VI
  \item[$\|\cdot\|$]    евклидова норма вектора
  \item[$\Fro{\cdot}$]  норма Фробениуса матрицы
  \item[$\tr(M)$]  след матрицы
  \item[$\col(M)$] линейная оболочка столбцов матрицы $M$
  \item[$\rank(M)$] ранг матрицы
  \item[$\sigma_{\min}(M)$,\, $\sigma_{\max}(M)$]
                   минимальное и максимальное сингулярные числа матрицы
  \item[$\lambda_{\min}(M)$,\, $\lambda_{\max}(M)$]
                   минимальное и максимальное собственные значения
                   симметричной матрицы
  \item[$\pinv{M}$] псевдообратная матрица Мура--Пенроуза
  \item[$\diag(\cdot)$] диагональная матрица
\end{description}

\section*{Квазиньютоновские обновления}
\addcontentsline{toc}{section}{Квазиньютоновские обновления}

\begin{description}[leftmargin=5.5em,labelwidth=5em,labelindent=0pt,font=\normalfont]
  \item[$x_k$]      текущая итерация алгоритма
  \item[$s_k$]      шаг: $s_k = x_{k+1} - x_k$
  \item[$y_k$]      разность значений: $y_k = F(x_{k+1}) - F(x_k)$
  \item[$d_k$]      направление поиска: $B_k\,d_k = -F(x_k)$
  \item[$B_k$]      текущая аппроксимация якобиана
  \item[$\widehat{B}_k$] базовая матрица в формуле обновления
                    (накопление: $\widehat{B}_k = B_k$; пересборка:
                    $\widehat{B}_k = \widetilde{B}$, обычно $I$)
  \item[$\widetilde{B}$] опорная (входная) матрица алгоритма
                    SP-Broyden, параметр функции; точка сброса
                    при пересборке. Не имеет итерационного индекса
                    и не зависит от $k$
  \item[$\Delk$]    изменение матрицы: $\Delk = B_{k+1} - \widehat{B}_k$
  \item[$S_m$]      матрица прошлых шагов:
                    $S_m = [s_{k-m+1}\mid\cdots\mid s_k]$
  \item[$Y_m$]      матрица прошлых разностей:
                    $Y_m = [y_{k-m+1}\mid\cdots\mid y_k]$
  \item[$V_m$]      матрица направлений обновления
                    (параметр семейства (\ref{eq:general}))
  \item[$E_k$]      ошибка аппроксимации якобиана: $E_k = B_k - J^*$
\end{description}

\section*{SP-Broyden и SS-U}
\addcontentsline{toc}{section}{SP-Broyden и SS-U}

\begin{description}[leftmargin=5.5em,labelwidth=5em,labelindent=0pt,font=\normalfont]
  \item[$m$]           ранг мультисекущего обновления в общей формуле
                        главы~\ref{ch:jacobian} ($m$ секущих условий
                        $B_{k+1} S_m = Y_m$, $S_m \in \RR^{n\times m}$);
                        используется только в главе~\ref{ch:jacobian}
  \item[$p$]           глубина сохраняемых \emph{прошлых} секущих условий
                        в SP-Broyden и его вариантах (главы~\ref{ch:sp-broyden-theory},
                        \ref{ch:symmetric}, \ref{ch:sp-broyden-viji}); число
                        столбцов окна на одну больше, чем $p$ ---
                        $S_p \in \RR^{n\times (p+1)}$ за счёт включения
                        текущей секущей $s_k$
  \item[$S_p$]         окно: текущий и $p$ прошлых шагов,
                        $S_p = [s_k\mid s_{k-1}\mid\cdots\mid s_{k-p}]
                        \in \RR^{n\times(p+1)}$
  \item[$Y_p$]         соответствующая матрица разностей $Y_p$
  \item[$G_p$]         грамиан окна: $G_p = S_p^\top S_p$
  \item[$v_k^*$]       оптимальный SP-вектор:
                        $v_k^* = S_p\,G_p^{-1}\,e_1$
  \item[$e_1$]         первый стандартный базисный вектор
                        $(1,0,\ldots,0)^\top$
  \item[$\Pi_k$]       ортогональный проектор на $\col(S_p)$:
                        $\Pi_k = S_p\,G_p^{-1}\,S_p^\top$
  \item[$\Pi_k^{\perp}$] дополнительный проектор: $\Pi_k^{\perp} = I - \Pi_k$
  \item[$\tau$]        порог числа обусловленности грамиана
                        (адаптивный выбор глубины $p$)
  \item[$\Ck$]         класс задач $\Qclass$: $L$-липшицев якобиан,
                        $\kappa$-обусловленность, память глубины $m$
  \item[$u_k^*$]       ведущий собственный вектор коррекции SS-U
                        (в $s_k^{\perp}$)
  \item[$\sigma_k^*$]  коэффициент ранг-1 коррекции SS-U
  \item[$\mathcal{U}$] базовое симметричное квазиньютоновское обновление
                        (SR1, BFGS, PSB и~т.\,п.) в конструкции SS-U
\end{description}

\section*{Задача с вариационным неравенством (глава 4)}
\addcontentsline{toc}{section}{Задача с вариационным неравенством (глава 4)}

\begin{description}[leftmargin=5.5em,labelwidth=5em,labelindent=0pt,font=\normalfont]
  \item[$X$]            замкнутое выпуклое множество, на котором
                         ставится VI
  \item[$\mu$]           константа сильной монотонности отображения $F$
  \item[$L_1$]           константа Липшица якобиана $\nabla F$
  \item[$J_0$]           линейная часть оператора в численных тестах
                          класса cubic-monotone и bilinear-saddle
                          ($F = J_0 x + \varepsilon \cdot R(x)$); это \emph{не} якобиан в решении
  \item[$v_k,\, z^{(s)}$] опорные точки сегментов VIJI-Restarted
  \item[$D$]             диаметр допустимого множества $X$
  \item[$D_s$]           радиус сегмента VIJI: оценка расстояния до $x^*$
  \item[$\bar{\rho}$]    коэффициент геометрического сжатия внутри сегмента
  \item[$\rhok$]         гарантированный коэффициент сжатия на шаге $k$
  \item[$M_0$]           равномерная оценка $\Fro{B_k - J^*}$
                          в анализе SP-Broyden внутри сегмента
  \item[$C_d$]           константа связи $\|d_k\| \ge C_d\,\|v_k - x^*\|$
                          (явное предположение Proposition~4.4)
  \item[$\varepsilon^*$] радиус локального режима,
                          в котором гарантируется оптимальная скорость
  \item[$\delta_k$]      управляемая ошибка якобиана:
                          $\|(B_k - J_k)(x_k - v_k)\| \le \delta_k\,\|x_k - v_k\|$
  \item[$r^*$]           число конечно-разностных поправок
                          (refinement-шагов) в SP-AFD
  \item[$\mathrm{GAP}$]  зазор двойственности VI
                          (критерий сходимости)
  \item[$N$,\, $T$]      число итераций
\end{description}

\section*{Соглашения}
\addcontentsline{toc}{section}{Соглашения}

\begin{itemize}[leftmargin=*]
  \item Индекс $k$ нумерует итерации алгоритма;
        индексы внутри окна пробегают $j = k-p,\ldots,k$.
  \item $I$~--- единичная матрица подходящей размерности.
  \item Если контекст не указан явно, все нормы матриц---фробениусовы,
        все нормы векторов---евклидовы.
  \item Константы обозначаются прописными латинскими буквами
        ($L,\, L_1,\, C_d,\, D,\, D_s,\, M_0$); параметры алгоритма---
        строчными греческими ($\tau,\, \mu,\, \bar{\rho}$);
        индексы~$k$ указывают итерационную зависимость.
\end{itemize}
