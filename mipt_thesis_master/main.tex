% --- TL 2026 workaround: pre-load etoolbox before the class --------------
% Класс mipt-thesis-bs использует \begingroup\globaldefs=1 для своих
% \renewcommand'ов. В TL 2026 \renewcommand под globaldefs=1 «протекает» во
% внутренние LaTeX3-хелперы и глобально засоряет hash-таблицу: при последующей
% загрузке etoolbox (через hyperref) его \newrobustcmd{\undef} падает с
% «Command \undef already defined» (а вместе с ним \appto, \preto, \ifinlist
% и т.д.). Если etoolbox загрузить заранее, эти команды уже существуют до
% globaldefs-блока и оказываются «защищены»: повторная загрузка пакета
% игнорируется, а пред-определённый \newrobustcmd внутри globaldefs-блока
% не вызывается. \RequirePackage до \documentclass — стандартный способ
% подгрузить пакет в исходную (нулевую) группу.
\RequirePackage{etoolbox}

\documentclass[oneside]{mipt-thesis-bs}

% --- TL 2026 workaround для hyperref/pd1enc.def:38 ----------------------------
% Ядро/patch в TeX Live 2026 при \DeclareFontEncoding{PD1}{}{} делает примерно
%   \advance\csname c@v@\f@version\endcsname by 1
% где \f@version ∈ {normal,bold,...}. Счётчики \c@v@normal/\c@v@bold не были
% инициализированы к этому моменту и \csname ... \endcsname выдаёт \relax →
% «You can't use `relax' after advance». В latex3/latex2e (release-2025-11-01
% и develop) такого кода в \DeclareFontEncoding@ нет — значит, его ставит
% кто-то из пакетов TL 2026 (возможно, latex-lab/NFSS-патч). Самое
% парсимонное решение — заранее объявить эти счётчики как \newcount, чтобы
% advance не падал. Делаем ДО \usepackage{hyperref}.
\makeatletter
\@ifundefined{c@v@normal}{\newcount\c@v@normal}{}%
\@ifundefined{c@v@bold}{\newcount\c@v@bold}{}%
\@ifundefined{c@v@italic}{\newcount\c@v@italic}{}%
\@ifundefined{c@v@bolditalic}{\newcount\c@v@bolditalic}{}%
\makeatother

% --- TL 2026 workaround для hyperref.sty:8303 ---------------------------------
% В hyperref 7.01p (TL 2026) строка 8303 вместо \providecommand* по ошибке
% использует \newcommand*{\footnoteautorefname}{\footnoteautorefname}. Команда
% к этому моменту уже определена (строка 2863 при загрузке английского языка),
% поэтому \newcommand* падает с «Command \footnoteautorefname already defined»,
% а каскад испорченных токенов ломает всё, что идёт после (в т.ч. \newcommand
% в пользовательской преамбуле и \newtheorem при \begin{document}).
%
% Обходим одной строкой: на время загрузки hyperref подменяем \newcommand на
% \providecommand. Семантика та же для корректных случаев, а багнутая строка
% 8303 становится безобидным no-op, потому что команда уже определена.
