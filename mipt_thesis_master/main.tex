% --- Окружение breakablealgorithm: не-float версия algorithm, которая
% --- разрешает разрыв страниц (стандартный приём, см. tex.stackexchange
% --- /a/100954). Сохраняет нумерацию через counter algorithm и заголовок
% --- через \caption. Применяется к длинным алгоритмам (SP-Broyden,
% --- SS-SR1), которые не помещаются на одной странице.
\makeatletter
\newenvironment{breakablealgorithm}
  {% \begin{breakablealgorithm}
   \begin{center}
     \refstepcounter{algorithm}%
     \hrule height.8pt depth0pt \kern2pt
     \renewcommand{\caption}[2][\relax]{%
       {\raggedright\textbf{\ALG@name~\thealgorithm} ##2\par}%
       \ifx\relax##1\relax
         \addcontentsline{loa}{algorithm}{\protect\numberline{\thealgorithm}##2}%
       \else
         \addcontentsline{loa}{algorithm}{\protect\numberline{\thealgorithm}##1}%
       \fi
       \kern2pt\hrule\kern2pt
     }
  }{%
     \kern2pt\hrule\relax
   \end{center}
  }
\makeatother

%--- Теоремные окружения ---------------------------------------------------
% ГОСТ-совместимый вариант: чёрный жирный serif-заголовок в основном
% семействе шрифтов (Latin Modern), без цвета и без sans-serif в теле
% научной работы. Дифференциация трёх классов утверждений — через
% штатные стили amsthm:
%   * miptmain (≡ plain)      — теоремы/леммы/следствия/утверждения/
%                                гипотезы: жирный заголовок на отдельной
%                                строке + курсивное тело;
%   * miptdef  (≡ definition) — определения и предположения: жирный
%                                заголовок на отдельной строке + прямое тело;
%   * miptrem  (≡ remark)     — замечания: курсивный заголовок обычного
%                                веса inline + прямое тело (намеренно
%                                «легче», без линий-разделителей).
% Заголовок на отдельной строке достигается передачей \newline в качестве
% «head/body sep» (7-й аргумент). Тонкие горизонтальные линии-разделители
% сверху и снизу добавляются ниже через \BeforeBeginEnvironment /
% \AfterEndEnvironment из etoolbox — для главных утверждений, определений
% и доказательств, но НЕ для замечаний.
% Сигнатура \newtheoremstyle:
%   {above}{below}{body font}{indent}{header font}{punct}{head/body sep}{custom}
\newtheoremstyle{miptmain}%
  {0.6\baselineskip}{0.4\baselineskip}{\itshape}{0pt}%
  {\large\bfseries}{.}{\newline}{}
\newtheoremstyle{miptdef}%
  {0.6\baselineskip}{0.4\baselineskip}{\normalfont}{0pt}%
  {\large\bfseries}{.}{\newline}{}
\newtheoremstyle{miptrem}%
  {0.6\baselineskip}{0.4\baselineskip}{\normalfont}{0pt}%
  {\itshape}{.}{0.5em}{}

\theoremstyle{miptmain}

% --- Без левого отступа: блоки теорем/лемм/доказательств идут по той же
% --- левой границе, что и основной текст. Дифференциация остаётся только
% --- по \newtheoremstyle (жирный заголовок \large на отдельной строке +
% --- курсив/прямой шрифт тела).

% --- Доказательства: жирный заголовок «Доказательство.» на отдельной
% --- строке и заметный QED -------------------------------------------------
% Стандартный amsthm печатает заголовок курсивом и в одну линию с телом;
% перепакуем proof-окружение с \bfseries и принудительным переводом строки
% после заголовка (\mbox{}\par\nobreak\noindent — стандартный приём,
% сохраняющий ignorespaces и не нарушающий trivlist). \blacksquare —
% заполненный квадрат, контрастнее дефолтного незаполненного.
\renewcommand{\proofname}{Доказательство}
\renewcommand{\qedsymbol}{\ensuremath{\blacksquare}}
\makeatletter
\renewenvironment{proof}[1][\proofname]{\par
  \pushQED{\qed}%
  \normalfont \topsep6\p@\@plus6\p@\relax
  \trivlist
  \item[\hskip\labelsep
        \large\bfseries
    #1\@addpunct{.}]\mbox{}\par\nobreak\noindent\ignorespaces
}{%
  \popQED\endtrivlist\@endpefalse
}
\makeatother
